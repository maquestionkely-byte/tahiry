  // ==================== SYNTHÈSE VOCALE AVEC VOIX GENRÉE ====================
  function trouverVoixFrancaise(homme) {
    const voix = window.speechSynthesis.getVoices();
    if (!voix || voix.length === 0) return null;
    
    if (homme) {
      let voixHomme = voix.find(v => 
        (v.lang === 'fr-FR' || v.lang === 'fr') && 
        (v.name.toLowerCase().includes('homme') || v.name.toLowerCase().includes('male') ||
         v.name.toLowerCase().includes('paul') || v.name.toLowerCase().includes('thomas'))
      );
      if (!voixHomme) voixHomme = voix.find(v => (v.lang === 'fr-FR' || v.lang === 'fr') && v.name.includes('Google'));
      if (!voixHomme) voixHomme = voix.find(v => v.lang === 'fr-FR' || v.lang === 'fr');
      return voixHomme;
    } else {
      let voixFemme = voix.find(v => 
        (v.lang === 'fr-FR' || v.lang === 'fr') && 
        (v.name.toLowerCase().includes('femme') || v.name.toLowerCase().includes('female') ||
         v.name.toLowerCase().includes('hortense') || v.name.toLowerCase().includes('julie'))
      );
      if (!voixFemme) voixFemme = voix.find(v => (v.lang === 'fr-FR' || v.lang === 'fr') && v.name.includes('Google'));
      if (!voixFemme) voixFemme = voix.find(v => v.lang === 'fr-FR' || v.lang === 'fr');
      return voixFemme;
    }
  }
  
  function parler(texte, prenom) {
    if (!window.speechSynthesis) return;
    
    const genre = getGenre(prenom);
    const textePropre = removeAllEmojis(texte);
    
    if (!textePropre.trim()) return;
    
    if (currentUtterance) window.speechSynthesis.cancel();
    
    const utterance = new SpeechSynthesisUtterance(textePropre);
    utterance.lang = "fr-FR";
    utterance.rate = 0.85;
    utterance.pitch = genre === 'garcon' ? 0.8 : 1.1;
    
    const voix = trouverVoixFrancaise(genre === 'garcon');
    if (voix) utterance.voice = voix;
    
    currentUtterance = utterance;
    utterance.onend = () => { currentUtterance = null; };
    window.speechSynthesis.speak(utterance);
  }
  
  // ==================== INDEXEDDB ====================
  function openDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, 2);
      request.onupgradeneeded = (e) => {
        const db = e.target.result;
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          db.createObjectStore(STORE_NAME);
        }
      };
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }
  
  async function saveToDB(content, version) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);
      store.put(content, FILE_KEY);
      store.put(version, VERSION_KEY);
      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
    });
  }
  
  async function loadFromDB() {
    const db = await openDB();
    return new Promise((resolve) => {
      const tx = db.transaction(STORE_NAME, 'readonly');
      const store = tx.objectStore(STORE_NAME);
      const reqContent = store.get(FILE_KEY);
      const reqVersion = store.get(VERSION_KEY);
      let content = null, version = null;
      let completed = 0;
      function check() {
        completed++;
        if (completed === 2) resolve({ content, version });
      }
      reqContent.onsuccess = () => { content = reqContent.result; check(); };
      reqVersion.onsuccess = () => { version = reqVersion.result; check(); };
      reqContent.onerror = () => check();
      reqVersion.onerror = () => check();
    });
  }

  async function getRemoteVersionDate() {
    try {
      const res = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
      if (!res.ok) return null;
      const text = await res.text();
      const blocks = text.split(/DERNIÈRE MISE À JOUR/);
      for (const block of blocks) {
        const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
        if (fileMatch && fileMatch[1].trim() === TARGET_FILE) {
          const dayMatch = block.match(/📅\s*([\d\/]+)/);
          const timeMatch = block.match(/⏰\s*([\d:]+)/);
          if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
        }
      }
      return null;
    } catch (e) { return null; }
  }

  function parseCompleteFile(rawText) {
    const lines = rawText.split(/\r?\n/);
    const rubriques = [];
    const rubriquesData = {};
    const rubriquesDescriptions = {};
    let currentRubrique = null;
    let i = 0;
    
    while (i < lines.length) {
      let line = lines[i];
      let trimmed = line.trim();
      
      const titreMatch = trimmed.match(/^[Tt]itre\s*:\s*(.+)$/);
      if (titreMatch) {
        currentRubrique = titreMatch[1].trim();
        if (!rubriques.includes(currentRubrique)) {
          rubriques.push(currentRubrique);
          rubriquesData[currentRubrique] = [];
          rubriquesDescriptions[currentRubrique] = "";
        }
        i++;
        if (i < lines.length) {
          let nextLine = lines[i].trim();
          const descMatch = nextLine.match(/^[Dd]escription\s*:\s*(.+)$/);
          if (descMatch) {
            rubriquesDescriptions[currentRubrique] = descMatch[1].trim();
            i++;
          }
        }
        continue;
      }
      
      if (currentRubrique && trimmed.length > 0) {
        const dialogueMatch = trimmed.match(/^([A-Za-zÀ-ÿ]+)\s*:\s*(.*)$/);
        if (dialogueMatch) {
          const nom = dialogueMatch[1].trim();
          let message = dialogueMatch[2];
          const genre = getGenre(nom);
          let traduction = "";
          let j = i + 1;
          let found = false;
          while (j < lines.length) {
            const next = lines[j];
            const nextTrim = next.trim();
            const tradMatch = nextTrim.match(/^[Tt]raduction\s*:\s*(.*)$/);
            if (tradMatch) {
              traduction = tradMatch[1];
              found = true;
              i = j;
              break;
            }
            message += "\n" + next;
            j++;
          }
          if (found) {
            let k = i + 1;
            while (k < lines.length) {
              const next = lines[k];
              const nextTrim = next.trim();
              if (nextTrim.match(/^[A-Za-zÀ-ÿ]+\s*:/)) break;
              if (nextTrim.match(/^[Tt]itre\s*:/)) break;
              k++;
            }
            i = k - 1;
          }
          rubriquesData[currentRubrique].push({
            nom: nom,
            message: message,
            traduction: traduction,
            genre: genre
          });
          i++;
          continue;
        }
      }
      i++;
    }
    
    const valid = rubriques.filter(r => rubriquesData[r] && rubriquesData[r].length > 0);
    return { rubriques: valid, rubriquesData, rubriquesDescriptions };
  }

  async function silentUpdate() {
    if (isUpdating || !navigator.onLine) return false;
    try {
      const remoteVer = await getRemoteVersionDate();
      if (!remoteVer) return false;
      if (currentVersionDate === remoteVer) return false;
      
      isUpdating = true;
      const newContent = await fetchRemoteTxt();
      if (!newContent || newContent.length < 100) throw new Error("Contenu invalide");
      
      await saveToDB(newContent, remoteVer);
      currentVersionDate = remoteVer;
      
      const parsed = parseCompleteFile(newContent);
      cachedRubriques = parsed.rubriques;
      cachedRubriquesData = parsed.rubriquesData;
      cachedDescriptions = parsed.rubriquesDescriptions;
      
      updateListOnly();
      
      isUpdating = false;
      showToast('✅ Dialogues mis à jour');
      return true;
    } catch (e) {
      console.error(e);
      isUpdating = false;
      return false;
    }
  }

  function fuzzyMatch(text, search) {
    if (!search.trim()) return true;
    const t = text.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    const s = search.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    return t.includes(s);
  }
  
  function filterRubriques() {
    if (!currentSearch.trim()) return cachedRubriques;
    const term = currentSearch.trim();
    if (/^\d+$/.test(term)) {
      const num = parseInt(term);
      const found = cachedRubriques.find((_, idx) => idx + 1 === num);
      return found ? [found] : [];
    }
    return cachedRubriques.filter(t => fuzzyMatch(t, term));
  }

      };

        currentVersionDate = cached.version;
        const parsed = parseCompleteFile(cached.content);
        cachedRubriques = parsed.rubriques;
        cachedRubriquesData = parsed.rubriquesData;
        cachedDescriptions = parsed.rubriquesDescriptions;
        
        if (cachedRubriques.length > 0) {
          renderList();
        } else {
          throw new Error("Fichier corrompu");
        }
      } else {
        if (!navigator.onLine) throw new Error("Pas de connexion internet");
        
        const fileContent = await fetchRemoteTxt();
        const remoteVersion = await getRemoteVersionDate();
        const finalVersion = remoteVersion || new Date().toLocaleString('fr-FR');
        
        await saveToDB(fileContent, finalVersion);
        currentVersionDate = finalVersion;
        
        const parsed = parseCompleteFile(fileContent);
        cachedRubriques = parsed.rubriques;
        cachedRubriquesData = parsed.rubriquesData;
        cachedDescriptions = parsed.rubriquesDescriptions;
        
        if (cachedRubriques.length === 0) {
          const fallback = getFallbackData();
          cachedRubriques = fallback.rubriques;
          cachedRubriquesData = fallback.rubriquesData;
          cachedDescriptions = fallback.rubriquesDescriptions;
        }
        
        renderList();
      }
      
      if (navigator.onLine) setTimeout(() => silentUpdate(), 2000);

      cachedRubriquesData = fallback.rubriquesData;
      cachedDescriptions = fallback.rubriquesDescriptions;
