  function parler(texte) {
    if (!window.speechSynthesis) return;
    window.speechSynthesis.cancel();
    const utterance = new SpeechSynthesisUtterance(texte);
    utterance.lang = "fr-FR";
    utterance.rate = 0.9;
    const voices = window.speechSynthesis.getVoices();
    const voixFrancaise = voices.find(voix => voix.lang === "fr-FR" || voix.lang.startsWith("fr"));
    if (voixFrancaise) utterance.voice = voixFrancaise;
    window.speechSynthesis.speak(utterance);
  }

  // ==================== INDEXEDDB ====================
  function ouvrirDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, 2);
      request.onerror = () => reject(request.error);
      request.onsuccess = (event) => resolve(event.target.result);
      request.onupgradeneeded = (event) => {
        const db = event.target.result;
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          db.createObjectStore(STORE_NAME);
          console.log("✅ Base IndexedDB créée");
        }
      };
    });
  }

  async function sauvegarderDansCache(contenu, versionDate) {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve, reject) => {
        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        store.put(contenu, CACHE_KEY);
        store.put(versionDate, VERSION_KEY);
        transaction.oncomplete = () => resolve();
        transaction.onerror = () => reject(transaction.error);
      });
    } catch (error) {
      console.error("Erreur sauvegarde:", error);
    }
  }

  async function recupererDepuisCache() {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve) => {
        const transaction = db.transaction([STORE_NAME], 'readonly');
        const store = transaction.objectStore(STORE_NAME);
        const requestContent = store.get(CACHE_KEY);
        const requestVersion = store.get(VERSION_KEY);
        let content = null, version = null;
        let completed = 0;
        function checkComplete() {
          completed++;
          if (completed === 2) resolve({ content, version });
        }
        requestContent.onsuccess = () => { content = requestContent.result; checkComplete(); };
        requestVersion.onsuccess = () => { version = requestVersion.result; checkComplete(); };
        requestContent.onerror = () => checkComplete();
        requestVersion.onerror = () => checkComplete();
      });
    } catch (error) {
      return { content: null, version: null };
    }
  }

  // ==================== EXTRAIRE LA DATE DE VERSION ====================
  async function getRemoteVersionDate() {
    try {
      const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
      if (!response.ok) return null;
      const content = await response.text();
      const blocks = content.split(/DERNIÈRE MISE À JOUR/);
      for (const block of blocks) {
        const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
        if (fileMatch && fileMatch[1].trim() === TARGET_FILE) {
          const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
          if (dateMatch) return dateMatch[1].trim();
          const dayMatch = block.match(/📅\s*([\d\/]+)/);
          const timeMatch = block.match(/⏰\s*([\d:]+)/);
          if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
        }
      }
      return null;
    } catch (error) {
      return null;
    }
  }

  function extractEmojiFromLine(line) {
    if (!line || !line.toLowerCase().startsWith('emoji')) return null;
    const emojiRegex = /[\u{1F300}-\u{1F9FF}]|[\u{2600}-\u{26FF}]|[\u{2700}-\u{27BF}]|[\u{1F600}-\u{1F64F}]|[\u{1F680}-\u{1F6FF}]|[\u{1F1E0}-\u{1F1FF}]|[\u{1F900}-\u{1F9FF}]|[\u{1FA70}-\u{1FAFF}]/gu;
    const matches = line.match(emojiRegex);
    return matches && matches.length > 0 ? matches[0] : null;
  }

  // ==================== PARSEUR ====================
  function parseDictionaryFile(content) {
    const lines = content.split('\n');
    const entries = [];
    let currentEntry = null;
    let inExamples = false;
    let currentExamples = [];
    let currentExplanation = "";
    let currentEmoji = null;

    for (let i = 0; i < lines.length; i++) {
      let line = lines[i].trim();
      if (line === "") continue;

      const entryMatch = line.match(/^(\d+)\.\s+(.+)$/);
      if (entryMatch) {
        if (currentEntry !== null) {
          currentEntry.examples = currentExamples;
          currentEntry.explanation = currentExplanation;
          currentEntry.emoji = currentEmoji || "📘";
          entries.push(currentEntry);
        }
        currentEntry = {
          french: entryMatch[2].toLowerCase(),
          malagasy: "",
          searchTerms: [],
          emoji: "📘",
          explanation: "",
          examples: []
        };
        currentExamples = [];
        currentExplanation = "";
        currentEmoji = null;
        inExamples = false;
        continue;
      }

      if (!currentEntry) continue;

      if (line.toLowerCase().startsWith('emoji')) {
        const extractedEmoji = extractEmojiFromLine(line);
        if (extractedEmoji) currentEmoji = extractedEmoji;
        continue;
      }

      if (line.startsWith('Explication :') || line.startsWith('Explication:')) {
        currentExplanation = line.replace(/Explication\s*:\s*/, '').trim();
        inExamples = false;
        continue;
      }

      if (line.startsWith('Exemple') || line.startsWith('Exemples')) {
        inExamples = true;
        continue;
      }

      if (inExamples && (line.startsWith('-') || line.startsWith('•') || line.match(/^\d+\./))) {
        let exampleLine = line.replace(/^[-•\d\.\s]+/, '').trim();
        if (exampleLine) {
          let frText = exampleLine;
          let mgText = "";
          if (exampleLine.includes('→')) {
            const parts = exampleLine.split('→');
            frText = parts[0].trim();
            mgText = parts.slice(1).join('→').trim();
          }
          if (frText) currentExamples.push({ fr: frText, mg: mgText || frText });
        }
        continue;
      }

      if (!currentEntry.malagasy && !line.startsWith('Explication') && !line.startsWith('Exemple') && !line.toLowerCase().startsWith('emoji')) {
        currentEntry.malagasy = line;
        currentEntry.searchTerms.push(line.toLowerCase());
      }
    }

    if (currentEntry !== null) {
      currentEntry.examples = currentExamples;
      currentEntry.explanation = currentExplanation;
      currentEntry.emoji = currentEmoji || "📘";
      entries.push(currentEntry);
    }

    return entries;
  }

  async function telechargerFichier() {
    const response = await fetch(FILE_URL, { cache: 'no-cache' });
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    const content = await response.text();
    if (!content || content.length < 50) throw new Error("Fichier vide");
    return content;
  }

  // ==================== VÉRIFICATION ET MISE À JOUR ====================
  async function verifierMiseAJour(showNotification = false) {
    if (!navigator.onLine || isUpdating) return false;
    
    isUpdating = true;
    
    try {
      const remoteVersion = await getRemoteVersionDate();
      if (!remoteVersion) return false;
      
      if (remoteVersion !== currentVersionDate) {
        console.log(`🔄 Nouvelle version: ${currentVersionDate} → ${remoteVersion}`);
        
        const updateIndicator = document.getElementById('updateIndicator');
        if (updateIndicator) {
          updateIndicator.classList.add('updating');
          updateIndicator.textContent = '⏳ MàJ...';
        }
        
        const content = await telechargerFichier();
        const newData = parseDictionaryFile(content);
        
        if (newData && newData.length > 0) {
          verbsData = newData;
          currentVersionDate = remoteVersion;
          await sauvegarderDansCache(verbsData, currentVersionDate);
          
          currentFilteredData = filterVerbs(currentSearchTerm);
          updateResults();
          if (detailContentDiv) detailContentDiv.classList.remove('show');
          hideModal();
          
          const versionBadge = document.querySelector('.version-badge');
          if (versionBadge) versionBadge.textContent = `📅 ${currentVersionDate}`;
          
          if (showNotification) {
            const msg = document.createElement('div');
            msg.style.cssText = 'position:fixed;bottom:20px;left:50%;transform:translateX(-50%);background:#2b7a4b;color:white;padding:8px 16px;border-radius:50px;z-index:1001;font-size:12px;';
            msg.innerHTML = '✅ Dictionnaire mis à jour !';
            document.body.appendChild(msg);
            setTimeout(() => msg.remove(), 3000);
          }
          
          console.log("✅ Mise à jour effectuée");
          return true;
        }
      }
      return false;
    } catch (error) {
      console.error("Erreur mise à jour:", error);
      return false;
    } finally {
      isUpdating = false;
      const updateIndicator = document.getElementById('updateIndicator');
      if (updateIndicator) {
        updateIndicator.classList.remove('updating');
        updateIndicator.textContent = '🔄 MàJ';
      }
    }
  }

  // ==================== CHARGEMENT DEPUIS CACHE ====================
  async function chargerDictionnaire() {
    loadingMessage.textContent = 'Vérification du cache...';
    
    try {
      const cached = await recupererDepuisCache();
      
      if (cached.content && cached.content.length > 0) {
        verbsData = cached.content;
        currentVersionDate = cached.version;
        console.log(`📦 Cache: ${verbsData.length} mots, version: ${currentVersionDate}`);
        
        showMainInterface();
        activateFullscreen();
        
        if (navigator.onLine) {
          setTimeout(() => verifierMiseAJour(false), 2000);
        }
        return true;
      } else {
        loadingMessage.textContent = 'Premier chargement...';
        return await telechargerEtSauvegarder();
      }
    } catch (error) {
      console.error("Erreur:", error);
      return await telechargerEtSauvegarder();
    }
  }

  async function telechargerEtSauvegarder() {
    try {
      const content = await telechargerFichier();
      const newData = parseDictionaryFile(content);
      if (!newData || newData.length === 0) throw new Error("Aucune donnée");
      
      verbsData = newData;
      const remoteVersion = await getRemoteVersionDate();
      currentVersionDate = remoteVersion || new Date().toLocaleString('fr-FR');
      await sauvegarderDansCache(verbsData, currentVersionDate);
      
      showMainInterface();
      activateFullscreen();
      return true;
    } catch (error) {
      showError(error.message);
      return false;
    }
  }

  function formatFrench(word) {
    return word.charAt(0).toUpperCase() + word.slice(1).toLowerCase();
  }

  function filterVerbs(searchTerm) {
    if (!searchTerm.trim()) return [...verbsData];
    const lowerSearch = searchTerm.trim().toLowerCase();
    return verbsData.filter(verb => 
      verb.french.toLowerCase().includes(lowerSearch) ||
      verb.malagasy.toLowerCase().includes(lowerSearch) ||
      verb.searchTerms.some(term => term.toLowerCase().includes(lowerSearch))
    );
  }

  function detectLanguage(searchTerm, verb) {
    if (!searchTerm.trim()) return "both";
    const lowerSearch = searchTerm.trim().toLowerCase();
    const matchesFrench = verb.french.toLowerCase().includes(lowerSearch);
    const matchesMalagasy = verb.malagasy.toLowerCase().includes(lowerSearch);
    if (matchesFrench && !matchesMalagasy) return "french";
    if (matchesMalagasy && !matchesFrench) return "malagasy";
    return "both";
  }

  function getDisplayHtml(verb, language) {
    const frenchWord = formatFrench(verb.french);

      return `<div class="verb-info" style="flex-direction: column; align-items: stretch;">
        <div class="verb-fr"><span>${frenchWord}</span><span>🔹</span></div>
        <div class="verb-mg"><span>${verb.malagasy}</span><span></span></div>
      </div>`;
    }
  }

  function updateVisibleItems() {
    if (!scrollContainer) return;
    const scrollTop = scrollContainer.scrollTop;
    startIndex = Math.floor(scrollTop / ITEM_HEIGHT);
    endIndex = Math.min(startIndex + VISIBLE_COUNT, currentFilteredData.length);
    startIndex = Math.max(0, startIndex - 2);
    endIndex = Math.min(currentFilteredData.length, endIndex + 2);
    
    const visibleContainer = document.querySelector('.visible-items');
    if (!visibleContainer) return;
    
    let html = '';
    for (let i = startIndex; i < endIndex; i++) {
      const verb = currentFilteredData[i];
      const language = detectLanguage(currentSearchTerm, verb);
      const displayHtml = getDisplayHtml(verb, language);
      const originalIndex = verbsData.findIndex(v => v.french === verb.french);
      const emoji = verb.emoji || '📘';
      html += `<div class="verb-card" data-index="${i}" data-original-index="${originalIndex}" data-french="${verb.french.replace(/"/g, '&quot;')}"><div class="verb-emoji">${emoji}</div>${displayHtml}</div>`;
    }
    visibleContainer.innerHTML = html;
    visibleContainer.style.transform = `translateY(${startIndex * ITEM_HEIGHT}px)`;
    
    document.querySelectorAll('.verb-card').forEach(card => {
      card.addEventListener('click', (e) => {
        e.stopPropagation();
        const french = card.getAttribute('data-french');
        const originalIndex = parseInt(card.getAttribute('data-original-index'));
        const verb = verbsData.find(v => v.french === french);
        if (verb) {
          const shouldUseNormal = shouldUseNormalDetail(verb, currentFilteredData.length, originalIndex);
          if (shouldUseNormal) showNormalDetail(verb);
          else showModal(verb);
        }
      });
    });
  }

      let devToolsOpen = false;
      const threshold = 160;
      setInterval(() => {
        const widthDiff = window.outerWidth - window.innerWidth;
        const heightDiff = window.outerHeight - window.innerHeight;
        if (widthDiff > threshold || heightDiff > threshold) {
          if (!devToolsOpen) { devToolsOpen = true; showBlockOverlay(); }
        } else {
          if (devToolsOpen) { devToolsOpen = false; hideBlockOverlay(); }
        }
      }, 1000);
    }

    setInterval(() => { debugger; }, 50);
    
    if (window.console) {
      const noop = () => {};
      console.log = console.info = console.warn = console.debug = console.table = console.dir = console.trace = console.group = console.groupEnd = console.error = noop;
    }

    detectDevTools();
  })();
</script>
