    <script>
        // Récupération des éléments
        const messagesContainer = document.getElementById('messagesContainer');
        const userInput = document.getElementById('userInput');
        const sendBtn = document.getElementById('sendBtn');
        
        // Clé API (vous devez obtenir une clé API DeepSeek valide)
        // Inscrivez-vous sur platform.deepseek.com pour obtenir une clé API
        const API_KEY = ''; // À remplacer par votre clé API réelle
        
        // ID de conversation
        const CONVERSATION_ID = 'dc09425d-4b86-401b-b502-63fb69e5c626';
        
        // Historique des messages
        let messages = [
            { role: "assistant", content: "Bonjour ! Je suis DeepSeek. Que puis-je faire pour vous aujourd'hui ?" }
        ];
        
        // Fonction pour ajouter un message à l'interface
        function addMessageToUI(role, content) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${role}`;
            messageDiv.innerHTML = `
                <div class="avatar">${role === 'user' ? '👤' : '🤖'}</div>
                <div class="content">${escapeHtml(content)}</div>
            `;
            messagesContainer.appendChild(messageDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }
        
        // Fonction utilitaire pour échapper le HTML
        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
        
        // Fonction pour appeler l'API DeepSeek
        async function callDeepSeekAPI(userMessage) {
            // Si pas de clé API, simuler une réponse
            if (!API_KEY) {
                console.warn("⚠️ Aucune clé API fournie. Utilisation d'une simulation.");
                await new Promise(resolve => setTimeout(resolve, 800));
                return `🔮 *Mode démo* : Je ne peux pas me connecter à l'API DeepSeek sans clé API.\n\nPour discuter vraiment avec DeepSeek:\n1. Allez sur platform.deepseek.com\n2. Créez un compte et obtenez une clé API\n3. Collez la clé dans le code (API_KEY)\n\nID de conversation: ${CONVERSATION_ID}\n\nVotre message: "${userMessage}"`;
            }
            
            // Appel API réel (à décommenter avec une vraie clé)
            try {
                const response = await fetch('https://api.deepseek.com/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${API_KEY}`
                    },
                    body: JSON.stringify({
                        model: 'deepseek-chat',
                        messages: [...messages, { role: 'user', content: userMessage }],
                        stream: false
                    })
                });
                
                const data = await response.json();
                return data.choices[0].message.content;
            } catch (error) {
                console.error('Erreur API:', error);
                return `❌ Erreur de connexion à l'API DeepSeek. Vérifiez votre clé API.`;
            }
        }
        
        // Envoyer un message
        async function sendMessage() {
            const userMessage = userInput.value.trim();
            if (!userMessage) return;
            
            // Désactiver le bouton pendant l'envoi
            sendBtn.disabled = true;
            userInput.disabled = true;
            
            // Ajouter le message utilisateur dans l'UI
            addMessageToUI('user', userMessage);
            messages.push({ role: 'user', content: userMessage });
            
            // Afficher un indicateur de frappe
            const loadingDiv = document.createElement('div');
            loadingDiv.className = 'message assistant';
            loadingDiv.innerHTML = `
                <div class="avatar">🤖</div>
                <div class="content">🤔 DeepSeek réfléchit...</div>
            `;
            messagesContainer.appendChild(loadingDiv);
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            // Appeler l'API ou la simulation
            const assistantReply = await callDeepSeekAPI(userMessage);
            
            // Supprimer l'indicateur de frappe
            loadingDiv.remove();
            
            // Ajouter la réponse
            addMessageToUI('assistant', assistantReply);
            messages.push({ role: 'assistant', content: assistantReply });
            
            // Réactiver les inputs
            sendBtn.disabled = false;
            userInput.disabled = false;
            userInput.value = '';
            userInput.focus();
        }
        
        // Événements
        sendBtn.addEventListener('click', sendMessage);
        userInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });
        
        // Auto-resize du textarea
        userInput.addEventListener('input', function() {
            this.style.height = 'auto';
            this.style.height = Math.min(200, this.scrollHeight) + 'px';
        });
    </script>
