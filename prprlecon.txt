    // ======================= FONCTIONS API GITHUB =======================
    async function githubRequest(method, path, token, body = null) {
        const url = `https://api.github.com/repos/${REPO}/contents/${encodeURIComponent(path)}`;
        const headers = {
            'Authorization': `token ${token}`,
            'Accept': 'application/vnd.github.v3+json'
        };
        const options = { method, headers };
        if (body) {
            options.body = JSON.stringify(body);
            headers['Content-Type'] = 'application/json';
        }
        const response = await fetch(url, options);
        const data = await response.json();
        return { ok: response.ok, status: response.status, data };
    }
    
    async function getSha(path, token) {
        const { ok, data } = await githubRequest('GET', path, token);
        if (ok && data.sha) return data.sha;
        return null;
    }
    
    async function readFileContent(path, token) {
        const { ok, data } = await githubRequest('GET', path, token);
        if (!ok) return null;
        return decodeURIComponent(escape(atob(data.content)));
    }
    
    async function uploadToGithub(path, content, token, sha = null) {
        const body = {
            message: `Mise à jour de ${path}`,
            content: btoa(unescape(encodeURIComponent(content))),
            branch: BRANCH
        };
        if (sha) body.sha = sha;
        const { ok, data } = await githubRequest('PUT', path, token, body);
        if (!ok) return { success: false, error: data.message };
        return { success: true, data };
    }
    
    function generateDateLine(fileName) {
        const now = new Date();
        const day = now.getDate().toString().padStart(2, '0');
        const month = (now.getMonth() + 1).toString().padStart(2, '0');
        const year = now.getFullYear();
        const hours = now.getHours().toString().padStart(2, '0');
        const minutes = now.getMinutes().toString().padStart(2, '0');
        const seconds = now.getSeconds().toString().padStart(2, '0');
        
        const dateStr = `${day}/${month}/${year}`;
        const timeStr = `${hours}:${minutes}:${seconds}`;
        
        return `DERNIÈRE MISE À JOUR
📅 ${dateStr}
⏰ ${timeStr}
📆 ${dateStr} ${timeStr}
📁 Fichier: ${fileName}`;
    }
    
    async function updateDateFile(fileName, token) {
        try {
            let currentContent = await readFileContent(DATE_FILE_PATH, token);
            let newContent = "";
            let found = false;
            
            if (currentContent) {
                const blocks = currentContent.split(/(?=DERNIÈRE MISE À JOUR)/);
                for (let block of blocks) {
                    if (block.includes(`📁 Fichier: ${fileName}`)) {
                        newContent += generateDateLine(fileName) + "\n\n";
                        found = true;
                    } else {
                        newContent += block;
                        if (!block.endsWith("\n\n") && !block.endsWith("\n")) {
                            newContent += "\n";
                        }
                    }
                }
            }
            
            if (!found) {
                if (newContent && !newContent.endsWith("\n\n")) newContent += "\n";
                newContent += generateDateLine(fileName) + "\n\n";
            }
            
            newContent = newContent.replace(/\n{3,}/g, "\n\n").trim() + "\n";
            
            const existingSha = await getSha(DATE_FILE_PATH, token);
            const result = await uploadToGithub(DATE_FILE_PATH, newContent, token, existingSha);
            return result.success;
        } catch (error) {
            console.error(`Erreur: ${error.message}`);
            return false;
        }
    }

    function getCodeLecon(titre) {
        let nettoye = titre.normalize("NFD").replace(/[\u0300-\u036f]/g, "");
        nettoye = nettoye.replace(/[^a-zA-Z0-9\s]/g, "");
        let mots = nettoye.toLowerCase().split(/\s+/);
        
        if (mots.length >= 2) {
            return mots[0].substring(0, 2) + mots[1].substring(0, 2);
        } else if (mots.length === 1) {
            return mots[0].substring(0, 4);
        }
        return "xxxx";
    }
    
    // ======================= CHARGEMENT DEPUIS lebe.txt =======================
    async function chargerLeconsDepuisLebe() {
        try {
            const response = await fetch(LEBO_URL);
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const htmlContent = await response.text();
            
            const extractArray = (varName) => {
                const regex = new RegExp(`const ${varName} = \\[([\\s\\S]*?)\\];`, 'i');
                const match = htmlContent.match(regex);
                if (match) {
                    const items = [];
                    const itemRegex = /{\s*id:\s*(\d+),\s*titre:\s*"([^"]+)",\s*description:\s*"([^"]*)",\s*defaut:\s*"([^"]*)"\s*}/g;
                    let itemMatch;
                    while ((itemMatch = itemRegex.exec(match[1])) !== null) {
                        const titre = itemMatch[2];
                        const code = getCodeLecon(titre);
                        items.push({
                            id: parseInt(itemMatch[1]),
                            titre: titre,
                            description: itemMatch[3],
                            code: code,
                            fichier: code + 'lecon.txt'
                        });
                    }
                    return items;
                }
                return [];
            };
            
            leconsData.debutant = extractArray('leconsDebutant');
            leconsData.intermediaire = extractArray('leconsIntermediaire');
            leconsData.avance = extractArray('leconsAvance');

                afficherTableauSpecial(cat.nom.toLowerCase() + 'Table', cat.lecons);
            });
            
            document.getElementById('loadingIndicator').style.display = 'none';
        } catch (error) {
            document.getElementById('loadingIndicator').innerHTML = '❌ Erreur de chargement';
        }
    }

    async function ajouterTousQCM(code) {
        if (!GITHUB_TOKEN) { alert('❌ Entrez votre token GitHub'); return; }
        
        const btn = document.getElementById('btnAddAll');
        btn.textContent = '⏳ Envoi en cours...';
        btn.disabled = true;
        
        let successCount = 0;
        let errorCount = 0;
        
        for (let level = 1; level <= 5; level++) {
            const fileName = `${code}qcm${level}.txt`;
            const content = tempContents[level];
            
            if (!content) {
                errorCount++;
                continue;
            }
            
            try {
                const existingSha = await getSha(fileName, GITHUB_TOKEN);
                const result = await uploadToGithub(fileName, content, GITHUB_TOKEN, existingSha);
                
                if (result.success) {
                    await updateDateFile(fileName, GITHUB_TOKEN);
                    successCount++;
                } else {
                    errorCount++;
                }
            } catch (error) {
                errorCount++;
            }
        }
        
        btn.textContent = `✅ ${successCount}/5 envoyés (${errorCount} erreurs)`;
        setTimeout(() => {
            btn.textContent = '➕ Ajouter les 5 niveaux QCM';
            btn.disabled = false;
        }, 3000);
        
        showToast(`QCM: ${successCount} fichiers enregistrés`);
        if (successCount === 5) setTimeout(() => fermerModal(), 1500);
    }
    
    function setupUpload(modal, code) {
        const uploadArea = modal.querySelector('#uploadArea');
        const fileInput = modal.querySelector('#fileInput');
        const preview = modal.querySelector('#preview');
        
        uploadArea.onclick = () => fileInput.click();
        uploadArea.ondragover = (e) => { e.preventDefault(); uploadArea.style.borderColor = '#ffb347'; };
        uploadArea.ondragleave = () => { uploadArea.style.borderColor = '#cbd5e1'; };
        uploadArea.ondrop = (e) => {
            e.preventDefault();
            uploadArea.style.borderColor = '#cbd5e1';
            const file = e.dataTransfer.files[0];
            if (file) handleFileQCM(file, preview, selectedLevel);
        };
        fileInput.onchange = (e) => {
            if (e.target.files[0]) handleFileQCM(e.target.files[0], preview, selectedLevel);
        };
        
        const sendBtn = modal.querySelector('#sendBtn');
        sendBtn.onclick = () => uploadFichierQCM(sendBtn, code + 'qcm' + selectedLevel + '.txt', selectedLevel);
    }
    
    function handleFileQCM(file, preview, level) {
        const reader = new FileReader();
        reader.onload = (e) => {
            tempContents[level] = e.target.result;
            preview.style.display = 'block';
            preview.innerHTML = `<strong>Aperçu niveau ${level}:</strong><br>${tempContents[level].substring(0, 200)}${tempContents[level].length > 200 ? '...' : ''}`;
        };
        reader.readAsText(file);
    }
    
    async function uploadFichierQCM(btn, nomFichier, level) {
        if (!tempContents[level]) { alert(`❌ Veuillez sélectionner un fichier pour le niveau ${level}`); return; }
        if (!GITHUB_TOKEN) { alert('❌ Entrez votre token GitHub'); return; }
        
        btn.classList.add('loading');
        btn.disabled = true;
        
        try {
            const existingSha = await getSha(nomFichier, GITHUB_TOKEN);
            const result = await uploadToGithub(nomFichier, tempContents[level], GITHUB_TOKEN, existingSha);
            
            if (result.success) {
                btn.classList.remove('loading');
                btn.textContent = '✅ Effectué !';
                btn.classList.add('success');
                
                await updateDateFile(nomFichier, GITHUB_TOKEN);
                
                showToast(`✅ ${nomFichier} enregistré !`);
                setTimeout(() => fermerModal(), 1500);
            } else {
                throw new Error(result.error);
            }
        } catch (error) {
            btn.classList.remove('loading');
            btn.textContent = '❌ Erreur';
            btn.style.background = '#e74c3c';
            btn.disabled = false;
            setTimeout(() => {
                btn.textContent = '📤 Envoyer ce niveau';
                btn.style.background = '#ffb347';
            }, 2000);
            showToast(`❌ Erreur: ${error.message}`);
        }
    }
