<script>
  const DB_NAME = 'lebea_storage';
  const STORE_NAME = 'lebe_content';
  const DB_VERSION = 18;

  async function debugDB() {
    const resultDiv = document.getElementById('result');
    
    try {
      const request = indexedDB.open(DB_NAME, DB_VERSION);
      
      request.onerror = () => {
        resultDiv.innerHTML = `<p class="error">❌ Erreur ouverture DB: ${request.error?.message}</p>`;
      };
      
      request.onsuccess = (e) => {
        const db = e.target.result;
        const transaction = db.transaction([STORE_NAME], 'readonly');
        const store = transaction.objectStore(STORE_NAME);
        
        // Lire toutes les clés
        const getAllKeys = store.getAllKeys();
        getAllKeys.onsuccess = () => {
          const keys = getAllKeys.result;
          resultDiv.innerHTML = `<p class="success">✅ Base ouverte avec succès</p>`;
          resultDiv.innerHTML += `<p><strong>📋 Clés trouvées dans la DB :</strong> ${keys.length > 0 ? keys.join(', ') : 'AUCUNE'}</p>`;
          
          if (keys.length === 0) {
            resultDiv.innerHTML += `<p class="error">⚠️ La base de données est VIDE !</p>`;
            resultDiv.innerHTML += `<p>Le HTML principal n'a pas encore stocké le contenu.</p>`;
            resultDiv.innerHTML += `<p>👉 Ouvrez d'abord le HTML principal pour qu'il télécharge et stocke lebe.txt</p>`;
            return;
          }
          
          // Lire chaque clé
          keys.forEach(async (key) => {
            const getValue = store.get(key);
            getValue.onsuccess = () => {
              const value = getValue.result;
              const taille = value ? value.length : 0;
              resultDiv.innerHTML += `<p><strong>🔑 Clé "${key}"</strong> : ${taille > 0 ? `${taille} caractères ✅` : 'VIDE ❌'}</p>`;
              if (key === 'htmlContent' && value) {
                resultDiv.innerHTML += `<p class="success">✅ La clé "htmlContent" existe et contient ${taille} caractères</p>`;
                resultDiv.innerHTML += `<details><summary>Aperçu (100 premiers caractères)</summary><pre>${value.substring(0, 500)}...</pre></details>`;
              }
            };
          });
        };
        
        getAllKeys.onerror = () => {
          resultDiv.innerHTML += `<p class="error">❌ Impossible de lire les clés</p>`;
        };
      };
      
    } catch (err) {
      resultDiv.innerHTML = `<p class="error">❌ Erreur: ${err.message}</p>`;
    }
  }
  
  debugDB();
</script>
