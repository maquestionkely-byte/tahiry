<script>
  const DB_NAME = 'lebea_storage';
  const STORE_NAME = 'lebe_content';
  const DB_VERSION = 18;
  const WORKER_URL = 'https://falytokoa.ahofaly62.workers.dev';

  let typesExercicesCache = null;
  let codeLeconCache = null;

  function ouvrirDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION);
      request.onerror = () => reject(request.error);
      request.onsuccess = (e) => resolve(e.target.result);
      request.onupgradeneeded = (e) => {
        if (!e.target.result.objectStoreNames.contains(STORE_NAME)) {
          e.target.result.createObjectStore(STORE_NAME);
        }
      };
    });
  }

  async function lireFichierDB(nomFichier) {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve) => {
        const tx = db.transaction([STORE_NAME], 'readonly');
        const store = tx.objectStore(STORE_NAME);
        const request = store.get(nomFichier);
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => resolve(null);
      });
    } catch { return null; }
  }

  async function lireLebeTxt() {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve) => {
        const tx = db.transaction([STORE_NAME], 'readonly');
        const store = tx.objectStore(STORE_NAME);
        const request = store.get('htmlContent');
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => resolve(null);
      });
    } catch { return null; }
  }

  function getCodeLecon(titre) {
    let nettoye = titre.normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    nettoye = nettoye.replace(/[^a-zA-Z0-9\s]/g, "");
    let mots = nettoye.toLowerCase().split(/\s+/);
    if (mots.length >= 2) {
      return mots[0].substring(0, 2) + mots[1].substring(0, 2);
    } else if (mots.length === 1) {
      return mots[0].substring(0, 4);
    }
    return "xxxx";
  }

  function extraireTypesExercices(contenuLebe) {
    const types = [];
    const parser = new DOMParser();
    const doc = parser.parseFromString(contenuLebe, 'text/html');
    
    const exoCards = doc.querySelectorAll('.exo-card');
    exoCards.forEach(card => {
      const titre = card.querySelector('div:first-child')?.textContent || '';
      const desc = card.querySelector('div:last-child')?.textContent || '';
      let code = '';
      const matchCode = (desc + titre).match(/\b(qcm|mal|vra|tex|ass)\b/i);
      if (matchCode) code = matchCode[1].toLowerCase();
      
      if (titre) {
        types.push({
          code: code,
          nom: titre.replace(/[📝]/g, '').trim(),
          description: desc
        });
      }
    });
    
    return types;
  }

  async function collecterExercicesSilencieusement() {
    const loadingDiv = document.getElementById('loading');
    loadingDiv.style.display = 'block';
    
    try {
      const contenuLebe = await lireLebeTxt();
      if (!contenuLebe) {
        loadingDiv.style.display = 'none';
        return false;
      }
      
      typesExercicesCache = extraireTypesExercices(contenuLebe);
      codeLeconCache = getCodeLecon("Types de phrases");
      
      if (typesExercicesCache.length === 0) {
        loadingDiv.style.display = 'none';
        return false;
      }
      
      for (let exo of typesExercicesCache) {
        const nomFichier = `${codeLeconCache}${exo.code}.txt`;
        const contenu = await lireFichierDB(nomFichier);
        exo.disponible = !!contenu;
        if (!contenu) {
          try {
            const url = `${WORKER_URL}/${nomFichier}`;
            const response = await fetch(url);
            if (response.ok) {
              const nouveauContenu = await response.text();
              const db = await ouvrirDB();
              const tx = db.transaction([STORE_NAME], 'readwrite');
              tx.objectStore(STORE_NAME).put(nouveauContenu, nomFichier);
              tx.oncomplete = () => { exo.disponible = true; };
              tx.commit();
            }
          } catch(e) {}
        }
      }
      
      loadingDiv.style.display = 'none';
      return true;
      
    } catch (error) {
      loadingDiv.style.display = 'none';
      return false;
    }
  }

  // Modal d'exercices avec son propre bouton Retour
  function afficherModalExercices() {
    if (!typesExercicesCache || typesExercicesCache.length === 0) {
      document.getElementById('errorMsg').style.display = 'block';
      return;
    }
    
    const modal = document.createElement('div');
    modal.style.cssText = `
      position: fixed; top: 0; left: 0; width: 100%; height: 100%;
      background: rgba(0,0,0,0.5); backdrop-filter: blur(4px);
      z-index: 20000; display: flex; align-items: center; justify-content: center;
    `;
    
    const modalContent = document.createElement('div');
    modalContent.style.cssText = `
      background: white; border-radius: 32px; max-width: 500px;
      width: 90%; max-height: 80vh; overflow-y: auto;
      box-shadow: 0 20px 40px rgba(0,0,0,0.3);
    `;
    
    modalContent.innerHTML = `
      <div style="background: #1f5442; color: white; padding: 16px 20px; border-radius: 32px 32px 0 0; display: flex; justify-content: space-between; align-items: center;">
        <h3 style="margin:0;">✏️ Exercices disponibles</h3>
        <button id="closeModalBtn" style="background: #ef4444; color: white; border: none; width: 32px; height: 32px; border-radius: 50%; font-size: 1.2rem; cursor: pointer;">✕</button>
      </div>
      <div style="padding: 20px;" id="modalExercicesList"></div>
      <div style="padding: 0 20px 20px 20px; text-align: center;">
        <button id="retourModalBtn" style="background: #0055a4; color: white; border: none; padding: 10px 24px; border-radius: 60px; font-size: 0.9rem; font-weight: bold; cursor: pointer;">↩️ Retour</button>
      </div>
    `;
    
    modal.appendChild(modalContent);
    document.body.appendChild(modal);
    
    const listContainer = modal.querySelector('#modalExercicesList');
    
    for (const exo of typesExercicesCache) {
      const card = document.createElement('div');
      card.style.cssText = `
        background: ${exo.disponible ? '#f8fafc' : '#e2e8f0'};
        border-radius: 20px; padding: 16px; margin-bottom: 12px;
        cursor: ${exo.disponible ? 'pointer' : 'not-allowed'};
        border-left: 5px solid ${exo.disponible ? '#2c7a5a' : '#94a3b8'};
        transition: all 0.2s; opacity: ${exo.disponible ? 1 : 0.6};
      `;
      card.innerHTML = `<h4 style="color:#1f5442;">📝 ${exo.nom}</h4><p style="font-size:0.85rem; color:#5a6e7c;">${exo.description}</p>`;
      
      if (exo.disponible) {
        card.onclick = async () => {
          const nomFichier = `${codeLeconCache}${exo.code}.txt`;
          const contenu = await lireFichierDB(nomFichier);
          if (contenu) {
            const blob = new Blob([contenu], { type: 'text/html' });
            const url = URL.createObjectURL(blob);
            window.open(url, '_blank');
          }
          modal.remove();
        };
        card.onmouseover = () => { card.style.background = '#fef3e2'; card.style.transform = 'translateX(5px)'; };
        card.onmouseout = () => { card.style.background = '#f8fafc'; card.style.transform = 'translateX(0)'; };
      }
      
      listContainer.appendChild(card);
    }
    
    // Boutons du modal
    modal.querySelector('#closeModalBtn').onclick = () => modal.remove();
    modal.querySelector('#retourModalBtn').onclick = () => modal.remove();
    modal.onclick = (e) => { if (e.target === modal) modal.remove(); };
  }

  // Lancement
  (async function init() {
    await collecterExercicesSilencieusement();
  })();

  // Bouton exercice
  document.getElementById('btnExercice').onclick = () => {
    if (typesExercicesCache && typesExercicesCache.length > 0) {
      afficherModalExercices();
    } else {
      document.getElementById('errorMsg').style.display = 'block';
    }
  };

  // Bouton retour principal : ferme l'onglet
  document.getElementById('btnRetour').onclick = () => {
    window.close();
  };
</script>
