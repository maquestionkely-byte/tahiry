<script>
  const DB_NAME = 'lebea_storage';
  const STORE_NAME = 'lebe_content';
  const DB_VERSION = 18;

  function ouvrirDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION);
      request.onerror = () => reject(request.error);
      request.onsuccess = (e) => resolve(e.target.result);
      request.onupgradeneeded = (e) => {
        if (!e.target.result.objectStoreNames.contains(STORE_NAME)) {
          e.target.result.createObjectStore(STORE_NAME);
        }
      };
    });
  }

  async function lireLebeTxt() {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve) => {
        const tx = db.transaction([STORE_NAME], 'readonly');
        const store = tx.objectStore(STORE_NAME);
        const request = store.get('htmlContent');
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => resolve(null);
      });
    } catch { return null; }
  }

  document.getElementById('btnExercice').onclick = async () => {
    const errorDiv = document.getElementById('errorMsg');
    errorDiv.style.display = 'none';
    
    const contenu = await lireLebeTxt();
    
    if (contenu) {
      // Ajouter un script dans le contenu pour rediriger vers la leçon "Types de phrases"
      const modifiedContent = contenu.replace('</body>', `
        <script>
          // Attendre que la page soit chargée
          window.addEventListener('load', function() {
            // Chercher la leçon "Types de phrases" dans la liste Débutant
            setTimeout(function() {
              // Aller à l'écran Débutant
              if (typeof showScreen === 'function') {
                showScreen('debutant');
              }
              // Attendre que les leçons soient chargées
              setTimeout(function() {
                // Chercher et cliquer sur la barre d'exercice de "Types de phrases"
                const toutesLesBarres = document.querySelectorAll('.barre-exercice');
                for (let barre of toutesLesBarres) {
                  if (barre.getAttribute('data-lecon-titre') === 'Types de phrases') {
                    barre.click();
                    break;
                  }
                }
              }, 500);
            }, 500);
          });
        <\/script>
      </body>`);
      
      const blob = new Blob([modifiedContent], { type: 'text/html' });
      const url = URL.createObjectURL(blob);
      window.open(url, '_blank');
    } else {
      errorDiv.style.display = 'block';
    }
  };
</script>
