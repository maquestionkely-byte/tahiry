<script>
  const DB_NAME = 'lebea_storage';
  const STORE_NAME = 'lebe_content';
  const DB_VERSION = 18;

  function ouvrirDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION);
      request.onerror = () => reject(request.error);
      request.onsuccess = (e) => resolve(e.target.result);
      request.onupgradeneeded = (e) => {
        if (!e.target.result.objectStoreNames.contains(STORE_NAME)) {
          e.target.result.createObjectStore(STORE_NAME);
        }
      };
    });
  }

  async function lireFichierDB(nomFichier) {
    try {
      const db = await ouvrirDB();
      return new Promise((resolve) => {
        const tx = db.transaction([STORE_NAME], 'readonly');
        const store = tx.objectStore(STORE_NAME);
        const request = store.get(nomFichier);
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => resolve(null);
      });
    } catch { return null; }
  }

  function getCodeLecon(titre) {
    let nettoye = titre.normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    nettoye = nettoye.replace(/[^a-zA-Z0-9\s]/g, "");
    let mots = nettoye.toLowerCase().split(/\s+/);
    if (mots.length >= 2) {
      return mots[0].substring(0, 2) + mots[1].substring(0, 2);
    } else if (mots.length === 1) {
      return mots[0].substring(0, 4);
    }
    return "typh";
  }

  const TITRE_LECON = "Types de phrases";
  const CODE_LECON = getCodeLecon(TITRE_LECON);

  function extraireTypesExercices(contenuLebe) {
    const types = [];
    const parser = new DOMParser();
    const doc = parser.parseFromString(contenuLebe, 'text/html');
    
    const exoCards = doc.querySelectorAll('.exo-card');
    exoCards.forEach(card => {
      const titre = card.querySelector('h4')?.textContent || '';
      const desc = card.querySelector('p')?.textContent || '';
      let code = '';
      const matchCode = (desc + titre).match(/\b(qcm|mal|vra|tex|ass)\b/i);
      if (matchCode) code = matchCode[1].toLowerCase();
      
      if (titre) {
        types.push({
          code: code,
          nom: titre.replace(/[📝]/g, '').trim(),
          description: desc
        });
      }
    });
    
    return types;
  }

  async function afficherModalExercices() {
    const contenuLebe = await lireFichierDB('htmlContent');
    
    if (!contenuLebe) {
      alert('Aucune donnée trouvée. Veuillez ouvrir l\'application principale d\'abord.');
      return;
    }
    
    const typesExercices = extraireTypesExercices(contenuLebe);
    
    if (typesExercices.length === 0) {
      alert('Aucun exercice trouvé dans le cours.');
      return;
    }
    
    const modal = document.createElement('div');
    modal.className = 'modal-exercices';
    modal.innerHTML = `
      <div class="modal-exercices-content">
        <div class="modal-header">
          <h3>✏️ Exercices disponibles</h3>
          <button class="close-modal">✕</button>
        </div>
        <div class="modal-body" id="modalExercicesList"></div>
      </div>
    `;
    document.body.appendChild(modal);
    
    const listContainer = modal.querySelector('#modalExercicesList');
    listContainer.innerHTML = '';
    
    for (const exo of typesExercices) {
      const nomFichier = exo.code ? `${CODE_LECON}${exo.code}.txt` : '';
      const card = document.createElement('div');
      card.className = 'exo-card';
      card.innerHTML = `<h4>📝 ${exo.nom}</h4><p>${exo.description || 'Cliquez pour commencer'}</p>`;
      
      card.onclick = async () => {
        let contenuExo = null;
        if (nomFichier) contenuExo = await lireFichierDB(nomFichier);
        
        if (contenuExo) {
          const blob = new Blob([contenuExo], { type: 'text/html' });
          const url = URL.createObjectURL(blob);
          window.open(url, '_blank');
        } else {
          alert(`Exercice non disponible: ${exo.nom}`);
        }
      };
      listContainer.appendChild(card);
    }
    
    modal.querySelector('.close-modal').onclick = () => modal.remove();
    modal.onclick = (e) => { if (e.target === modal) modal.remove(); };
  }

  document.getElementById('btnExercice').onclick = afficherModalExercices;
</script>
