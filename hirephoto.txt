<script>
    // ============================================================
    // 🔥 CONFIGURATIONS

    // CONFIG 1 : IMAGA (spécial)
    const IMAGA = {

    const RESAKASARY = {

    };

    // ============================================================
    // 📋 LISTE DES GALERIES - NOMS FICHIERS FLEXIBLES
    // ============================================================
    const GALLERIES = [
        // -------- IMAGA (spécial) --------
        {

            prefix: 'imaga',
            title: '📖 Connaissance {numero}',

            screenSub: 'savoir · partage · découverte'
        },

        // -------- RESAKASARY --------
        {

        },

        // 👇 FICHIER SANS "numero" au début - mylife.txt

        },

        // 👇 FICHIER SANS "numero" au début - voyages.txt
        {

            screenSub: 'exploration · aventures · horizons'
        },

        // 👇 AUTRE EXEMPLE - nature.txt
        {
            id: 'nature',
            label: '🌿 Nature & Paysages',
            verification: 'nature.txt',
            prefix: 'nature',
            title: '🌿 Paysage {numero}',
            screenTitle: '🌿 Nature & Paysages',
            screenSub: 'calme · sérénité · beauté'
        }
,
    {
        id: 'oita',
        label: '💪 😍 lover',
        verification: 'okay.txt',
        prefix: 'ot',
        title: 'oui',
        screenTitle: '💪 😍 lover',
        screenSub: 'yes'
    }
,
    {
        id: 'mety',
        label: '😍 loove,be',
        verification: 'metyka.txt',
        prefix: 'photore',
        title: 'mety tsara',
        screenTitle: '😍 loove,be',
        screenSub: 'vu de vu'
    }
//ataovy
    ];

    // ============================================================
    // TRAITEMENT AUTOMATIQUE
    // ============================================================
    GALLERIES.forEach(g => {
        // Si verification est présent, on construit l'URL avec RESAKASARY
        if (g.verification) {
            g.manifest = `https://resakasary.ahofaly62.workers.dev/${g.verification}`;
            g.worker = RESAKASARY.worker;
        }
        // Si ni manifest ni verification, on met RESAKASARY par défaut
        if (!g.manifest && !g.verification) {
            g.manifest = `https://resakasary.ahofaly62.workers.dev/numerosary.txt`;
            g.worker = RESAKASARY.worker;
        }
    });

    // ============================================================
    // LE RESTE EST AUTOMATIQUE
    // ============================================================

    let currentGallery = null;
    let currentPhotoList = [];
    let currentIndex = 0;

    const selectionScreen = document.getElementById('selectionScreen');
    const buttonsContainer = document.getElementById('buttonsContainer');
    const galleryApp = document.getElementById('galleryApp');
    const galleryTitle = document.getElementById('galleryTitle');
    const gallerySub = document.getElementById('gallerySub');
    const photoCounterSpan = document.getElementById('photoCounter');
    const galleryContainer = document.getElementById('galleryGrid');
    const lightboxDiv = document.getElementById('lightbox');
    const lbImage = document.getElementById('lbImage');
    const closeLb = document.getElementById('closeLb');
    const prevLb = document.getElementById('prevLb');
    const nextLb = document.getElementById('nextLb');
    const lbCounter = document.getElementById('lbCounter');
    const btnBack = document.getElementById('btnBack');
    const heroHeader = document.getElementById('heroHeader');

    // ---------- GÉNÉRER LES BOUTONS ----------
    GALLERIES.forEach(gallery => {
        const btn = document.createElement('button');
        btn.className = 'btn-gallery';
        btn.textContent = gallery.label;
        btn.dataset.id = gallery.id;
        btn.addEventListener('click', () => loadGallery(gallery.id));
        buttonsContainer.appendChild(btn);
    });

    // ---------- RETOUR (en haut à gauche) ----------
    btnBack.addEventListener('click', () => {
        galleryApp.classList.add('hidden');
        selectionScreen.style.display = 'flex';
        selectionScreen.style.opacity = '1';
    });

    // ---------- HEADER DISPARITION AU SCROLL ----------
    let lastScrollY = 0;
    function handleHeaderScroll() {
        if (!heroHeader) return;
        if (window.scrollY > lastScrollY && window.scrollY > 50) {
            heroHeader.classList.add('hide-header');
        } else if (window.scrollY < lastScrollY || window.scrollY <= 30) {
            heroHeader.classList.remove('hide-header');
        }
        lastScrollY = window.scrollY;
    }
    window.addEventListener('scroll', handleHeaderScroll, { passive: true });

    // ---------- INDEXEDDB ----------
    async function openDB(galleryId) {
        return new Promise((resolve, reject) => {
            const req = indexedDB.open(`Gallery_${galleryId}`, 3);
            req.onerror = () => reject(req.error);
            req.onsuccess = (e) => resolve(e.target.result);
            req.onupgradeneeded = (e) => {
                const db = e.target.result;
                if (!db.objectStoreNames.contains('photos')) {
                    db.createObjectStore('photos', { keyPath: 'filename' });
                }
                if (!db.objectStoreNames.contains('metadata')) {
                    db.createObjectStore('metadata', { keyPath: 'key' });
                }
            };
        });
    }

    async function savePhoto(galleryId, filename, blob, numero) {
        const db = await openDB(galleryId);
        return new Promise((resolve, reject) => {
            const tx = db.transaction(['photos'], 'readwrite');
            tx.objectStore('photos').put({ filename, blob, numero });
            tx.oncomplete = resolve;
            tx.onerror = () => reject(tx.error);
        });
    }

    async function getAllPhotos(galleryId) {
        const db = await openDB(galleryId);
        return new Promise((resolve, reject) => {
            const req = db.transaction(['photos'], 'readonly').objectStore('photos').getAll();
            req.onsuccess = () => resolve(req.result || []);
            req.onerror = () => reject(req.error);
        });
    }

    async function getStoredNum(galleryId) {
        const db = await openDB(galleryId);
        return new Promise((resolve) => {
            const req = db.transaction(['metadata'], 'readonly').objectStore('metadata').get('numerook');
            req.onsuccess = () => resolve(req.result ? req.result.value : 0);
        });
    }

    async function setStoredNum(galleryId, value) {
        const db = await openDB(galleryId);
        return new Promise((resolve) => {
            const tx = db.transaction(['metadata'], 'readwrite');
            tx.objectStore('metadata').put({ key: 'numerook', value: value });
            tx.oncomplete = resolve;
        });
    }

    // ---------- MANIFEST ----------
    async function fetchManifest(gallery) {
        try {
            const resp = await fetch(gallery.manifest, { cache: 'no-cache' });
            if (!resp.ok) return [];
            const text = await resp.text();
            const lines = text.split(/\r?\n/).filter(l => l.trim());
            const map = new Map();
            const regex = new RegExp(`^${gallery.prefix}(\\d+)\\.([a-zA-Z0-9]+)$`);
            for (const line of lines) {
                const match = line.match(regex);
                if (match) {
                    const num = parseInt(match[1]);
                    if (!map.has(num)) map.set(num, { filename: line, numero: num });
                }
            }
            return Array.from(map.values()).sort((a,b) => a.numero - b.numero);
        } catch(e) { return []; }
    }

    // ---------- SYNC ----------
    async function syncPhotos(galleryId, manifest, localNum) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        if (!manifest.length) return 0;
        const maxRemote = manifest[manifest.length-1].numero;
        if (maxRemote <= localNum) return 0;
        const toDownload = manifest.filter(p => p.numero > localNum);
        let downloaded = 0;
        for (const photo of toDownload) {
            const url = `${gallery.worker}${photo.filename}`;
            try {
                const res = await fetch(url);
                if (res.ok) {
                    await savePhoto(galleryId, photo.filename, await res.blob(), photo.numero);
                    downloaded++;
                }
            } catch(e) {}
        }
        if (downloaded > 0) await setStoredNum(galleryId, maxRemote);
        return downloaded;
    }

    // ---------- RENDER ----------
    async function renderGallery(galleryId) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        const stored = await getAllPhotos(galleryId);
        if (!stored.length) {
            galleryContainer.innerHTML = `<div class="loader-area"><div class="spinner"></div></div>`;
            photoCounterSpan.textContent = '0';
            return;
        }
        const sorted = stored.sort((a,b) => b.numero - a.numero);
        currentPhotoList = sorted;
        photoCounterSpan.textContent = sorted.length;

        let html = '';
        for (let i = 0; i < sorted.length; i++) {
            const item = sorted[i];
            const url = URL.createObjectURL(item.blob);
            const title = gallery.title.replace('{numero}', item.numero);
            html += `
                <div class="photo-card" data-index="${i}">
                    <div class="img-wrap"><img src="${url}" loading="lazy"></div>
                    <div class="photo-title">${title}</div>
                </div>
            `;
        }
        galleryContainer.innerHTML = html;
        document.querySelectorAll('.photo-card').forEach(card => {
            card.addEventListener('click', () => openLightbox(parseInt(card.dataset.index)));
        });
    }

    // ---------- LOAD ----------
    async function loadGallery(galleryId) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        currentGallery = galleryId;
        galleryTitle.textContent = gallery.screenTitle || gallery.label;
        gallerySub.textContent = gallery.screenSub || '';
        selectionScreen.style.opacity = '0';
        setTimeout(async () => {
            selectionScreen.style.display = 'none';
            galleryApp.classList.remove('hidden');
            await renderGallery(galleryId);
            const manifest = await fetchManifest(gallery);
            if (manifest.length) {
                const localNum = await getStoredNum(galleryId);
                const maxRemote = manifest[manifest.length-1].numero;
                if (maxRemote > localNum) {
                    await syncPhotos(galleryId, manifest, localNum);
                    await renderGallery(galleryId);
                }
            }
        }, 300);
    }

    // ---------- LIGHTBOX ----------
    function openLightbox(index) {
        if (!currentPhotoList.length) return;
        currentIndex = index;
        const url = URL.createObjectURL(currentPhotoList[index].blob);
        lbImage.src = url;
        lightboxDiv.classList.add('active');
        lbCounter.textContent = `${index+1} / ${currentPhotoList.length}`;
    }
    function closeLightbox() { lightboxDiv.classList.remove('active'); }
    function nextPhoto() {
        if (currentIndex < currentPhotoList.length - 1) {
            currentIndex++;
            lbImage.src = URL.createObjectURL(currentPhotoList[currentIndex].blob);
            lbCounter.textContent = `${currentIndex+1} / ${currentPhotoList.length}`;
        }
    }
    function prevPhoto() {
        if (currentIndex > 0) {
            currentIndex--;
            lbImage.src = URL.createObjectURL(currentPhotoList[currentIndex].blob);
            lbCounter.textContent = `${currentIndex+1} / ${currentPhotoList.length}`;
        }
    }
    closeLb.addEventListener('click', closeLightbox);
    prevLb.addEventListener('click', prevPhoto);
    nextLb.addEventListener('click', nextPhoto);
    lightboxDiv.addEventListener('click', (e) => { if (e.target === lightboxDiv) closeLightbox(); });
    document.addEventListener('keydown', (e) => {
        if (!lightboxDiv.classList.contains('active')) return;
        if (e.key === 'Escape') closeLightbox();
        if (e.key === 'ArrowRight') nextPhoto();
        if (e.key === 'ArrowLeft') prevPhoto();
    });
</script>
