        // ==================== INDEXEDDB ====================
        function ouvrirDB() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.open(DB_NAME, 2);
                request.onerror = () => reject(request.error);
                request.onsuccess = (event) => resolve(event.target.result);
                request.onupgradeneeded = (event) => {
                    const db = event.target.result;
                    if (!db.objectStoreNames.contains(STORE_NAME)) {
                        db.createObjectStore(STORE_NAME);
                    }
                };
            });
        }

        async function sauvegarderDansCache(cartesData, version) {
            try {
                const db = await ouvrirDB();
                return new Promise((resolve, reject) => {
                    const transaction = db.transaction([STORE_NAME], 'readwrite');
                    const store = transaction.objectStore(STORE_NAME);
                    store.put(cartesData, STORE_KEY);
                    if (version) store.put(version, VERSION_KEY);
                    transaction.oncomplete = () => resolve();
                    transaction.onerror = () => reject(transaction.error);
                });
            } catch (error) {
                console.error("Erreur sauvegarde:", error);
            }
        }

        async function recupererDepuisCache() {
            try {
                const db = await ouvrirDB();
                return new Promise((resolve) => {
                    const transaction = db.transaction([STORE_NAME], 'readonly');
                    const store = transaction.objectStore(STORE_NAME);
                    const reqContent = store.get(STORE_KEY);
                    const reqVersion = store.get(VERSION_KEY);
                    let content = null, version = null;
                    let completed = 0;
                    function check() {
                        completed++;
                        if (completed === 2) resolve({ content, version });
                    }
                    reqContent.onsuccess = () => { content = reqContent.result; check(); };
                    reqVersion.onsuccess = () => { version = reqVersion.result; check(); };
                    reqContent.onerror = () => check();
                    reqVersion.onerror = () => check();
                });
            } catch (error) {
                return { content: null, version: null };
            }
        }

        async function supprimerBaseCompletement() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.deleteDatabase(DB_NAME);
                request.onsuccess = () => resolve();
                request.onerror = () => reject(request.error);
            });
        }

        async function getCacheSize() {
            try {
                const db = await ouvrirDB();
                return new Promise((resolve) => {
                    const transaction = db.transaction([STORE_NAME], 'readonly');
                    const store = transaction.objectStore(STORE_NAME);
                    const request = store.get(STORE_KEY);
                    request.onsuccess = () => {
                        const data = request.result;
                        if (data) {
                            const size = new Blob([JSON.stringify(data)]).size;
                            const kb = (size / 1024).toFixed(1);
                            cacheSizeSpan.textContent = `📦 ${kb} Ko`;
                        } else {
                            cacheSizeSpan.textContent = `📦 vide`;
                        }
                        resolve();
                    };
                    request.onerror = () => resolve();
                });
            } catch (e) {
                cacheSizeSpan.textContent = `📦 --`;
            }
        }

        // ==================== VERSION DISTANTE ====================
        async function getRemoteVersionDate() {
            try {
                const res = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
                if (!res.ok) return null;
                const text = await res.text();
                const blocks = text.split(/DERNIÈRE MISE À JOUR/);
                for (const block of blocks) {
                    const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                    if (fileMatch && fileMatch[1].trim() === TARGET_FILE) {
                        const dayMatch = block.match(/📅\s*([\d\/]+)/);
                        const timeMatch = block.match(/⏰\s*([\d:]+)/);
                        if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
                    }
                }
                return null;
            } catch (e) { 
                return null; 
            }
        }

        // ==================== PARSEUR ====================
        function parserFichierTexte(content) {
            const cartesResult = [];
            const lignes = content.split('\n');
            
            let currentCarte = null;
            let currentExpressions = [];
            
            for (let i = 0; i < lignes.length; i++) {
                let ligne = lignes[i].trim();
                if (ligne === '') continue;
                
                const titreMatch = ligne.match(/^Titre\s*:\s*(.+)$/i);
                if (titreMatch) {
                    if (currentCarte && currentExpressions.length > 0) {
                        cartesResult.push({
                            titre: currentCarte,
                            expressions: [...currentExpressions]
                        });
                    }
                    currentCarte = titreMatch[1].trim();
                    currentExpressions = [];
                    continue;
                }
                
                const numMatch = ligne.match(/^(\d+)\.\s*(.+?)\s*:?\s*$/);
                if (numMatch && currentCarte) {
                    let expressionFr = numMatch[2].trim();
                    expressionFr = expressionFr.replace(/:$/, '').trim();
                    
                    let traduction = '';
                    let nextLigne = i + 1 < lignes.length ? lignes[i+1].trim() : '';
                    if (nextLigne.match(/^traduction\s*:/i)) {
                        traduction = nextLigne.replace(/^traduction\s*:\s*/i, '').trim();
                        i++;
                    }
                    
                    currentExpressions.push({
                        num: parseInt(numMatch[1]),
                        fr: expressionFr,
                        mg: traduction || ''
                    });
                    continue;
                }
            }
            
            if (currentCarte && currentExpressions.length > 0) {
                cartesResult.push({
                    titre: currentCarte,
                    expressions: [...currentExpressions]
                });
            }
            
            return cartesResult;
        }

        async function telechargerFichier() {
            const response = await fetch(TXT_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            if (!content || content.length < 50) throw new Error("Fichier vide");
            return content;
        }

        // ==================== MISE À JOUR AUTOMATIQUE SILENCIEUSE ====================
        async function miseAJourAutomatique() {
            if (!navigator.onLine || isUpdating) return false;
            
            try {
                const remoteVersion = await getRemoteVersionDate();
                if (!remoteVersion) return false;
                
                // Si la version distante est différente de la version locale
                if (currentVersionDate !== remoteVersion) {
                    console.log("🔄 Nouvelle version détectée:", remoteVersion);
                    isUpdating = true;
                    
                    // Télécharger le nouveau fichier
                    const nouveauContenu = await telechargerFichier();
                    const nouvellesCartes = parserFichierTexte(nouveauContenu);
                    
                    if (nouvellesCartes && nouvellesCartes.length > 0) {
                        // Remplacer complètement les données
                        cartes = nouvellesCartes;
                        currentVersionDate = remoteVersion;
                        
                        // Sauvegarder dans le cache
                        await sauvegarderDansCache(cartes, remoteVersion);
                        
                        // Mettre à jour l'affichage
                        statusText.innerHTML = "📦 " + cartes.length + " cartes";
                        if (versionText) versionText.textContent = `📅 ${remoteVersion}`;
                        await getCacheSize();
                        afficherListe();
                        
                        // Notification discrète (optionnelle)
                        showNotificationSilencieuse("", "#FFFFFF");
                        console.log("✅ Mise à jour automatique effectuée: " + remoteVersion);
                    }
                    isUpdating = false;
                    return true;
                }
            } catch (error) {
                console.error("Erreur mise à jour auto:", error);
                isUpdating = false;
            }
            return false;
        }

        function ouvrirCarte(index) {

            currentCarteData = JSON.parse(JSON.stringify(cartes[index]));
            
            const carte = currentCarteData;
            const expressions = carte.expressions;
            
            let expressionsHtml = '';
            for (let i = 0; i < 4; i++) {
                const exp = expressions[i] || { fr: '', mg: '' };
                expressionsHtml += `
                    <div class="expression">
                        <div contenteditable="true" class="fr" id="fr${i+1}">${escapeHtml(exp.fr || '')}</div>
                        <div contenteditable="true" class="mg" id="mg${i+1}">${escapeHtml(exp.mg || '')}</div>
                    </div>
                `;
            }
            
            const carteHtml = `
                <div class="carte-container" style="background-image: url('${CARTE_IMAGE_URL}'); background-size: contain; background-position: center; background-repeat: no-repeat; background-color: #1a1a2e;">
                    <div class="contenu-carte">
                        <div class="titre-carte">
                            <h1>${carte.titre.includes('AMOUR') ? '💖 ' + escapeHtml(carte.titre) : '🌙 ' + escapeHtml(carte.titre)}</h1>
                            <p>4 expressions du cœur</p>
                        </div>
                        <div class="expressions">
                            ${expressionsHtml}
                        </div>
                        <div class="message-bas">
                            💌 À la personne que j'aime 💌
                        </div>
                    </div>
                </div>
            `;
            
            carteDisplay.innerHTML = carteHtml;
            
            for (let i = 1; i <= 4; i++) {
                const frEl = document.getElementById(`fr${i}`);
                const mgEl = document.getElementById(`mg${i}`);
                if (frEl) frEl.addEventListener('blur', () => sauvegarderCarteCourante());
                if (mgEl) mgEl.addEventListener('blur', () => sauvegarderCarteCourante());
            }
            
            screenList.classList.add('hidden');
            screenCarte.classList.remove('hidden');
        }
        
        function escapeHtml(str) {
            if (!str) return '';
            return str.replace(/[&<>]/g, function(m) {
                if (m === '&') return '&amp;';
                if (m === '<') return '&lt;';
                if (m === '>') return '&gt;';
                return m;
            });
        }
        
        async function sauvegarderCarteCourante() {
            if (!currentCarteData) return;
            
            const nouvellesExpressions = [];
            for (let i = 1; i <= 4; i++) {
                const frEl = document.getElementById(`fr${i}`);
                const mgEl = document.getElementById(`mg${i}`);
                nouvellesExpressions.push({
                    num: i,
                    fr: frEl ? frEl.innerText : '',
                    mg: mgEl ? mgEl.innerText : ''
                });
            }
            
            currentCarteData.expressions = nouvellesExpressions;
            cartes[currentCarteIndex] = currentCarteData;
            await sauvegarderDansCache(cartes, currentVersionDate);
            await getCacheSize();
            
            toast.style.opacity = '1';
            setTimeout(() => { toast.style.opacity = '0'; }, 1000);
        }

        async function chargerDonnees() {
            prechargerImage();
            
            const cached = await recupererDepuisCache();
            if (cached && cached.content && cached.content.length > 0) {
                cartes = cached.content;
                currentVersionDate = cached.version;
                statusText.innerHTML = "" + cartes.length + " cartes";
                if (versionText && currentVersionDate) versionText.textContent = ``;
                await getCacheSize();
                afficherListe();
                
                // Lance la vérification automatique après 2 secondes
                if (navigator.onLine) {
                    setTimeout(() => {
                        miseAJourAutomatique();
                    }, 2000);
                    
                    // Vérifie toutes les 60 secondes
                    setInterval(() => {
                        miseAJourAutomatique();
                    }, 60000);
                }
                return;
            }
            
            statusText.innerHTML = "🌐 Téléchargement...";
            try {
                const content = await telechargerFichier();
                const remoteVersion = await getRemoteVersionDate();
                cartes = parserFichierTexte(content);
                currentVersionDate = remoteVersion || new Date().toLocaleString('fr-FR');
                await sauvegarderDansCache(cartes, currentVersionDate);
                statusText.innerHTML = "✅ " + cartes.length + " cartes";
                if (versionText) versionText.textContent = `📅 ${currentVersionDate}`;
                await getCacheSize();
                afficherListe();
            } catch (error) {
                statusText.innerHTML = "❌ Erreur";
                carteListContainer.innerHTML = `<div class="loading"><p>❌ Impossible de charger.</p><button onclick="location.reload()" style="margin-top:15px;padding:8px 20px;border-radius:30px;border:none;background:#3b82f6;color:white;">🔄 Réessayer</button></div>`;
            }
        }
