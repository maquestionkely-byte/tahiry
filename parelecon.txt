    // ==================== CONFIGURATION ====================
    const WORKER_URL = 'https://falytokoa.ahofaly62.workers.dev';
    const FILE_URL = WORKER_URL + '/prprlecon.txt';
    const DATE_INFO_URL = WORKER_URL + '/date_rehetra.txt';
    const DB_NAME = 'prprlecon_storage';
    const STORE_NAME = 'prprlecon_content';
    const DB_VERSION = 18;
    const TARGET_FILE = 'prprlecon.txt';

    let currentHtmlContent = null;
    let currentVersionDate = null;
    let isUpdating = false;
    let isFirstLoad = true;
    let pendingDownload = false;
    let networkListenerAttached = false;

    // ======================= INDEXEDDB =======================
    function ouvrirDB() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(DB_NAME, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (db.objectStoreNames.contains(STORE_NAME)) {
                    db.deleteObjectStore(STORE_NAME);
                }
                db.createObjectStore(STORE_NAME);
            };
        });
    }

    async function stockerDansIndexedDB(contenuHTML, versionDate) {
        const db = await ouvrirDB();
        return new Promise((resolve, reject) => {
            const transaction = db.transaction([STORE_NAME], 'readwrite');
            const store = transaction.objectStore(STORE_NAME);
            store.put(contenuHTML, 'htmlContent');
            store.put(versionDate, 'versionDate');
            transaction.oncomplete = () => resolve();
            transaction.onerror = () => reject(transaction.error);
        });
    }

    async function recupererDepuisIndexedDB() {
        const db = await ouvrirDB();
        return new Promise((resolve) => {
            const transaction = db.transaction([STORE_NAME], 'readonly');
            const store = transaction.objectStore(STORE_NAME);
            const htmlRequest = store.get('htmlContent');
            const dateRequest = store.get('versionDate');
            let htmlContent = null;
            let versionDate = null;
            let completed = 0;
            function checkComplete() {
                completed++;
                if (completed === 2) resolve({ htmlContent, versionDate });
            }
            htmlRequest.onsuccess = () => { htmlContent = htmlRequest.result; checkComplete(); };
            dateRequest.onsuccess = () => { versionDate = dateRequest.result; checkComplete(); };
            htmlRequest.onerror = () => checkComplete();
            dateRequest.onerror = () => checkComplete();
        });
    }

    // ======================= EXTRAIRE LA DATE DE VERSION =======================
    async function getRemoteVersionDate() {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === TARGET_FILE) {
                    const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (dateMatch) return dateMatch[1].trim();
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
                }
            }
            return null;
        } catch (error) {
            return null;
        }
    }

    // ======================= TÉLÉCHARGER LE CONTENU =======================
    async function telechargerContenuHeraa() {
        const response = await fetch(FILE_URL, { cache: 'no-cache' });
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Contenu invalide");
        return content;
    }

    // ======================= AFFICHER LE CONTENU =======================
    function afficherContenu(htmlContent) {
        try {
            document.body.innerHTML = '';
            document.body.style.margin = '0';
            document.body.style.padding = '0';
            
            const iframe = document.createElement('iframe');
            iframe.style.position = 'fixed';
            iframe.style.top = '0';
            iframe.style.left = '0';
            iframe.style.width = '100%';
            iframe.style.height = '100%';
            iframe.style.border = 'none';
            iframe.style.zIndex = '9999';
            iframe.style.backgroundColor = 'white';
            iframe.srcdoc = htmlContent;
            document.body.appendChild(iframe);
            
            attachNetworkListeners();
        } catch (error) {
            console.error("Erreur:", error);
        }
    }

    // ======================= TÉLÉCHARGEMENT COMPLET =======================
    async function effectuerTelechargement() {
        if (isUpdating) return false;
        isUpdating = true;
        
        try {
            const remoteVersionDate = await getRemoteVersionDate();
            const finalVersionDate = remoteVersionDate || new Date().toLocaleString('fr-FR');
            const nouveauContenu = await telechargerContenuHeraa();
            await stockerDansIndexedDB(nouveauContenu, finalVersionDate);
            currentHtmlContent = nouveauContenu;
            currentVersionDate = finalVersionDate;
            afficherContenu(nouveauContenu);
            isUpdating = false;
            return true;
        } catch (error) {
            isUpdating = false;
            throw error;
        }
    }

    // ======================= PREMIER TÉLÉCHARGEMENT =======================
    async function premierTelechargement() {
        if (isUpdating) return;
        isUpdating = true;
        
        try {
            const remoteVersionDate = await getRemoteVersionDate();
            const finalVersionDate = remoteVersionDate || new Date().toLocaleString('fr-FR');
            const nouveauContenu = await telechargerContenuHeraa();
            await stockerDansIndexedDB(nouveauContenu, finalVersionDate);
            currentHtmlContent = nouveauContenu;
            currentVersionDate = finalVersionDate;
            afficherContenu(nouveauContenu);
            isFirstLoad = false;
            isUpdating = false;
        } catch (error) {
            isUpdating = false;
            if (error.message === "Failed to fetch" || error.name === "NetworkError" || error.message.includes("fetch")) {
                pendingDownload = true;
            }
        }
    }

    // ======================= MISE À JOUR SILENCIEUSE =======================
    async function miseAJourSilencieuse() {
        if (isUpdating) return;
        
        try {
            const remoteVersionDate = await getRemoteVersionDate();
            if (!remoteVersionDate) return;
            
            const cache = await recupererDepuisIndexedDB();
            const storedVersionDate = cache.versionDate;
            
            if (storedVersionDate && storedVersionDate === remoteVersionDate) return;
            
            const nouveauContenu = await telechargerContenuHeraa();
            await stockerDansIndexedDB(nouveauContenu, remoteVersionDate);
            currentHtmlContent = nouveauContenu;
            currentVersionDate = remoteVersionDate;
            afficherContenu(nouveauContenu);
        } catch (error) {
            console.error("Erreur mise à jour:", error);
        }
    }

    // ======================= ÉCOUTEURS RÉSEAU =======================
    function attachNetworkListeners() {
        if (networkListenerAttached) return;
        
        window.addEventListener('online', () => {
            if (pendingDownload) {
                pendingDownload = false;
                setTimeout(() => premierTelechargement(), 500);
            } else if (!isFirstLoad && currentHtmlContent) {
                setTimeout(() => miseAJourSilencieuse(), 1000);
            }
        });
        
        window.addEventListener('offline', () => {
            if (isFirstLoad && !currentHtmlContent) {
                pendingDownload = true;
            }
        });
        
        if (!navigator.onLine && isFirstLoad && !currentHtmlContent) {
            pendingDownload = true;
        }
        
        networkListenerAttached = true;
    }

    // ======================= CHARGEMENT PRINCIPAL =======================
    async function charger() {
        try {
            const cache = await recupererDepuisIndexedDB();
            
            if (cache.htmlContent && cache.versionDate) {
                isFirstLoad = false;
                currentHtmlContent = cache.htmlContent;
                currentVersionDate = cache.versionDate;
                afficherContenu(cache.htmlContent);
                attachNetworkListeners();
                setTimeout(() => miseAJourSilencieuse(), 2000);
            } else {
                isFirstLoad = true;
                attachNetworkListeners();
                if (navigator.onLine) {
                    await premierTelechargement();
                } else {
                    pendingDownload = true;
                }
            }
        } catch (error) {
            attachNetworkListeners();
            if (navigator.onLine) {
                setTimeout(() => premierTelechargement(), 1000);
            } else {
                pendingDownload = true;
            }
        }
    }

    charger();
