            <button id="sendBtn" onclick="sendMessage()">Envoyer</button>

    <script>
        // Éléments DOM
        const chatMessages = document.getElementById('chatMessages');
        const userInput = document.getElementById('userInput');
        const sendBtn = document.getElementById('sendBtn');
        const statusBadge = document.getElementById('statusBadge');
        
        let isWaiting = false;
        
        // Gestion de la touche Entrée
        function handleKeyPress(event) {
            if (event.key === 'Enter' && !isWaiting) {
                sendMessage();
            }
        }
        
        // Ajouter un message dans l'interface
        function addMessage(content, isUser = false) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${isUser ? 'user-message' : 'bot-message'}`;
            
            const avatar = document.createElement('div');
            avatar.className = `avatar ${isUser ? 'user-avatar' : 'bot-avatar'}`;
            avatar.textContent = isUser ? '👤' : '🤖';
            
            const contentDiv = document.createElement('div');
            contentDiv.className = 'message-content';
            contentDiv.innerHTML = formatMessage(content);
            
            messageDiv.appendChild(avatar);
            messageDiv.appendChild(contentDiv);
            
            chatMessages.appendChild(messageDiv);
            chatMessages.scrollTop = chatMessages.scrollHeight;
            
            return contentDiv;
        }
        
        // Formater le message (liens, gras, etc.)
        function formatMessage(text) {
            // Convertir les URLs en liens cliquables
            const urlRegex = /(https?:\/\/[^\s]+)/g;
            text = text.replace(urlRegex, '<a href="$1" target="_blank" style="color: #667eea; text-decoration: underline;">$1</a>');
            
            // Convertir le texte en paragraphes
            text = text.replace(/\n/g, '<br>');
            
            return text;
        }
        
        // Afficher l'indicateur de frappe
        function showTypingIndicator() {
            const typingDiv = document.createElement('div');
            typingDiv.className = 'typing-indicator';
            typingDiv.id = 'typingIndicator';
            typingDiv.innerHTML = '<span></span><span></span><span></span>';
            
            // Ajouter l'avatar fantôme
            const wrapper = document.createElement('div');
            wrapper.style.display = 'flex';
            wrapper.style.gap = '12px';
            
            const ghostAvatar = document.createElement('div');
            ghostAvatar.className = 'avatar bot-avatar';
            ghostAvatar.textContent = '🤖';
            ghostAvatar.style.opacity = '0.5';
            
            wrapper.appendChild(ghostAvatar);
            wrapper.appendChild(typingDiv);
            
            chatMessages.appendChild(wrapper);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }
        
        function hideTypingIndicator() {
            const indicator = document.querySelector('.typing-indicator');
            if (indicator && indicator.parentElement) {
                indicator.parentElement.remove();
            }
        }
        
        // Mettre à jour le statut
        function updateStatus(text, isError = false) {
            statusBadge.textContent = text;
            statusBadge.style.background = isError ? 'rgba(239, 68, 68, 0.2)' : 'rgba(255,255,255,0.2)';
            setTimeout(() => {
                if (!isWaiting) {
                    statusBadge.textContent = '✅ Prêt';
                    statusBadge.style.background = 'rgba(255,255,255,0.2)';
                }
            }, 2000);
        }
        
        // Message principal - Puter.js tourne en arrière-plan
        async function sendMessage() {
            const message = userInput.value.trim();
            if (!message || isWaiting) return;
            
            // Afficher le message utilisateur
            addMessage(message, true);
            userInput.value = '';
            isWaiting = true;
            sendBtn.disabled = true;
            updateStatus('🤔 Réflexion en cours...');
            
            // Afficher l'indicateur de frappe
            showTypingIndicator();
            
            try {
                // Appel à Puter.js (tourne EN ARRIÈRE-PLAN)
                // Pendant que ça tourne, l'interface HTML reste réactive
                const response = await puter.ai.chat(message);
                
                // Cacher l'indicateur
                hideTypingIndicator();
                
                // Afficher la réponse
                addMessage(response, false);
                updateStatus('✅ Réponse reçue');
                
            } catch (error) {
                console.error('Erreur Puter:', error);
                hideTypingIndicator();
                addMessage(`❌ Désolé, une erreur est survenue : ${error.message || 'Problème de connexion'}`, false);
                updateStatus('❌ Erreur', true);
            } finally {
                isWaiting = false;
                sendBtn.disabled = false;
                userInput.focus();
            }
        }
        
        // Effacer la conversation
        function clearChat() {
            if (isWaiting) return;
            
            // Garder seulement le premier message du bot
            while (chatMessages.children.length > 1) {
                chatMessages.removeChild(chatMessages.lastChild);
            }
            
            // Réinitialiser le message d'accueil
            const firstMessage = chatMessages.children[0];
            if (firstMessage) {
                const contentDiv = firstMessage.querySelector('.message-content');
                if (contentDiv) {
                    contentDiv.innerHTML = 'Bonjour ! 👋<br><br>Je suis entièrement gratuit, sans clé API. Posez-moi toutes vos questions !';
                }
            }
            
            updateStatus('🧹 Conversation effacée');
        }
        
        // Vérifier que Puter.js est chargé
        if (typeof puter === 'undefined') {
            console.error('Puter.js non chargé');
            addMessage('⚠️ Erreur de chargement de Puter.js. Rafraîchissez la page.', false);
        } else {
            console.log('✅ Puter.js prêt');
        }
        
        // Focus initial
        userInput.focus();
    </script>
