    function ouvrirDB(nomDB) {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(nomDB, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains('contenu')) db.createObjectStore('contenu');
                if (!db.objectStoreNames.contains('version')) db.createObjectStore('version');
            };
        });
    }
    
    async function stockerContenuEtVersion(nomDB, contenu, version) {
        const db = await ouvrirDB(nomDB);
        return new Promise((resolve, reject) => {
            const transaction = db.transaction(['contenu', 'version'], 'readwrite');
            transaction.objectStore('contenu').put(contenu, 'htmlContent');
            transaction.objectStore('version').put(version, 'versionDate');
            transaction.oncomplete = () => resolve();
            transaction.onerror = () => reject(transaction.error);
        });
    }
    
    async function recupererVersion(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['version'], 'readonly');
                const request = transaction.objectStore('version').get('versionDate');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch { return null; }
    }
    
    async function recupererContenu(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['contenu'], 'readonly');
                const request = transaction.objectStore('contenu').get('htmlContent');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch { return null; }
    }
    
    async function telechargerFichier(nomFichier) {
        const url = `${WORKER_URL}/${nomFichier}`;
        const response = await fetch(url, { cache: 'no-cache' });
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Contenu invalide");
        return content;
    }
    
    async function getRemoteVersionDate(nomFichier) {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === nomFichier) {
                    const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (dateMatch) return dateMatch[1].trim();
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
                }
            }
            return null;
        } catch { return null; }
    }

async function verifierTousLesFichiers() {
    try {
        // 1. Télécharger date_rehetra.txt une seule fois
        const response = await fetch(DATE_INFO_URL + '?t=' + Date.now());
        const contenuDateRehetra = await response.text();
        
        // 2. Construire le dictionnaire des versions serveur
        const versionServeur = new Map();
        const blocks = contenuDateRehetra.split(/DERNIÈRE MISE À JOUR/);
        
        for (const block of blocks) {
            const fichierMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
            const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
            if (fichierMatch && dateMatch) {
                versionServeur.set(fichierMatch[1].trim(), dateMatch[1].trim());
            }
        }
        
        // 3. Générer tous les fichiers à vérifier
        const tousLesFichiers = genererTousLesFichiersAVerifier();
        
        // 4. Collecter les fichiers qui ont changé
        const fichiersModifies = [];
        
        for (const fichier of tousLesFichiers) {
            const versionServeurFichier = versionServeur.get(fichier.nom);
            if (!versionServeurFichier) continue;
            
            const versionLocale = await recupererVersion(fichier.dbName);
            
            if (versionLocale !== versionServeurFichier) {
                fichiersModifies.push({
                    nom: fichier.nom,
                    dbName: fichier.dbName,
                    nouvelleVersion: versionServeurFichier
                });
            }
        }
        
        // 5. Mettre à jour uniquement les fichiers modifiés (par lots de 5)
        if (fichiersModifies.length > 0) {
            const LOT_TAILLE = 5;
            
            for (let i = 0; i < fichiersModifies.length; i += LOT_TAILLE) {
                const lot = fichiersModifies.slice(i, i + LOT_TAILLE);
                
                const promises = lot.map(async (fichier) => {
                    try {
                        const nouveauContenu = await telechargerFichier(fichier.nom);
                        await stockerContenuEtVersion(fichier.dbName, nouveauContenu, fichier.nouvelleVersion);
                    } catch (error) {
                        // Erreur silencieuse
                    }
                });
                
                await Promise.all(promises);
                
                // Petite pause entre les lots
                if (i + LOT_TAILLE < fichiersModifies.length) {
                    await new Promise(r => setTimeout(r, 50));
                }
            }
        }
        
    } catch (error) {
        // Silence total en cas d'erreur
    }
}

    async function chargerLecon(titre, nomFichier, urlComplet, contenuDefaut) {
    const modal = document.createElement('div');
    modal.className = 'modal-fullscreen';
    modal.innerHTML = `
    <div class="modal-fullscreen-header">

    try {
        let contenu = await recupererContenu(nomFichier.replace('.txt', ''));
        
        if (!contenu) {
            // ✏️ MODIFICATION: Ajout du deviceModel dans l'URL
            const urlAvecDevice = `${WORKER_URL}/${nomFichier}?deviceModel=${encodeURIComponent(deviceModel)}`;
            const response = await fetch(urlAvecDevice);
            
            // ✏️ MODIFICATION: Gestion du code requis (403)
            if (response.status === 403) {
                const errorData = await response.json();
                if (errorData.needsCode) {
                    modal.remove();
                    pendingRequest = { 
                        fileName: nomFichier, 
                        niveau: errorData.niveau || getNiveauParFichier(nomFichier),
                        deviceModel: deviceModel,
                        titre: titre,
                        url: urlAvecDevice
                    };
                    document.getElementById('modalNiveau').innerText = pendingRequest.niveau;
                    document.getElementById('modalFileName').innerText = nomFichier;
                    document.getElementById('codeModal').classList.add('active');
                    return;
                }
                throw new Error(errorData.message || 'Accès refusé');
            }
            
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            contenu = await response.text();
            await stockerEnIndexedDB(nomFichier, contenu);
        }
        
        modal.remove();
        const blob = new Blob([contenu], { type: 'text/html' });
        const blobUrl = URL.createObjectURL(blob);
        localStorage.setItem('dernierContexteLecon', JSON.stringify({ leconTitre: titre, nomFichier: nomFichier }));
        localStorage.setItem('pagePrincipaleUrl', window.location.href);
        window.location.href = blobUrl;
        
    } catch (error) {
        modal.remove();
        // ✏️ MODIFICATION: Petit message discret au lieu de grosse erreur
        const notif = document.createElement('div');
        notif.className = 'update-notification error';
        notif.style.background = '#c0392b';
        notif.innerHTML = `⚠️ Mbola tsy vita téléchargé !`;
        document.body.appendChild(notif);
        setTimeout(() => notif.remove(), 3000);
    }
}

// ======================= VÉRIFICATION DES NIVEAUX DANS LA DB =======================
async function verifierNiveauxDansDB(leconTitre) {
    const niveauxTrouves = [];
    const codeLecon = getCodeLecon(leconTitre);
    
    // Tous les types d'exercices à vérifier
    const tousLesTypes = TYPES_EXERCICE_CONFIG.map(t => t.code);
    
    // Vérifier les niveaux 1 à 5 pour chaque type
    for (const niveau of [1, 2, 3, 4, 5]) {
        for (const typeCode of tousLesTypes) {
            const nomFichier = `${codeLecon}${typeCode}${niveau}.txt`;
            const dbName = nomFichier.replace('.txt', '');
            
            try {
                // Vérifier si le contenu existe dans IndexedDB
                const contenuExistant = await recupererContenu(dbName);
                if (contenuExistant && contenuExistant.length > 50) {
                    if (!niveauxTrouves.includes(niveau)) {
                        niveauxTrouves.push(niveau);
                    }
                    break; // Sortir de la boucle des types si on a trouvé ce niveau
                }
            } catch (e) {
                // Ignorer les erreurs
            }
        }
    }
    
    return niveauxTrouves;
}

            }
        }
    }
});

    async function stockerImage(url, blob) {
        return new Promise((resolve) => {
            const request = indexedDB.open('image_cache', 7);
            request.onupgradeneeded = (e) => { if (!e.target.result.objectStoreNames.contains('images')) e.target.result.createObjectStore('images'); };
            request.onsuccess = (e) => {
                const db = e.target.result;
                const tx = db.transaction('images', 'readwrite');
                tx.objectStore('images').put(blob, url);
                tx.oncomplete = () => { db.close(); resolve(); };
            };
        });
    }
    
    async function chargerImage(element, url) {
        const request = indexedDB.open('image_cache', 7);
        request.onsuccess = async (e) => {
            const db = e.target.result;
            if (db.objectStoreNames.contains('images')) {
                const tx = db.transaction('images', 'readonly');
                const get = tx.objectStore('images').get(url);
                get.onsuccess = () => {
                    if (get.result) { element.src = URL.createObjectURL(get.result); db.close(); return; }
                    db.close();
                    fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
                };
            } else {
                db.close();
                fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
            }
        };
    }
