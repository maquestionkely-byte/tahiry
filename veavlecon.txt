<script>
    // Base de questions : mélange d'utilisation du présent (avoir/être)
    const questionsData = [
        { text: "Nous ______ contents de te voir.", correct: "sommes", verb: "être" },
        { text: "Elles ______ deux chiens adorables.", correct: "ont", verb: "avoir" },
        { text: "Je ______ fatigué aujourd'hui.", correct: "suis", verb: "être" },
        { text: "Tu ______ un joli sourire.", correct: "as", verb: "avoir" },
        { text: "Il ______ toujours à l'heure.", correct: "est", verb: "être" },
        { text: "Vous ______ raison dans ce débat.", correct: "avez", verb: "avoir" },
        { text: "Marie et Sophie ______ les meilleures amies.", correct: "sont", verb: "être" },
        { text: "Mon frère ______ 15 ans.", correct: "a", verb: "avoir" },
        { text: "Nous ______ déjà mangé.", correct: "avons", verb: "avoir" },
        { text: "Ils ______ très talentueux.", correct: "sont", verb: "être" }
    ];

    // On va prendre 6 questions aléatoires pour que ce soit dynamique mais pédagogique
    let currentQuestions = [];
    let userAnswers = new Array(6).fill(null); // stocke la réponse sélectionnée (texte)
    let answerStatus = new Array(6).fill(null); // 'correct', 'wrong' ou null
    let score = 0;

    let totalQ = 0;

    function shuffleArray(arr) {
        for (let i = arr.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
    }

    function initQuiz() {
        // Sélectionner 6 questions aléatoires parmi le pool
        const shuffledAll = shuffleArray([...questionsData]);
        currentQuestions = shuffledAll.slice(0, 6);
        totalQ = currentQuestions.length;
        document.getElementById('totalQuestions').innerText = totalQ;
        userAnswers = new Array(totalQ).fill(null);
        answerStatus = new Array(totalQ).fill(null);
        score = 0;
        updateScoreDisplay();
        renderQuestions();
    }

    function renderQuestions() {
        const container = document.getElementById('questionsContainer');
        if (!container) return;
        container.innerHTML = '';
        currentQuestions.forEach((q, idx) => {
            // définir les options possibles selon le verbe (pour l'exercice cohérent)
            let options = [];
            if (q.verb === 'avoir') {
                options = ['ai', 'as', 'a', 'avons', 'avez', 'ont'];
            } else {
                options = ['suis', 'es', 'est', 'sommes', 'êtes', 'sont'];
            }
            // on s'assure que la bonne réponse est toujours présente
            if (!options.includes(q.correct)) {
                options.push(q.correct);
                options = [...new Set(options)];
            }
            // mélanger les options pour chaque question
            const shuffledOpts = shuffleArray([...options]);

            const questionDiv = document.createElement('div');
            questionDiv.className = 'question';
            questionDiv.setAttribute('data-qidx', idx);
            
            const questionText = document.createElement('div');
            questionText.className = 'question-text';
            questionText.innerHTML = `<span style="color:#a5f3fc;">${idx+1}.</span> ${q.text}`;
            
            const optionsDiv = document.createElement('div');
            optionsDiv.className = 'options';
            
            shuffledOpts.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = 'option-btn';
                btn.innerText = opt;
                btn.setAttribute('data-value', opt);
                // si déjà une réponse sélectionnée
                if (userAnswers[idx] === opt) {
                    // Appliquer la classe visuelle selon statut
                    if (answerStatus[idx] === 'correct') {
                        btn.classList.add('correct-glow');
                    } else if (answerStatus[idx] === 'wrong') {
                        btn.classList.add('wrong-glow');
                    }
                }
                btn.addEventListener('click', (e) => {
                    e.stopPropagation();
                    handleAnswer(idx, opt, q.correct);
                });
                optionsDiv.appendChild(btn);
            });
            
            const feedbackSpan = document.createElement('div');
            feedbackSpan.className = 'feedback';
            if (answerStatus[idx] === 'correct') {
                feedbackSpan.innerHTML = '✅ Bonne réponse !';
                feedbackSpan.style.color = '#86efac';
            } else if (answerStatus[idx] === 'wrong') {
                feedbackSpan.innerHTML = `❌ Mauvais. La bonne réponse était : ${q.correct}.`;
                feedbackSpan.style.color = '#fca5a5';
            } else {
                feedbackSpan.innerHTML = '🔍 Clique sur une réponse.';
                feedbackSpan.style.color = '#94a3b8';
            }
            
            questionDiv.appendChild(questionText);
            questionDiv.appendChild(optionsDiv);
            questionDiv.appendChild(feedbackSpan);
            container.appendChild(questionDiv);
        });
    }

    function handleAnswer(qIdx, selected, correctAnswer) {
        // Si déjà répondu correctement, on ne permet pas de changer ? Pour plus de clarté, on permet de changer avant validation ? On va autoriser le changement mais mettre à jour le score proprement.
        // On traite le nouveau statut
        const wasCorrect = (answerStatus[qIdx] === 'correct');
        const isNowCorrect = (selected === correctAnswer);
        
        // Mettre à jour userAnswers et status
        const oldAnswer = userAnswers[qIdx];
        userAnswers[qIdx] = selected;
        
        if (isNowCorrect) {
            if (!wasCorrect) {
                // si c'était faux ou null avant, on augmente le score
                answerStatus[qIdx] = 'correct';
                score++;
            } else {
                // déjà correct, ne rien changer
                answerStatus[qIdx] = 'correct';
            }
        } else {
            // réponse fausse
            if (wasCorrect) {
                // précédemment correct, on perd le point
                answerStatus[qIdx] = 'wrong';
                score--;
            } else {
                answerStatus[qIdx] = 'wrong';
            }
        }
        
        // S'assurer que le score ne descende pas sous 0
        if (score < 0) score = 0;
        updateScoreDisplay();
        // Re-rendu pour actualiser les couleurs des boutons et feedback
        refreshQuestionUI();
    }
    
    function refreshQuestionUI() {
        const container = document.getElementById('questionsContainer');
        if (!container) return;
        const questionDivs = container.querySelectorAll('.question');
        questionDivs.forEach((div, idx) => {
            if (idx >= currentQuestions.length) return;
            const q = currentQuestions[idx];
            const optionsDiv = div.querySelector('.options');
            const feedbackSpan = div.querySelector('.feedback');
            
            // Mettre à jour les classes des boutons
            const btns = optionsDiv.querySelectorAll('.option-btn');
            btns.forEach(btn => {
                const val = btn.getAttribute('data-value');
                btn.classList.remove('correct-glow', 'wrong-glow');
                if (userAnswers[idx] === val) {
                    if (answerStatus[idx] === 'correct') {
                        btn.classList.add('correct-glow');
                    } else if (answerStatus[idx] === 'wrong') {
                        btn.classList.add('wrong-glow');
                    }
                }
            });
            
            // Mettre à jour feedback
            if (answerStatus[idx] === 'correct') {
                feedbackSpan.innerHTML = '✅ Bonne réponse !';
                feedbackSpan.style.color = '#86efac';
            } else if (answerStatus[idx] === 'wrong') {
                feedbackSpan.innerHTML = `❌ Mauvais. La bonne réponse était : ${q.correct}.`;
                feedbackSpan.style.color = '#fca5a5';
            } else {
                feedbackSpan.innerHTML = '🔍 Clique sur une réponse.';
                feedbackSpan.style.color = '#94a3b8';
            }
        });
        updateScoreDisplay();
    }
    
    function updateScoreDisplay() {
        const scoreSpan = document.getElementById('scoreValue');
        if (scoreSpan) scoreSpan.innerText = score;
    }
    
    function resetQuiz() {
        // Réinitialiser complètement les réponses et le score
        userAnswers = new Array(totalQ).fill(null);
        answerStatus = new Array(totalQ).fill(null);
        score = 0;
        updateScoreDisplay();
        // re-rendu (efface les selections)
        refreshQuestionUI();
        // Pour chaque question remettre à l'état initial (feedback neutre)
        const container = document.getElementById('questionsContainer');
        if (container) {
            const questionDivs = container.querySelectorAll('.question');
            questionDivs.forEach((div, idx) => {
                const feedbackSpan = div.querySelector('.feedback');
                if (feedbackSpan) {
                    feedbackSpan.innerHTML = '🔍 Clique sur une réponse.';
                    feedbackSpan.style.color = '#94a3b8';
                }
                const btns = div.querySelectorAll('.option-btn');
                btns.forEach(btn => {
                    btn.classList.remove('correct-glow', 'wrong-glow');
                });
            });
        }
        // reset userAnswers & answerStatus après rendu (déjà fait)
        // On s'assure que le score reste bien 0
        score = 0;
        updateScoreDisplay();
        // on force le refresh complet des états internes (pour éviter incohérence)
        for(let i=0;i<totalQ;i++){
            answerStatus[i] = null;
            userAnswers[i] = null;
        }
        refreshQuestionUI(); // re-synchro
    }
    
    // Transition : afficher l'écran d'attente 2 secondes puis montrer l'exercice
    function startExerciseAfterDelay() {
        const waitingDiv = document.getElementById('waitingScreen');
        const exerciseDiv = document.getElementById('exerciseContainer');
        if (waitingDiv && exerciseDiv) {
            setTimeout(() => {
                waitingDiv.style.opacity = '0';
                setTimeout(() => {
                    waitingDiv.style.display = 'none';
                    exerciseDiv.style.display = 'block';
                    // Initialiser l'exercice seulement maintenant
                    initQuiz();
                    const resetBtn = document.getElementById('resetQuizBtn');
                    if (resetBtn) {
                        // enlever les anciens listeners pour éviter doublons
                        const newReset = resetBtn.cloneNode(true);
                        resetBtn.parentNode.replaceChild(newReset, resetBtn);
                        newReset.addEventListener('click', () => {
                            resetQuiz();
                        });
                    }
                }, 500);
            }, 1800); // 1.8 secondes d'attente agréable (animation simple)
        } else {
            // fallback
            if(exerciseDiv) exerciseDiv.style.display = 'block';
            if(waitingDiv) waitingDiv.style.display = 'none';
            initQuiz();
        }
    }
    
    // Lancer au chargement
    window.addEventListener('DOMContentLoaded', () => {
        startExerciseAfterDelay();
    });
</script>
