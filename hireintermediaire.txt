    function extractTechnicalModel(ua) {
        const patterns = [
            /;\s([A-Za-z0-9\-\s]+?)\sBuild\//,
            /;\s([A-Za-z0-9\-_]+?)\)/,
            /;\s([A-Za-z0-9\-_]+?);/,
            /\(([A-Za-z0-9\-_]+?);\sAndroid/
        ];
        for (let pattern of patterns) {
            const match = ua.match(pattern);
            if (match && match[1] && !match[1].match(/Android|AppleWebKit|Chrome|Mobile|Safari|Version|Gecko|Linux|arm|arm64|aarch64/i)) {
                return match[1];
            }
        }
        return "Unknown";
    }

                if (hints.model) model = hints.model;
            } catch(e) {}
        }
        
        deviceModel = model || "Unknown";
        console.log("📱 Modèle détecté:", deviceModel);
        return deviceModel;
    }

    async function fetchWithTimeout(url, options, timeout = 15000) {
        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), timeout);
        try {
            const response = await fetch(url, { ...options, signal: controller.signal });
            clearTimeout(timeoutId);
            return response;
        } catch (error) {
            clearTimeout(timeoutId);
            throw error;
        }
    }

    let db = null;
    const DB_NAME = 'Discussion_intermed';
    const DB_VERSION = 3;
    const STORE_NAME = 'messages';
    
    function openDB() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(DB_NAME, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = () => {
                db = request.result;
                resolve(db);
            };
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains(STORE_NAME)) {
                    db.createObjectStore(STORE_NAME, { keyPath: 'id', autoIncrement: true });
                }
            };
        });
    }
    
    async function saveMessageToDB(message) {
        return new Promise((resolve, reject) => {
            if (!db) { reject('DB non initialisée'); return; }
            const transaction = db.transaction([STORE_NAME], 'readwrite');
            const objectStore = transaction.objectStore(STORE_NAME);
            const request = objectStore.add({
                text: message.text,
                type: message.type,
                timestamp: message.timestamp,
                time: message.time,
                date: message.date
            });
            request.onsuccess = () => resolve(request.result);
            request.onerror = () => reject(request.error);
        });
    }
    
    async function loadMessagesFromDB() {
        return new Promise((resolve, reject) => {
            if (!db) { reject('DB non initialisée'); return; }
            const transaction = db.transaction([STORE_NAME], 'readonly');
            const objectStore = transaction.objectStore(STORE_NAME);
            const request = objectStore.getAll();
            request.onsuccess = () => {
                const messages = request.result;
                messages.sort((a, b) => a.timestamp - b.timestamp);
                resolve(messages);
            };
            request.onerror = () => reject(request.error);
        });
    }

            const request = objectStore.clear();
            request.onsuccess = () => resolve();
            request.onerror = () => reject(request.error);
        });
    }
    
    async function saveUserInfoToDB(name, lieu, gender) {
        return new Promise((resolve, reject) => {
            if (!db) { reject('DB non initialisée'); return; }
            const transaction = db.transaction([STORE_NAME], 'readwrite');
            const objectStore = transaction.objectStore(STORE_NAME);
            // Supprimer anciennes infos
            const getAllRequest = objectStore.getAll();
            getAllRequest.onsuccess = () => {
                const messages = getAllRequest.result;
                const userInfoMessages = messages.filter(msg => msg.type === 'user_info');
                userInfoMessages.forEach(msg => {
                    objectStore.delete(msg.id);
                });
                const request = objectStore.add({
                    type: 'user_info',
                    name: name,
                    lieu: lieu,
                    gender: gender,
                    timestamp: Date.now()
                });
                request.onsuccess = () => resolve();
                request.onerror = () => reject(request.error);
            };
            getAllRequest.onerror = () => reject(getAllRequest.error);
        });
    }

                const messages = request.result;
                const userInfo = messages.find(msg => msg.type === 'user_info');
                if (userInfo) {
                    resolve({ name: userInfo.name, lieu: userInfo.lieu, gender: userInfo.gender });
                } else {
                    resolve(null);
                }
            };
            request.onerror = () => reject(request.error);
        });
    }

    function checkUnlockCode(message) {
        // Nettoyer le message (enlever espaces, mettre en majuscules)
        const cleanMessage = message.trim().toUpperCase();
        
        // Vérifier si le message correspond à LIBR + LETTRE (LIBRA, LIBRB, etc.)
        const librMatch = cleanMessage.match(/^LIBR[A-Z]$/);
        if (librMatch && UNLOCK_CODES[cleanMessage]) {
            const duration = UNLOCK_CODES[cleanMessage];
            const durationHours = duration / (60 * 60 * 1000);
            const durationMinutes = duration / (60 * 1000);
            
            // Calculer la durée de déblocage
            unlockUntilTime = Date.now() + duration;
            
            // Sauvegarder dans localStorage
            localStorage.setItem('unlockUntilTime', unlockUntilTime.toString());
            
            // Réinitialiser les compteurs de blocage
            if (isBlocked) {
                // Arrêter le blocage en cours
                if (countdownInterval) {
                    clearInterval(countdownInterval);
                    countdownInterval = null;
                }
                isBlocked = false;
                blockEndTime = null;
                localStorage.removeItem('blockEndTime');
            }
            
            // Réinitialiser les timestamps
            messageTimestamps = [];
            
            return {
                valid: true,
                duration: duration,
                durationDisplay: `${durationHours} heure(s)`,
                code: cleanMessage,
                type: 'LIBR'
            };
        }
        
        // Vérifier si le message correspond à LOSO + LETTRE (LOSOA, LOSOB, etc.)
        const losoMatch = cleanMessage.match(/^LOSO[A-Z]$/);
        if (losoMatch && LOSO_CODES[cleanMessage]) {
            const duration = LOSO_CODES[cleanMessage];
            const durationMinutes = duration / (60 * 1000);
            
            let durationDisplay = '';
            if (durationMinutes >= 60) {
                const hours = Math.floor(durationMinutes / 60);
                const mins = durationMinutes % 60;
                if (mins > 0) {
                    durationDisplay = `${hours}h ${mins}min`;
                } else {
                    durationDisplay = `${hours} heure(s)`;
                }
            } else {
                durationDisplay = `${durationMinutes} minutes`;
            }
            
            // Calculer la durée de déblocage
            unlockUntilTime = Date.now() + duration;
            
            // Sauvegarder dans localStorage
            localStorage.setItem('unlockUntilTime', unlockUntilTime.toString());
            
            // Réinitialiser les compteurs de blocage
            if (isBlocked) {
                // Arrêter le blocage en cours
                if (countdownInterval) {
                    clearInterval(countdownInterval);
                    countdownInterval = null;
                }
                isBlocked = false;
                blockEndTime = null;
                localStorage.removeItem('blockEndTime');
            }
            
            // Réinitialiser les timestamps
            messageTimestamps = [];
            
            return {
                valid: true,
                duration: duration,
                durationDisplay: durationDisplay,
                code: cleanMessage,
                type: 'LOSO'
            };
        }
        
        return { valid: false };
    }

           function startCountdown() {
        if (countdownInterval) clearInterval(countdownInterval);
        
        countdownInterval = setInterval(async () => {
            if (!isBlocked || !blockEndTime) {
                if (countdownInterval) clearInterval(countdownInterval);
                return;
            }
            
            const now = Date.now();
            const remaining = blockEndTime - now;
            
            if (remaining <= 0) {
                clearInterval(countdownInterval);
                countdownInterval = null;
                await endBlock();
            } else {
                const remainingSeconds = Math.ceil(remaining / 1000);
                const minutes = Math.floor(remainingSeconds / 60);
                const seconds = remainingSeconds % 60;
                const timeStr = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                
                // Mettre à jour TOUS les messages de blocage
                const botMessages = document.querySelectorAll('.message-wrapper.bot');
                
                for (let i = 0; i < botMessages.length; i++) {
                    const messageDiv = botMessages[i].querySelector('.message');
                    if (!messageDiv) continue;
                    
                    const htmlContent = messageDiv.innerHTML;
                    
                    // Vérifier si c'est un message de blocage (contenant du texte spécifique)
                    if (htmlContent.includes('Nasiana fiantoana kely') || 
                        htmlContent.includes('Message tsy lasa') ||
                        htmlContent.includes('Mbola miandry kely ô')) {

                        newContent = newContent.replace(/⏰ Temps restant : \d{2}:\d{2} ⏰/, `⏰ Temps restant : ${timeStr} ⏰`);

            unlockUntilTime = parseInt(savedUnlockTime);

        document.getElementById('welcomeScreen').classList.add('hide');
        
        // Afficher le chat
        chatContainer.style.display = 'flex';
        
        // Activer le plein écran
        enterFullscreen();
        
        // Charger l'historique
        await loadChatHistory();
        
        // Vérifier la connexion
        checkConnection();
        setInterval(checkConnection, 15000);
        
        // Ajouter un message de bienvenue si aucun message
        if (chatBox.children.length === 0) {
            const genderWord = userGender === 'homme' ? 'un homme' : 'une femme';
            let welcomeMsg = `\n\n\n\n\n\n\n\n\n\nSalut ${userName} !😊\n\nJe suis Nirina, ravie de te rencontrer ! `;
            welcomeMsg = cleanBotMessage(welcomeMsg);
            await addMessage(welcomeMsg, 'bot');
        }
        
        // Focus sur l'input
        setTimeout(() => {
            messageInput.focus();
            forceScrollToInput();
        }, 300);
    }

    async function checkExistingUser() {
        await openDB();
        const userInfo = await loadUserInfoFromDB();
        
        if (userInfo && userInfo.name && userInfo.lieu && userInfo.gender) {
            userName = userInfo.name;
            userLieu = userInfo.lieu;
            userGender = userInfo.gender;
            
            console.log("✅ Utilisateur trouvé localement:", userName, userGender, userLieu);
            
            // Afficher l'écran de bienvenue avec bouton Commencer
            document.getElementById('registerScreen').classList.add('hide');
            showWelcomeScreen();
        } else {
            console.log("👤 Nouvel utilisateur - affichage de la sélection de niveau");
                    }
    }

function setupChatEvents() {
    // Empêcher le bouton de prendre le focus (cause de la fermeture du clavier)
    sendButton.addEventListener('mousedown', (e) => {
        e.preventDefault();  // Empêche le bouton de recevoir le focus
    });
    
    sendButton.addEventListener('touchstart', (e) => {
        e.preventDefault();  // Empêche le bouton de recevoir le focus sur mobile

            messageInput.focus();
            forceScrollToInput();
        }, 10);

        e.preventDefault();
        sendMessage();

            const currentHeight = window.visualViewport.height;
            const windowHeight = window.innerHeight;
            const keyboardHeight = windowHeight - currentHeight;
            if (keyboardHeight > 60) {
                isKeyboardVisible = true;
                currentKeyboardHeight = keyboardHeight;
                updateHeaderPosition(currentKeyboardHeight);
                forceScrollToInput();

            }
        }

            return false;
        }

        e.preventDefault();
        return false;
    });

        chatBox.scrollTop = chatBox.scrollHeight;
        forceScrollToInput();
    });
    observer.observe(chatBox, { childList: true, subtree: true });

    async function init() {
        await getDeviceModel();
        await openDB();  // Ouvre la base de données
        
        checkBlockedOnLoad();  // Vérifie le blocage
        
        const usernameInput = document.getElementById('usernameInput');
        const lieuInput = document.getElementById('lieuInput');
        
        function setupAutoCapitalize(inputElement) {
            inputElement.addEventListener('input', function(e) {
                const start = this.selectionStart;
                const end = this.selectionEnd;
                const oldValue = this.value;
                const capitalized = capitalizeFirstLetter(oldValue);
                if (oldValue !== capitalized) {
                    this.value = capitalized;
                    this.setSelectionRange(start, end);
                }
            });
        }
        
        setupAutoCapitalize(usernameInput);
        setupAutoCapitalize(lieuInput);
        setupGenderSelection();
        
        document.getElementById('registerBtn').addEventListener('click', handleRegister);
        
        usernameInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                e.preventDefault();
                lieuInput.focus();
            }
        });
        lieuInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') handleRegister();
        });
        
        document.getElementById('registerScreen').classList.remove('hide');
        
        // Bouton Commencer
        btnStartChat.addEventListener('click', () => {
            startChat();
        });
        
        setupChatEvents();
        setDarkMode(false);
        
        await checkExistingUser();
    }
