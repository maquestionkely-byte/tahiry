    const WORKER_URL = 'https://falytokoa.ahofaly62.workers.dev';

    const DB_VERSION = 21;
    
    // Facebook
    const FB_URL = 'https://www.facebook.com/profile.php?id=61555313604553';

    const COURSES = {
        'hiredebutant.txt': { db: 'hiredebutant', name: 'Débutant' },
        'hireintermediaire.txt': { db: 'hireintermediaire', name: 'Intermédiaire' }

    // ======================= INDEXEDDB =======================
    function ouvrirDB(dbName) {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(dbName, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains('content')) {
                    db.createObjectStore('content');
                }
            };
        });
    }

    async function stockerDansIndexedDB(dbName, contenuHTML, versionDate) {
        const db = await ouvrirDB(dbName);
        return new Promise((resolve, reject) => {
            const transaction = db.transaction(['content'], 'readwrite');
            const store = transaction.objectStore('content');
            store.put(contenuHTML, 'htmlContent');
            store.put(versionDate, 'versionDate');
            transaction.oncomplete = () => resolve();
            transaction.onerror = () => reject(transaction.error);
        });
    }

    async function recupererDepuisIndexedDB(dbName) {
        try {
            const db = await ouvrirDB(dbName);
            return new Promise((resolve) => {
                const transaction = db.transaction(['content'], 'readonly');
                const store = transaction.objectStore('content');
                const htmlRequest = store.get('htmlContent');
                const dateRequest = store.get('versionDate');
                let htmlContent = null;
                let versionDate = null;
                let completed = 0;
                function checkComplete() {
                    completed++;
                    if (completed === 2) resolve({ htmlContent, versionDate });
                }
                htmlRequest.onsuccess = () => { htmlContent = htmlRequest.result; checkComplete(); };
                dateRequest.onsuccess = () => { versionDate = dateRequest.result; checkComplete(); };
                htmlRequest.onerror = () => checkComplete();
                dateRequest.onerror = () => checkComplete();
            });
        } catch (error) {
            return { htmlContent: null, versionDate: null };
        }
    }

    // ======================= EXTRAIRE DATE VERSION =======================
    async function getRemoteVersionDateForFile(fileName) {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === fileName) {
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    
                    if (dayMatch && timeMatch) {
                        return `${dayMatch[1]} ${timeMatch[1]}`;
                    }
                    
                    const altDateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (altDateMatch) {
                        return altDateMatch[1].trim();
                    }
                }
            }
            return null;
        } catch (error) {
            return null;
        }
    }

    // ======================= TÉLÉCHARGER CONTENU =======================
    async function telechargerContenu(fileName) {
        const fileUrl = `${WORKER_URL}/${fileName}`;
        const response = await fetch(fileUrl, { cache: 'no-cache' });
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Contenu invalide");
        return content;
    }

    // ======================= AFFICHER CONTENU =======================

    // ======================= CHARGER COURS (Débutant/Intermédiaire) =======================
    async function chargerCours(fileName, dbName, courseName) {
        if (isUpdating) return;
        isUpdating = true;
        
        try {
            loadingMessage.innerText = `😁 Ouvrir ${courseName}...`;
            loadingOverlay.classList.add('active');
            
            const cache = await recupererDepuisIndexedDB(dbName);
            const remoteVersionDate = await getRemoteVersionDateForFile(fileName);
            
            let contenu = null;
            
            if (cache.htmlContent && remoteVersionDate && cache.versionDate === remoteVersionDate) {
                contenu = cache.htmlContent;
                currentVersionDate = cache.versionDate;
            } else {
                contenu = await telechargerContenu(fileName);
                if (remoteVersionDate) {
                    await stockerDansIndexedDB(dbName, contenu, remoteVersionDate);
                    currentVersionDate = remoteVersionDate;
                }
            }
            
            currentDbName = dbName;
            afficherContenu(contenu);
            loadingOverlay.classList.remove('active');
            isUpdating = false;
            
        } catch (error) {
            loadingOverlay.classList.remove('active');
            isUpdating = false;
            alert(`❌ Erreur: ${error.message}`);
        }
    }

    // ======================= OUVRIR FACEBOOK (HIRESAKA) =======================
    function ouvrirFacebook() {
        // Supprimer le fichier hiredirect.txt de IndexedDB si existant
        const supprimerFichierHiresaka = async () => {
            try {
                // Vérifier si la DB hiredirect existe
                const db = await ouvrirDB('hiredirect');
                if (db) {
                    // Supprimer le contenu stocké
                    const transaction = db.transaction(['content'], 'readwrite');
                    const store = transaction.objectStore('content');
                    store.delete('htmlContent');
                    store.delete('versionDate');
                    transaction.oncomplete = () => console.log("Fichier HIRESAKA supprimé");
                }
            } catch (e) {
                console.log("Aucun fichier HIRESAKA à supprimer");
            }
        };
        
        supprimerFichierHiresaka();
        
        // Ouvrir Facebook
        window.open(FB_URL, '_blank');
    }

    // ======================= CHARGEMENT AUTO DÉBUTANT/INTERMÉDIAIRE =======================
    async function chargementAutoSilencieux() {
        for (const [fileName, config] of Object.entries(COURSES)) {
            try {
                const remoteVersionDate = await getRemoteVersionDateForFile(fileName);
                if (!remoteVersionDate) continue;
                
                const cache = await recupererDepuisIndexedDB(config.db);
                
                if (!cache.versionDate || cache.versionDate !== remoteVersionDate) {
                    const nouveauContenu = await telechargerContenu(fileName);
                    await stockerDansIndexedDB(config.db, nouveauContenu, remoteVersionDate);
                }
            } catch (error) {
                // Silencieux
            }
        }
    }

        // Boutons Débutant et Intermédiaire

        if (intermediaireBtn) {
            intermediaireBtn.addEventListener('click', () => {

        if (hiresakaBtn) {
            hiresakaBtn.addEventListener('click', (e) => {
                e.stopPropagation();
                ouvrirFacebook();
            });
        }
    }
