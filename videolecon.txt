    function filterVideos(query) {
        searchTerm = query.toLowerCase().trim();
        const cards = document.querySelectorAll('.media-card');
        let visibleCount = 0;
        let hasVisible = false;

        cards.forEach((card, index) => {
            if (index >= currentMediaList.length) return;
            const item = currentMediaList[index];
            const title = (item.titre || '').toLowerCase();
            const filename = (item.filename || '').toLowerCase();
            
            const match = !searchTerm || 
                         title.includes(searchTerm) || 
                         filename.includes(searchTerm);
            
            if (match) {
                card.classList.remove('hidden-card');
                visibleCount++;
                hasVisible = true;
            } else {
                card.classList.add('hidden-card');
            }
        });

            if (!noResult) {

            noResult.style.display = 'none';
        }

        currentFilteredList = currentMediaList.filter((item, index) => {
            const card = cards[index];
            return card && !card.classList.contains('hidden-card');
        });
    }

    // ---------- INDEXEDDB ----------
    async function openDB(galleryId) {
        return new Promise((resolve, reject) => {
            const req = indexedDB.open(`Gallery_${galleryId}`, 3);
            req.onerror = () => reject(req.error);
            req.onsuccess = (e) => resolve(e.target.result);
            req.onupgradeneeded = (e) => {
                const db = e.target.result;
                if (!db.objectStoreNames.contains('medias')) {
                    db.createObjectStore('medias', { keyPath: 'filename' });
                }
                if (!db.objectStoreNames.contains('metadata')) {
                    db.createObjectStore('metadata', { keyPath: 'key' });
                }
            };
        });
    }

    async function saveMedia(galleryId, filename, blob, numero, titre) {
        const db = await openDB(galleryId);
        return new Promise((resolve, reject) => {
            const tx = db.transaction(['medias'], 'readwrite');
            tx.objectStore('medias').put({ filename, blob, numero, titre });
            tx.oncomplete = resolve;
            tx.onerror = () => reject(tx.error);
        });
    }

            const req = db.transaction(['medias'], 'readonly').objectStore('medias').getAll();
            req.onsuccess = () => resolve(req.result || []);
            req.onerror = () => reject(req.error);
        });
    }

    async function getStoredNum(galleryId) {
        const db = await openDB(galleryId);
        return new Promise((resolve) => {
            const req = db.transaction(['metadata'], 'readonly').objectStore('metadata').get('numerook');
            req.onsuccess = () => resolve(req.result ? req.result.value : 0);
        });
    }

    async function setStoredNum(galleryId, value) {
        const db = await openDB(galleryId);
        return new Promise((resolve) => {
            const tx = db.transaction(['metadata'], 'readwrite');
            tx.objectStore('metadata').put({ key: 'numerook', value: value });
            tx.oncomplete = resolve;
        });
    }

    // ---------- MANIFEST ----------
    async function fetchManifest(gallery) {
        try {
            const resp = await fetch(gallery.manifest, { cache: 'no-cache' });
            if (!resp.ok) return [];
            const text = await resp.text();
            const lines = text.split(/\r?\n/).filter(l => l.trim() !== '');
            
            const map = new Map();
            let currentNum = null;
            let currentFilename = null;
            let currentTitre = '';
            const regex = new RegExp(`^${gallery.prefix}(\\d+)\\.([a-zA-Z0-9]+)$`);

            for (const line of lines) {
                const match = line.match(regex);
                if (match) {
                    if (currentNum !== null && currentFilename !== null) {
                        map.set(currentNum, { 
                            filename: currentFilename, 
                            numero: currentNum,
                            titre: currentTitre || `Vidéo ${currentNum}`
                        });
                    }
                    currentNum = parseInt(match[1]);
                    currentFilename = line;
                    currentTitre = '';
                } else if (currentNum !== null) {
                    const titreMatch = line.match(/^Titre\s*:\s*(.+)$/i);
                    if (titreMatch) {
                        currentTitre = titreMatch[1].trim();
                    }
                }
            }

            if (currentNum !== null && currentFilename !== null) {
                map.set(currentNum, { 
                    filename: currentFilename, 
                    numero: currentNum,
                    titre: currentTitre || `Vidéo ${currentNum}`
                });
            }

            return Array.from(map.values()).sort((a,b) => a.numero - b.numero);
        } catch(e) { 
            console.error('Erreur fetchManifest:', e);
            return []; 
        }
    }

    // ---------- SYNC ----------
    async function syncMedias(galleryId, manifest, localNum) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        if (!manifest.length) return 0;
        const maxRemote = manifest[manifest.length-1].numero;
        if (maxRemote <= localNum) return 0;
        const toDownload = manifest.filter(p => p.numero > localNum);
        let downloaded = 0;
        for (const media of toDownload) {
            const url = `${gallery.worker}${media.filename}`;
            try {
                const res = await fetch(url);
                if (res.ok) {
                    await saveMedia(galleryId, media.filename, await res.blob(), media.numero, media.titre);
                    downloaded++;
                }
            } catch(e) {}
        }
        if (downloaded > 0) await setStoredNum(galleryId, maxRemote);
        return downloaded;
    }

    // ---------- RENDER ----------
    async function renderGallery(galleryId) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        const stored = await getAllMedias(galleryId);
        if (!stored.length) {
            galleryContainer.innerHTML = `<div class="loader-area"><div class="spinner"></div></div>`;
            mediaCounterSpan.textContent = '0';
            return;
        }
        const sorted = stored.sort((a,b) => b.numero - a.numero);
        currentMediaList = sorted;
        currentFilteredList = sorted;
        mediaCounterSpan.textContent = sorted.length;

        let html = '';
        for (let i = 0; i < sorted.length; i++) {
            const item = sorted[i];
            const url = URL.createObjectURL(item.blob);
            const title = gallery.title.replace('{titre}', item.titre || `Vidéo ${item.numero}`);
            html += `
                <div class="media-card" data-index="${i}">
                    <div class="video-wrap">
                        <video src="${url}" muted playsinline preload="metadata"></video>
                        <div class="video-play-icon">▶</div>
                    </div>
                    <div class="media-title">${title}</div>
                </div>
            `;
        }
        galleryContainer.innerHTML = html;
        
        document.querySelectorAll('.media-card').forEach(card => {
            card.addEventListener('click', () => {
                const index = parseInt(card.dataset.index);
                const filteredIndex = currentFilteredList.findIndex(
                    item => item.filename === currentMediaList[index].filename
                );
                openLightbox(filteredIndex >= 0 ? filteredIndex : 0);
            });
        });

        if (searchTerm) {
            filterVideos(searchTerm);
        }
    }

    // ---------- LOAD ----------
    async function loadGallery(galleryId) {
        const gallery = GALLERIES.find(g => g.id === galleryId);
        currentGallery = galleryId;
        galleryTitle.textContent = gallery.screenTitle || gallery.label;
        gallerySub.textContent = gallery.screenSub || '';
        selectionScreen.style.opacity = '0';
        setTimeout(async () => {
            selectionScreen.style.display = 'none';
            galleryApp.classList.remove('hidden');
            
            await renderGallery(galleryId);
            
            const manifest = await fetchManifest(gallery);
            if (manifest.length) {
                const localNum = await getStoredNum(galleryId);
                const maxRemote = manifest[manifest.length-1].numero;
                if (maxRemote > localNum) {
                    await syncMedias(galleryId, manifest, localNum);
                    await renderGallery(galleryId);
                }
            }
        }, 300);
    }

    // ---------- LIGHTBOX ----------
    function openLightbox(index) {
        if (!currentFilteredList.length) return;
        currentIndex = index;
        const item = currentFilteredList[index];
        const url = URL.createObjectURL(item.blob);
        lbContent.innerHTML = `<video class="lb-video" src="${url}" controls autoplay></video>`;
        lightboxDiv.classList.add('active');
        lbCounter.textContent = `${index+1} / ${currentFilteredList.length}`;
    }
    
    function closeLightbox() { 
        lightboxDiv.classList.remove('active');
        const video = lbContent.querySelector('video');
        if (video) video.pause();
    }
    
    function nextMedia() {
        if (currentIndex < currentFilteredList.length - 1) {
            currentIndex++;
            const oldVideo = lbContent.querySelector('video');
            if (oldVideo) oldVideo.pause();
            const item = currentFilteredList[currentIndex];
            const url = URL.createObjectURL(item.blob);
            lbContent.innerHTML = `<video class="lb-video" src="${url}" controls autoplay></video>`;
            lbCounter.textContent = `${currentIndex+1} / ${currentFilteredList.length}`;
        }
    }
    
    function prevMedia() {
        if (currentIndex > 0) {
            currentIndex--;
            const oldVideo = lbContent.querySelector('video');
            if (oldVideo) oldVideo.pause();
            const item = currentFilteredList[currentIndex];
            const url = URL.createObjectURL(item.blob);
            lbContent.innerHTML = `<video class="lb-video" src="${url}" controls autoplay></video>`;
            lbCounter.textContent = `${currentIndex+1} / ${currentFilteredList.length}`;
        }
    }
    
    closeLb.addEventListener('click', closeLightbox);
    prevLb.addEventListener('click', prevMedia);
    nextLb.addEventListener('click', nextMedia);
    lightboxDiv.addEventListener('click', (e) => { if (e.target === lightboxDiv) closeLightbox(); });
    document.addEventListener('keydown', (e) => {
        if (!lightboxDiv.classList.contains('active')) return;
        if (e.key === 'Escape') closeLightbox();
        if (e.key === 'ArrowRight') nextMedia();
        if (e.key === 'ArrowLeft') prevMedia();
    });
