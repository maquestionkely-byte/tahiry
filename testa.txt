<script>
    const questions = [
        {
            text: "Julie _____ 10 ans.",
            options: ["est", "a", "suis", "ont"],
            correct: "a",
            explanation: "Avoir + âge : Julie a 10 ans."
        },
        {
            text: "Nous _____ faim.",
            options: ["sommes", "ont", "avons", "est"],
            correct: "avons",
            explanation: "Avoir faim : Nous avons faim."
        },
        {
            text: "Ils _____ peur du noir.",
            options: ["sont", "ont", "avez", "est"],
            correct: "ont",
            explanation: "Avoir peur : Ils ont peur du noir."
        },
        {
            text: "Tu _____ raison.",
            options: ["es", "as", "a", "suis"],
            correct: "as",
            explanation: "Avoir raison : Tu as raison."
        },
        {
            text: "Je _____ fatigué.",
            options: ["ai", "est", "suis", "sont"],
            correct: "suis",
            explanation: "Être fatigué : Je suis fatigué."
        }
    ];

    let userAnswers = new Array(5).fill(null);

    function renderQuestions() {
        const container = document.getElementById('questionsContainer');
        container.innerHTML = '';
        questions.forEach((q, idx) => {
            const card = document.createElement('div');
            card.className = 'question-card';
            card.innerHTML = `
                <div class="question-text">${idx+1}. ${q.text}</div>
                ${q.options.map(opt => `
                    <div class="option" data-qidx="${idx}" data-opt="${opt}">
                        ${opt}
                    </div>
                `).join('')}
                <div class="feedback" id="fb-${idx}"></div>
            `;
            container.appendChild(card);
        });

        document.querySelectorAll('.option').forEach(optDiv => {
            optDiv.addEventListener('click', (e) => {
                const qidx = parseInt(optDiv.dataset.qidx);
                const selected = optDiv.dataset.opt;
                userAnswers[qidx] = selected;
                updateSelectionUI(qidx, selected);
                // Afficher feedback immédiat
                const fbDiv = document.getElementById(`fb-${qidx}`);
                const isCorrect = (selected === questions[qidx].correct);
                if (isCorrect) {
                    fbDiv.innerHTML = "✅ Bonne réponse !";
                    fbDiv.className = "feedback";
                } else {
                    fbDiv.innerHTML = `❌ Faux. Bonne réponse : ${questions[qidx].correct}`;
                    fbDiv.className = "feedback wrong";
                }
            });
        });
    }

    function updateSelectionUI(qidx, selected) {
        const card = document.querySelectorAll('.question-card')[qidx];
        const options = card.querySelectorAll('.option');
        options.forEach(opt => {
            if (opt.dataset.opt === selected) {
                opt.classList.add('selected');
            } else {
                opt.classList.remove('selected');
            }
        });
    }

    function showCorrection() {
        let score = 0;
        for (let i = 0; i < questions.length; i++) {
            if (userAnswers[i] === questions[i].correct) score++;
        }
        document.getElementById('scoreDiv').innerHTML = `🎯 Votre score : ${score} / 5 (${score*20}%)`;

        let correctionHtml = `<h3>📋 Corrigé détaillé</h3>`;
        questions.forEach((q, i) => {
            const user = userAnswers[i] || "(non répondue)";
            const isOk = (user === q.correct);
            correctionHtml += `
                <p><strong>${i+1}. ${q.text}</strong><br>
                Votre réponse : ${user}<br>
                ✅ Bonne réponse : ${q.correct}<br>
                📖 Explication : ${q.explanation}<br>
                ${isOk ? '✔️ Correct' : '❌ Erreur'}</p>
                <hr>
            `;
        });
        document.getElementById('correctionDiv').innerHTML = correctionHtml;
        document.getElementById('correctionDiv').style.display = 'block';
    }

    renderQuestions();
    document.getElementById('corrigeBtn').addEventListener('click', showCorrection);
</script>
