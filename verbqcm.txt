    // ==================== INDEXEDDB ====================
    function ouvrirDB() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(DB_NAME, 2);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains(STORE_NAME)) {
                    db.createObjectStore(STORE_NAME);
                    console.log("✅ Base IndexedDB créée");
                }
            };
        });
    }
    
    async function sauvegarderDansCache(levelId, contenu, versionDate) {
        try {
            const db = await ouvrirDB();
            return new Promise((resolve, reject) => {
                const transaction = db.transaction([STORE_NAME], 'readwrite');
                const store = transaction.objectStore(STORE_NAME);
                store.put(contenu, `content_${levelId}`);
                store.put(versionDate, `version_${levelId}`);
                transaction.oncomplete = () => {
                    console.log(`💾 Niveau ${levelId} sauvegardé`);
                    resolve();
                };
                transaction.onerror = () => reject(transaction.error);
            });
        } catch (error) {
            console.error("Erreur sauvegarde:", error);
            throw error;
        }
    }
    
    async function recupererDepuisCache(levelId) {
        try {
            const db = await ouvrirDB();
            return new Promise((resolve) => {
                const transaction = db.transaction([STORE_NAME], 'readonly');
                const store = transaction.objectStore(STORE_NAME);
                const requestContent = store.get(`content_${levelId}`);
                const requestVersion = store.get(`version_${levelId}`);
                
                let content = null;
                let version = null;
                let completed = 0;
                
                function checkComplete() {
                    completed++;
                    if (completed === 2) {
                        resolve({ content, version });
                    }
                }
                
                requestContent.onsuccess = () => { content = requestContent.result; checkComplete(); };
                requestVersion.onsuccess = () => { version = requestVersion.result; checkComplete(); };
                requestContent.onerror = () => checkComplete();
                requestVersion.onerror = () => checkComplete();
            });
        } catch (error) {
            console.error("Erreur lecture cache:", error);
            return { content: null, version: null };
        }
    }
    
    async function verifierNiveauEnCache(levelId) {
        try {
            const db = await ouvrirDB();
            return new Promise((resolve) => {
                const transaction = db.transaction([STORE_NAME], 'readonly');
                const store = transaction.objectStore(STORE_NAME);
                const request = store.get(`content_${levelId}`);
                request.onsuccess = () => {
                    resolve(request.result !== undefined && request.result !== null);
                };
                request.onerror = () => resolve(false);
            });
        } catch {
            return false;
        }
    }
    
    // ==================== EXTRAIRE LA DATE DE VERSION ====================
    async function getRemoteVersionDate(fileName) {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) return null;
            const content = await response.text();
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === fileName) {
                    const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (dateMatch) return dateMatch[1].trim();
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
                }
            }
            return null;
        } catch (error) {
            console.warn("Impossible de récupérer la version distante:", error);
            return null;
        }
    }
    
    // ==================== PARSEUR DU FICHIER TEXTE ====================
    function parseTextFile(content) {
        const lines = content.split('\n');
        let title = "QCM - Français";
        const questionsList = [];
        let currentQuestion = null;
        let currentOptions = [];
        let currentCorrect = null;
        let currentTranslation = null;
        
        for (let i = 0; i < lines.length; i++) {
            let line = lines[i].trim();
            if (line === "") continue;
            
            if (line.startsWith('TITRE:')) {
                title = line.replace('TITRE:', '').trim();
                continue;
            }
            
            const questionMatch = line.match(/^(\d+)\.\s+(.+)$/);
            if (questionMatch) {
                if (currentQuestion !== null && currentOptions.length > 0) {
                    questionsList.push({
                        q: currentQuestion,
                        options: [...currentOptions],
                        correct: currentCorrect,
                        mg: currentTranslation || ""
                    });
                }
                currentQuestion = questionMatch[2];
                currentOptions = [];
                currentCorrect = null;
                currentTranslation = null;
                continue;
            }
            
            const starOptionMatch = line.match(/^\*([A-D])\)\s+(.+)$/);
            if (starOptionMatch && currentQuestion !== null) {
                currentCorrect = starOptionMatch[1];
                currentOptions.push(starOptionMatch[2].trim());
                continue;
            }
            
            const normalOptionMatch = line.match(/^([A-D])\)\s+(.+)$/);
            if (normalOptionMatch && currentQuestion !== null) {
                currentOptions.push(normalOptionMatch[2].trim());
                continue;
            }
            
            if (line.startsWith('TRADUCTION:') && currentQuestion !== null) {
                currentTranslation = line.replace('TRADUCTION:', '').trim();
                continue;
            }
        }
        
        if (currentQuestion !== null && currentOptions.length > 0) {
            questionsList.push({
                q: currentQuestion,
                options: [...currentOptions],
                correct: currentCorrect,
                mg: currentTranslation || ""
            });
        }
        
        return { title, questions: questionsList };
    }
    
    // ==================== TÉLÉCHARGER LE FICHIER ====================
    async function telechargerFichier(fileName) {
        const url = BASE_URL + fileName;
        const response = await fetch(url, { cache: 'no-cache' });
        if (!response.ok) throw new Error(`HTTP ${response.status} - Fichier ${fileName} non trouvé`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Fichier vide ou trop court");
        return content;
    }
    
    // ==================== CHARGER UN NIVEAU ====================
    async function loadLevel(level, forceDownload = false) {
        appContainer.innerHTML = `
            <div class="loading-container" id="loadingDiv">
                <div class="spinner"></div>
                <h2>Chargement du ${level.display}</h2>
                <p id="loadingMessage">Vérification du cache...</p>
            </div>
        `;
        
        const loadingMessage = document.getElementById('loadingMessage');
        
        try {
            const remoteVersion = await getRemoteVersionDate(level.file);
            const cached = await recupererDepuisCache(level.id);
            let content = null;
            let needUpdate = forceDownload;
            
            if (cached.content && !forceDownload) {
                content = cached.content;
                currentVersionDate = cached.version;
                
                if (remoteVersion && cached.version !== remoteVersion) {
                    needUpdate = true;
                    if (loadingMessage) loadingMessage.innerText = 'Nouvelle version disponible...';
                }
            }
            
            if (needUpdate || !content) {
                if (loadingMessage) loadingMessage.innerText = `Téléchargement...`;
                content = await telechargerFichier(level.file);
                const newVersion = remoteVersion || new Date().toLocaleString('fr-FR');
                await sauvegarderDansCache(level.id, content, newVersion);
                currentVersionDate = newVersion;
            }
            
            if (loadingMessage) loadingMessage.innerText = 'Analyse du fichier...';
            const parsed = parseTextFile(content);
            questions = parsed.questions;
            quizTitle = parsed.title;
            
            if (!questions || questions.length === 0) {
                throw new Error("Aucune question trouvée");
            }
            
            afficherQuiz(level);
            
        } catch (error) {
            appContainer.innerHTML = `
                <div class="quiz-container">
                    <div class="quiz-header">
                        <h1>❌ Erreur</h1>
                        <button class="back-btn" onclick="location.reload()">🏠 Accueil</button>
                    </div>
                    <div class="question-area">
                        <div class="text-center" style="padding: 2rem;">
                            <p style="color: #c0392b; margin-bottom: 20px;">${error.message}</p>
                            <button onclick="location.reload()" style="background: #ffb347; border: none; padding: 12px 25px; border-radius: 60px; cursor: pointer;">🔄 Réessayer</button>
                        </div>
                    </div>
                </div>
            `;
        }
    }

        const liveScore = document.getElementById('liveScore');
        if (qCounter) qCounter.innerText = `Question ${currentIdx+1} / ${questions.length}`;
        if (liveScore) liveScore.innerText = `🎯 Score : ${currentScore} / ${questions.length}`;
    }
    
    function handleAnswer(selectedLetter) {
        if (isLocked[currentIdx] || quizFinished) return;
        isLocked[currentIdx] = true;
        const q = questions[currentIdx];
        if (selectedLetter === q.correct) currentScore++;
        userSelections[currentIdx] = selectedLetter;
        renderCurrent();
        setTimeout(() => {
            if (currentIdx + 1 < questions.length) {
                currentIdx++;
                renderCurrent();
            } else {
                finishQuiz();
            }
        }, 380);
    }
    
    function finishQuiz() {
        quizFinished = true;
        const totalCorrect = currentScore;
        const percent = Math.round((totalCorrect / questions.length) * 100);
        
        let listHtml = `<div class="final-report">
            <div class="final-header">📊 RÉSULTAT FINAL</div>
            <div class="text-center"><span class="final-score">${totalCorrect} / ${questions.length} (${percent}%)</span></div>
            ${currentVersionDate ? `<div class="text-center" style="font-size: 0.7rem; color: #888; margin-top: 5px;">📅 Version: ${currentVersionDate}</div>` : ''}
            <div class="questions-list">
                <h3>📋 Corrigé complet :</h3>`;
        
        for (let i = 0; i < questions.length; i++) {
            const q = questions[i];
            const userChoice = userSelections[i];
            const lettersIndex = ['A','B','C','D'];
            const isOk = (userChoice === q.correct);
            const userText = userChoice ? q.options[lettersIndex.indexOf(userChoice)] : "(non répondue)";
            const correctAnswerText = q.options[lettersIndex.indexOf(q.correct)];
            const borderClass = isOk ? "correct" : "wrong";
            listHtml += `
                <div class="question-summary ${borderClass}">
                    <strong>${isOk ? "✅" : "❌"} Q${i+1} : ${q.q}</strong><br>
                    <span>✏️ Votre réponse : ${userChoice || "—"} → "${userText}"</span><br>
                    <span>📌 Bonne réponse : ${q.correct} → "${correctAnswerText}"</span>
                    ${q.mg ? `<br><span style="font-size:0.8rem;">🇲🇬 Malagasy: ${q.mg}</span>` : ''}
                </div>
            `;
        }
        listHtml += `<button class="restart-btn" id="restartQuizBtn">🔄 Recommencer le quiz</button>`;
        listHtml += `</div></div>`;
        
        const quizContent = document.getElementById('quizContent');
        if (quizContent) {
            quizContent.innerHTML = `<div class="progress-bar"><span class="badge">Terminé !</span><span class="badge score-badge">🎯 Score final : ${totalCorrect} / ${questions.length}</span></div>${listHtml}`;
        }
        
        const restartBtn = document.getElementById('restartQuizBtn');
        if (restartBtn) restartBtn.addEventListener('click', () => resetQuiz());
    }
    
    function resetQuiz() {
        currentIdx = 0;
        userSelections = new Array(questions.length).fill(null);
        isLocked = new Array(questions.length).fill(false);
        currentScore = 0;
        quizFinished = false;
        renderCurrent();
    }

    async function showLevelsScreen() {
        // Vérifier quels niveaux sont en cache
        const cacheStatus = {};
        for (const level of LEVELS) {
            cacheStatus[level.id] = await verifierNiveauEnCache(level.id);
        }
        
        let levelsHtml = `
            <div class="levels-container">
                <div class="levels-header">
                    <h1>📚 Choisissez votre niveau</h1>
                    <p>Sélectionnez un niveau pour commencer le QCM</p>
                    <button class="refresh-btn" id="refreshCacheBtn">🔄 Vérifier les niveaux téléchargés</button>
                </div>
                <div class="levels-grid">
        `;
        
        LEVELS.forEach(level => {
            const isCached = cacheStatus[level.id];
            const cachedClass = isCached ? 'cached' : '';
            const cachedBadge = isCached ? '<div class="cached-badge">✓ Téléchargé</div>' : '';
            
            levelsHtml += `
                <div class="level-card ${cachedClass}" data-level-id="${level.id}" data-level-file="${level.file}" data-level-name="${level.display}">
                    <div class="level-number">${level.display}</div>
                    ${cachedBadge}
                    <div class="level-badge">📖 Commencer</div>
                </div>
            `;
        });
        
        levelsHtml += `
                </div>
                <footer>📁 Les niveaux avec bordure verte sont déjà téléchargés</footer>
            </div>
        `;
        
        appContainer.innerHTML = levelsHtml;
        
        // Ajouter les événements de clic
        document.querySelectorAll('.level-card').forEach(card => {
            card.addEventListener('click', () => {
                const levelId = parseInt(card.dataset.levelId);
                const level = LEVELS.find(l => l.id === levelId);
                if (level) {
                    loadLevel(level);
                }
            });
        });
        
        // Bouton rafraîchir
        const refreshBtn = document.getElementById('refreshCacheBtn');
        if (refreshBtn) {
            refreshBtn.addEventListener('click', () => showLevelsScreen());
        }
    }
    
    // ==================== LANCEMENT ====================
    showLevelsScreen();
