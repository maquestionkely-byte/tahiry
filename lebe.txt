    // ======================= FONCTIONS INDEXEDDB =======================
    function ouvrirDB(nomDB) {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(nomDB, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains('contenu')) db.createObjectStore('contenu');
                if (!db.objectStoreNames.contains('version')) db.createObjectStore('version');
            };
        });
    }
    
    async function stockerContenuEtVersion(nomDB, contenu, version) {
        const db = await ouvrirDB(nomDB);
        return new Promise((resolve, reject) => {
            const transaction = db.transaction(['contenu', 'version'], 'readwrite');
            transaction.objectStore('contenu').put(contenu, 'htmlContent');
            transaction.objectStore('version').put(version, 'versionDate');
            transaction.oncomplete = () => resolve();
            transaction.onerror = () => reject(transaction.error);
        });
    }
    
    async function recupererVersion(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['version'], 'readonly');
                const request = transaction.objectStore('version').get('versionDate');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch { return null; }
    }
    
    async function recupererContenu(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['contenu'], 'readonly');
                const request = transaction.objectStore('contenu').get('htmlContent');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch { return null; }
    }
    
    async function telechargerFichier(nomFichier) {
        const url = `${WORKER_URL}/${nomFichier}`;
        const response = await fetch(url, { cache: 'no-cache' });
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Contenu invalide");
        return content;
    }
    
    async function getRemoteVersionDate(nomFichier) {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === nomFichier) {
                    const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (dateMatch) return dateMatch[1].trim();
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
                }
            }
            return null;
        } catch { return null; }
    }

    }

    async function verifierTousLesFichiers() {
        console.log("🚀 Début de la vérification automatique...");
        const tousLesFichiers = genererTousLesFichiersAVerifier();
        let verifies = 0, misAJour = 0, erreurs = 0;
        
        for (const fichier of tousLesFichiers) {
            try {
                const remoteVersion = await getRemoteVersionDate(fichier.nom);
                if (!remoteVersion) continue;
                
                const storedVersion = await recupererVersion(fichier.dbName);
                
                if (storedVersion && storedVersion === remoteVersion) {
                    verifies++;
                } else {
                    console.log(`🆕 Mise à jour: ${fichier.nom}`);
                    const nouveauContenu = await telechargerFichier(fichier.nom);
                    await stockerContenuEtVersion(fichier.dbName, nouveauContenu, remoteVersion);
                    misAJour++;
                    verifies++;
                }
                await new Promise(r => setTimeout(r, 100));
            } catch (error) {
                erreurs++;
            }
        }
        
        console.log(`✅ Vérifiés: ${verifies}, Mis à jour: ${misAJour}, Erreurs: ${erreurs}`);
        if (misAJour > 0) showUpdateNotification(`${misAJour} fichier(s) mis à jour`);
    }
    
    function showUpdateNotification(message, isError = false) {

        notif.innerHTML = isError ? `⚠️ ${message}` : `✅ ${message}`;
        document.body.appendChild(notif);
        setTimeout(() => notif.remove(), 5000);
    }

    async function telechargerEtOuvrirExercice(leconTitre, typeCode, niveau) {
        const codeLecon = getCodeLecon(leconTitre);
        const nomFichier = `${codeLecon}${typeCode}.txt`;
        const url = `${WORKER_URL}/${nomFichier}`;
        
        const modal = document.createElement('div');
        modal.className = 'modal-fullscreen';
        modal.innerHTML = `
            <div class="modal-fullscreen-header"><h2>📥 Chargement</h2><button class="btn-close-fullscreen" id="closeLoadingBtn">✕</button></div>
            <div class="modal-fullscreen-body" style="display:flex;align-items:center;justify-content:center;"><p>Chargement de ${nomFichier}...</p></div>
        `;
        document.body.appendChild(modal);
        document.getElementById('closeLoadingBtn').addEventListener('click', () => modal.remove());
        
        try {
            let contenu = await recupererContenu(nomFichier.replace('.txt', ''));
            if (!contenu) {
                const response = await fetch(url);
                if (!response.ok) throw new Error();
                contenu = await response.text();
                await stockerEnIndexedDB(nomFichier, contenu);
            }
            modal.remove();
            const blob = new Blob([contenu], { type: 'text/html' });
            const blobUrl = URL.createObjectURL(blob);
            sessionStorage.setItem('exerciceInfo', JSON.stringify({ leconTitre: leconTitre, typeCode: typeCode, niveau: niveau }));
            window.location.href = blobUrl;
        } catch (error) {
            modal.remove();
            alert(`❌ Fichier non trouvé: ${nomFichier}\nCréez ce fichier sur le serveur.`);
        }
    }

    // ======================= IMAGES =======================
    async function stockerImage(url, blob) {
        return new Promise((resolve) => {
            const request = indexedDB.open('image_cache', 7);
            request.onupgradeneeded = (e) => { if (!e.target.result.objectStoreNames.contains('images')) e.target.result.createObjectStore('images'); };
            request.onsuccess = (e) => {
                const db = e.target.result;
                const tx = db.transaction('images', 'readwrite');
                tx.objectStore('images').put(blob, url);
                tx.oncomplete = () => { db.close(); resolve(); };
            };
        });
    }
    
    async function chargerImage(element, url) {
        const request = indexedDB.open('image_cache', 7);
        request.onsuccess = async (e) => {
            const db = e.target.result;
            if (db.objectStoreNames.contains('images')) {
                const tx = db.transaction('images', 'readonly');
                const get = tx.objectStore('images').get(url);
                get.onsuccess = () => {
                    if (get.result) { element.src = URL.createObjectURL(get.result); db.close(); return; }
                    db.close();
                    fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
                };
            } else {
                db.close();
                fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
            }
        };
    }
