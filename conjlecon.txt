// ========== VARIABLES GLOBALES ==========
let verbesOriginaux = [];
let verbesFiltres = [];
let currentVerb = "";
let tempsData = {};
let isListVisible = false;

// ========== CONFIGURATION ==========
const FILE_URL = 'https://passage.ahofaly62.workers.dev/hireverbes.txt';
const DB_NAME = 'verbes_cache';
const STORE_NAME = 'verbes_contenus';
const CACHE_KEY = 'verbes_content';

// ========== FONCTIONS CACHE ==========
function ouvrirDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, 1);
        request.onerror = () => reject(request.error);
        request.onsuccess = (event) => resolve(event.target.result);
        request.onupgradeneeded = (event) => {
            const db = event.target.result;
            if (!db.objectStoreNames.contains(STORE_NAME)) {
                db.createObjectStore(STORE_NAME);
                console.log("✅ Base IndexedDB créée");
            }
        };
    });
}

async function sauvegarderDansCache(contenu) {
    try {
        const db = await ouvrirDB();
        return new Promise((resolve, reject) => {
            const transaction = db.transaction([STORE_NAME], 'readwrite');
            const store = transaction.objectStore(STORE_NAME);
            const request = store.put(contenu, CACHE_KEY);
            request.onsuccess = () => { console.log("💾 Sauvegardé"); resolve(); };
            request.onerror = () => reject(request.error);
        });
    } catch (error) {
        console.error("Erreur sauvegarde:", error);
    }
}

async function recupererDepuisCache() {
    try {
        const db = await ouvrirDB();
        return new Promise((resolve) => {
            const transaction = db.transaction([STORE_NAME], 'readonly');
            const store = transaction.objectStore(STORE_NAME);
            const request = store.get(CACHE_KEY);
            request.onsuccess = () => resolve(request.result);
            request.onerror = () => resolve(null);
        });
    } catch (error) {
        console.error("Erreur lecture cache:", error);
        return null;
    }
}

// ========== DÉTECTION GROUPE ==========
function detecterGroupe(verbe) {
    const v = verbe.toLowerCase();
    const verbesIrreguliers = ["être", "avoir", "aller", "faire", "dire", "prendre", "venir", "voir", "vouloir", "pouvoir", "devoir", "savoir", "connaître", "mettre", "croire", "tenir", "suivre", "partir", "sortir", "dormir", "servir", "sentir", "mentir", "courir", "mourir", "battre", "rompre", "conduire", "produire", "traduire", "craindre", "peindre", "joindre"];
    if (verbesIrreguliers.includes(v)) return 3;
    if (v === "aller") return 3;
    if (v.endsWith("er")) return 1;
    if (v.endsWith("ir")) {
        const radical = v.slice(0, -2);
        const verbes2eGroupe = ["fin", "chois", "grand", "réuss", "bât", "rempl", "pâl", "jaun", "roug", "nourr", "agir", "réfléchir", "obéir", "définir", "punir"];
        if (verbes2eGroupe.some(g => radical === g || radical.endsWith(g))) return 2;
        return 3;
    }
    return 3;
}

function getGroupeText(groupe) {
    if (groupe === 1) return "📚 1er groupe (-er)";
    if (groupe === 2) return "📚 2ème groupe (-ir)";
    return "📚 3ème groupe (irrégulier)";
}

// ========== CHARGEMENT DES VERBES ==========
async function chargerVerbes() {
    const container = document.getElementById('verbButtonsContainer');
    container.innerHTML = '<div class="loading-message">⏳ Chargement des verbes...</div>';
    
    try {
        let cached = await recupererDepuisCache();
        
        if (cached && cached.length > 100) {
            console.log("✅ Utilisation du cache");
            const verbes = traiterContenuVerbes(cached);
            verbesOriginaux = verbes;
            verbesFiltres = [...verbes];
            renderVerbList();
            return;
        }
        
        console.log("🌐 Chargement depuis le réseau...");
        const response = await fetch(FILE_URL);
        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const text = await response.text();
        
        if (text && text.length > 100) {
            await sauvegarderDansCache(text);
            const verbes = traiterContenuVerbes(text);
            verbesOriginaux = verbes;
            verbesFiltres = [...verbes];
            renderVerbList();
        } else {
            throw new Error("Fichier vide ou invalide");
        }
        
    } catch (error) {
        console.error("Erreur:", error);
        container.innerHTML = `<div class="empty-message">❌ Erreur: ${error.message}</div>`;
    }
}

function traiterContenuVerbes(text) {
    const lines = text.split('\n');
    const verbes = [];
    for (let line of lines) {
        line = line.trim().toLowerCase();
        if (line && !line.includes('titre') && !line.includes(':') && line.length > 2) {
            verbes.push({ name: line, groupe: detecterGroupe(line) });
        }
    }
    console.log(`📚 ${verbes.length} verbes chargés`);
    return verbes;
}

function renderVerbList() {
    const container = document.getElementById('verbButtonsContainer');
    const resultsCount = document.getElementById('resultsCount');
    
    if (verbesFiltres.length === 0) {
        container.innerHTML = '<div class="empty-message">🔍 Aucun verbe trouvé.</div>';
        resultsCount.textContent = '';
        return;
    }
    
    resultsCount.textContent = `${verbesFiltres.length} verbe${verbesFiltres.length > 1 ? 's' : ''}`;
    container.innerHTML = '';
    
    for (const verbe of verbesFiltres) {
        const div = document.createElement('div');
        div.className = 'verb-item';
        const groupeClass = `groupe-${verbe.groupe}`;
        const groupeText = verbe.groupe === 1 ? "1er Grp" : (verbe.groupe === 2 ? "2ème Grp" : "3ème Grp");
        div.innerHTML = `<span class="verb-name">${verbe.name}</span><span class="verb-group-badge ${groupeClass}">${groupeText}</span>`;
        div.addEventListener('click', () => openConjugationScreen(verbe.name));
        container.appendChild(div);
    }
}

function filterVerbes(searchTerm) {
    const term = searchTerm.toLowerCase().trim();
    verbesFiltres = term === '' ? [...verbesOriginaux] : verbesOriginaux.filter(verbe => verbe.name.includes(term));
    renderVerbList();
}

// ========== SYSTÈME D'AUXILIAIRE INTELLIGENT ==========
function conjuguerPasseCompose(verbe, sujet, contexte = {}) {
    const v = verbe.toLowerCase().trim();
    const pronom = sujet.pronom || sujet;
    const genre = sujet.genre || (pronom === "elle" || pronom === "elles" ? "f" : "m");
    const nombre = sujet.nombre || (pronom === "je" || pronom === "tu" || pronom === "il" || pronom === "elle" ? "s" : "p");
    const { aCOD = false, codAvant = null, estPronominal = false, estImpersonnel = false } = contexte;
    
    const verbesEtre = new Set([
        "aller", "venir", "arriver", "entrer", "rester", "tomber", "partir", "sortir",
        "monter", "descendre", "retourner", "rentrer", "passer", "revenir", "devenir",
        "parvenir", "naître", "mourir", "décéder", "apparaître", "disparaître",
        "advenir", "circonvenir", "contrevenir", "convenir", "intervenir", "survenir"
    ]);
    
    const verbesDoubleAux = new Set([
        "monter", "descendre", "sortir", "rentrer", "retourner", "passer", "entrer"
    ]);
    
    const participes = {
        "être": "été", "avoir": "eu", "prendre": "pris", "mettre": "mis", "dire": "dit",
        "écrire": "écrit", "faire": "fait", "voir": "vu", "vivre": "vécu", "naître": "né",
        "mourir": "mort", "ouvrir": "ouvert", "offrir": "offert", "courir": "couru"
    };
    
    let auxiliaire;
    if (estImpersonnel || v === "pleuvoir" || v === "neiger" || v === "falloir") {
        auxiliaire = "avoir";
    } else if (estPronominal || v.startsWith("se ")) {
        auxiliaire = "être";
    } else if (verbesEtre.has(v)) {
        auxiliaire = "être";
    } else if (verbesDoubleAux.has(v)) {
        auxiliaire = aCOD ? "avoir" : "être";
    } else {
        auxiliaire = "avoir";
    }
    
    let participe = participes[v];
    if (!participe) {
        if (v.endsWith("er")) participe = v.slice(0, -2) + "é";
        else if (v.endsWith("ir")) participe = v.slice(0, -2) + "i";
        else if (v.endsWith("re")) participe = v.slice(0, -2) + "u";
        else participe = v;
    }
    
    if (v === "mourir" && genre === "f") participe = "morte";
    if (v === "naître" && genre === "f") participe = "née";
    
    let participeAccorde = participe;
    if (auxiliaire === "être") {
        if (genre === "f" && nombre === "s") participeAccorde = participe + "e";
        else if (genre === "f" && nombre === "p") participeAccorde = participe + "es";
        else if (genre === "m" && nombre === "p") participeAccorde = participe + "s";
    } else if (auxiliaire === "avoir" && codAvant && codAvant.nature === "COD") {
        if (codAvant.genre === "f" && codAvant.nombre === "s") participeAccorde = participe + "e";
        else if (codAvant.genre === "f" && codAvant.nombre === "p") participeAccorde = participe + "es";
        else if (codAvant.genre === "m" && codAvant.nombre === "p") participeAccorde = participe + "s";
    }
    
  const conjugaisons = {
    avoir: { 
        "je": "ai",    // ← déjà contracté !
        "tu": "as", 
        "il": "a", 
        "elle": "a", 
        "nous": "avons", 
        "vous": "avez", 
        "ils": "ont", 
        "elles": "ont" 
    },
    être: { 
        "je": "suis",
        "tu": "es", 
        "il": "est", 
        "elle": "est", 
        "nous": "sommes", 
        "vous": "êtes", 
        "ils": "sont", 
        "elles": "sont" 
    }
};
    
    let auxConj = conjugaisons[auxiliaire][pronom];
    if (!auxConj) auxConj = `${pronom} ${auxiliaire}`;
    
    return { conjugaison: `${auxConj} ${participeAccorde}`, auxiliaire, participe: participeAccorde };
}

}

    }
    if (pronom === "que je" && verbe && /^[aeiouyâêîôûàèìòùéèëïü]/i.test(verbe)) {
        return "que j'";
    }
    return pronom;
}

// ========== FONCTIONS DE COLORATION ==========
function colorEnding(verbForm, tenseId, genre = null, nombre = null) {
    if (!verbForm) return verbForm;
    
    // 🟢 PASSÉ COMPOSÉ - Colorer les terminaisons du participe passé
    if (tenseId === 'passe_compose') {
        const isFeminine = (genre === 'f');
        const isPlural = (nombre === 'p');
        
        // === PARTICIPE EN -é (manger, parler) ===
        if (verbForm.endsWith('é')) {
            let base = verbForm.slice(0, -1);
            if (isFeminine && !isPlural) {
                // elle → mangée : é(rouge) + e(bleu)
                return base + `<span class="red-ending">é</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → mangées : é(rouge) + es(bleu)
                return base + `<span class="red-ending">é</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → mangés : é(rouge) + s(rouge)
                return base + `<span class="red-ending">é</span><span class="red-ending">s</span>`;
            }
            // il → mangé
            return base + `<span class="red-ending">é</span>`;
        }
        
        // === PARTICIPE EN -i (sortir, finir) ===
        if (verbForm.endsWith('i') && !verbForm.endsWith('ie') && !verbForm.endsWith('is')) {
            let base = verbForm.slice(0, -1);
            if (isFeminine && !isPlural) {
                // elle → sortie : i(rouge) + e(bleu)
                return base + `<span class="red-ending">i</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → sorties : i(rouge) + es(bleu)
                return base + `<span class="red-ending">i</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → sortis : i(rouge) + s(rouge)
                return base + `<span class="red-ending">i</span><span class="red-ending">s</span>`;
            }
            // il → sorti
            return base + `<span class="red-ending">i</span>`;
        }
        
        // === PARTICIPE EN -u (voir, pouvoir, vouloir) ===
        if (verbForm.endsWith('u') && !verbForm.endsWith('ue') && !verbForm.endsWith('us')) {
            let base = verbForm.slice(0, -1);
            if (isFeminine && !isPlural) {
                // elle → vue : u(rouge) + e(bleu)
                return base + `<span class="red-ending">u</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → vues : u(rouge) + es(bleu)
                return base + `<span class="red-ending">u</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → vus : u(rouge) + s(rouge)
                return base + `<span class="red-ending">u</span><span class="red-ending">s</span>`;
            }
            // il → vu
            return base + `<span class="red-ending">u</span>`;
        }
        
        // === PARTICIPE EN -is (prendre, mettre) ===
        if (verbForm.endsWith('is') && !verbForm.endsWith('ise')) {
            let base = verbForm.slice(0, -2);
            if (isFeminine && !isPlural) {
                // elle → prise : is(rouge) + e(bleu)
                return base + `<span class="red-ending">is</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → prises : is(rouge) + es(bleu)
                return base + `<span class="red-ending">is</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → pris : is(rouge) + s(rouge)
                return base + `<span class="red-ending">is</span><span class="red-ending">s</span>`;
            }
            // il → pris
            return base + `<span class="red-ending">is</span>`;
        }
        
        // === PARTICIPE EN -it (dire, écrire) ===
        if (verbForm.endsWith('it') && !verbForm.endsWith('ite')) {
            let base = verbForm.slice(0, -2);
            if (isFeminine && !isPlural) {
                // elle → dite : it(rouge) + e(bleu)
                return base + `<span class="red-ending">it</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → dites : it(rouge) + es(bleu)
                return base + `<span class="red-ending">it</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → dits : it(rouge) + s(rouge)
                return base + `<span class="red-ending">it</span><span class="red-ending">s</span>`;
            }
            // il → dit
            return base + `<span class="red-ending">it</span>`;
        }
        
        // === PARTICIPE EN -ert (ouvrir, couvrir) ===
        if (verbForm.endsWith('ert')) {
            let base = verbForm.slice(0, -3);
            if (isFeminine && !isPlural) {
                // elle → ouverte : ert(rouge) + e(bleu)
                return base + `<span class="red-ending">ert</span><span style="color:#2563eb;">e</span>`;
            }
            if (isFeminine && isPlural) {
                // elles → ouvertes : ert(rouge) + es(bleu)
                return base + `<span class="red-ending">ert</span><span style="color:#2563eb;">es</span>`;
            }
            if (!isFeminine && isPlural) {
                // ils → ouverts : ert(rouge) + s(rouge)
                return base + `<span class="red-ending">ert</span><span class="red-ending">s</span>`;
            }
            // il → ouvert
            return base + `<span class="red-ending">ert</span>`;
        }
        
        return verbForm;
    }
    
    // === FUTUR SIMPLE ===
    if (tenseId === 'futur_simple') {
        if (verbForm.endsWith('ai')) return verbForm.slice(0, -2) + `<span class="red-ending">ai</span>`;
        if (verbForm.endsWith('as')) return verbForm.slice(0, -2) + `<span class="red-ending">as</span>`;
        if (verbForm.endsWith('a')) return verbForm.slice(0, -1) + `<span class="red-ending">a</span>`;
        if (verbForm.endsWith('ons')) return verbForm.slice(0, -3) + `<span class="red-ending">ons</span>`;
        if (verbForm.endsWith('ez')) return verbForm.slice(0, -2) + `<span class="red-ending">ez</span>`;
        if (verbForm.endsWith('ont')) return verbForm.slice(0, -3) + `<span class="red-ending">ont</span>`;
    }
    
    // === CONDITIONNEL PRÉSENT ===
    if (tenseId === 'conditionnel_present') {
        if (verbForm.endsWith('ais')) return verbForm.slice(0, -3) + `<span class="red-ending">ais</span>`;
        if (verbForm.endsWith('ait')) return verbForm.slice(0, -3) + `<span class="red-ending">ait</span>`;
        if (verbForm.endsWith('ions')) return verbForm.slice(0, -4) + `<span class="red-ending">ions</span>`;
        if (verbForm.endsWith('iez')) return verbForm.slice(0, -3) + `<span class="red-ending">iez</span>`;
        if (verbForm.endsWith('aient')) return verbForm.slice(0, -5) + `<span class="red-ending">aient</span>`;
    }
    
    // === SUBJONCTIF PRÉSENT ===
    if (tenseId === 'subjonctif_present') {
        if (verbForm.endsWith('ions')) return verbForm.slice(0, -4) + `<span class="red-ending">ions</span>`;
        if (verbForm.endsWith('iez')) return verbForm.slice(0, -3) + `<span class="red-ending">iez</span>`;
        if (verbForm.endsWith('e')) return verbForm.slice(0, -1) + `<span class="red-ending">e</span>`;
        if (verbForm.endsWith('es')) return verbForm.slice(0, -2) + `<span class="red-ending">es</span>`;
        if (verbForm.endsWith('ent')) return verbForm.slice(0, -3) + `<span class="red-ending">ent</span>`;
    }
    
    // === IMPARFAIT ===
    if (tenseId === 'imparfait') {
        if (verbForm.endsWith('ais')) return verbForm.slice(0, -3) + `<span class="red-ending">ais</span>`;
        if (verbForm.endsWith('ait')) return verbForm.slice(0, -3) + `<span class="red-ending">ait</span>`;
        if (verbForm.endsWith('ions')) return verbForm.slice(0, -4) + `<span class="red-ending">ions</span>`;
        if (verbForm.endsWith('iez')) return verbForm.slice(0, -3) + `<span class="red-ending">iez</span>`;
        if (verbForm.endsWith('aient')) return verbForm.slice(0, -5) + `<span class="red-ending">aient</span>`;
    }
    
    // === PRÉSENT / IMPÉRATIF ===
    if (tenseId === 'present' || tenseId === 'imperatif') {
        if (verbForm.endsWith('e')) return verbForm.slice(0, -1) + `<span class="red-ending">e</span>`;
        if (verbForm.endsWith('es')) return verbForm.slice(0, -2) + `<span class="red-ending">es</span>`;
        if (verbForm.endsWith('ons')) return verbForm.slice(0, -3) + `<span class="red-ending">ons</span>`;
        if (verbForm.endsWith('ez')) return verbForm.slice(0, -2) + `<span class="red-ending">ez</span>`;
        if (verbForm.endsWith('ent')) return verbForm.slice(0, -3) + `<span class="red-ending">ent</span>`;
    }
    
    return verbForm;
}


function colorComposed(verbForm, tenseId) {
    let parts = verbForm.split(' ');
    if (parts.length >= 2) {
        // L'auxiliaire en vert
        let aux = `<span class="green-aux">${parts[0]}</span>`;
        
        // Le participe passé
        let participe = parts[1];
        
        // Enlever d'éventuelles balises HTML existantes
        participe = participe.replace(/<[^>]*>/g, '');
        
        // Colorer la terminaison du participe
        let coloredParticipe = colorEnding(participe, 'passe_compose');
        
        return `${aux} ${coloredParticipe}`;
    }
    return verbForm;
}

function isCompound(tenseId) {
    return ['passe_compose', 'plus_que_parfait', 'futur_anterieur', 'conditionnel_passe', 'subjonctif_passe', 'imperatif_passe', 'infinitif_passe', 'gerondif_passe'].includes(tenseId);
}

function colorVerb(verbForm, tenseId) {
    if (isCompound(tenseId)) return colorComposed(verbForm);
    return colorEnding(verbForm, tenseId);
}

function buildTable(tense) {
    let html = `<table class="conj-table">`;
    
    // 🔍 DÉTECTION : Est-ce que c'est le passé composé ?
    let estPasseCompose = (tense.id === 'passe_compose');
    
    // 🔍 Vérifier si l'auxiliaire "être" est utilisé (regarder la première ligne)
    let utiliseEtre = false;
    if (estPasseCompose && tense.conjugation && tense.conjugation.length > 0) {
        // Vérifier si le verbe contient "suis", "es", "est", "sommes", "êtes", "sont"
        const premierVerbe = tense.conjugation[0].v || "";
        utiliseEtre = premierVerbe.includes("suis") || 
                       premierVerbe.includes("es ") || 
                       premierVerbe.includes("est ") ||
                       premierVerbe.includes("sommes") ||
                       premierVerbe.includes("êtes") ||
                       premierVerbe.includes("sont");
    }
    
    for (let row of tense.conjugation) {
        let colored = colorVerb(row.v, tense.id);
        let pronom = row.p;
        
        // Gestion de l'élision pour "je"
        if (pronom === "je" && colored && /^[aeiouyâêîôûàèìòùéèëïü]/i.test(colored)) {
            pronom = "j'";
        }
        else if (pronom === "que je" && colored && /^[aeiouyâêîôûàèìòùéèëïü]/i.test(colored)) {
            pronom = "que j'";
        }
        
        html += `<tr><td class="pronoun-cell">${pronom}</td><td class="verb-cell">${colored}</td></tr>`;
    }
    html += `</table>`;
    
    // 📝 AFFICHER LA NOTE si c'est le passé composé avec l'auxiliaire "être"

function showAllConjugations() {
    const allTenses = [...tempsData.indicatif, ...tempsData.subjonctif, ...tempsData.conditionnel, ...tempsData.imperatif, ...tempsData.infinitif, ...tempsData.participe, ...tempsData.gerondif, ...tempsData.tournures];
    const allConjAreas = document.querySelectorAll('.conj-area');
    allConjAreas.forEach((area, index) => {
        if (index < allTenses.length) {
            area.innerHTML = buildTable(allTenses[index]);
            area.classList.add('visible');
        }
    });
    isListVisible = true;
    document.getElementById("btnVoirTous").innerHTML = "📖 fermer la liste";
}

function hideAllConjugations() {
    document.querySelectorAll('.conj-area').forEach(area => area.classList.remove('visible'));
    isListVisible = false;
    document.getElementById("btnVoirTous").innerHTML = "📖 voir la liste";
}

function toggleAllConjugations() {
    isListVisible ? hideAllConjugations() : showAllConjugations();
}
