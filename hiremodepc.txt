    <script>
        const btn = document.getElementById('btnHanomboka');
        const toast = document.getElementById('toastOk');
        const fsIndicator = document.getElementById('fsIndicator');
        let isFullscreen = false;

        btn.addEventListener('click', function() {
            // Afficher le toast OK
            toast.classList.add('show');
            setTimeout(() => {
                toast.classList.remove('show');
            }, 2500);

            // Mettre en plein écran
            if (!isFullscreen) {
                const elem = document.documentElement;
                const requestMethod = elem.requestFullscreen || elem.webkitRequestFullscreen || elem.msRequestFullscreen;
                if (requestMethod) {
                    requestMethod.call(elem).then(() => {
                        isFullscreen = true;
                        fsIndicator.classList.add('show');
                        setTimeout(() => fsIndicator.classList.remove('show'), 3000);
                    }).catch(function(err) {
                        console.log("Erreur plein écran:", err);
                    });
                }
            }
        });

        // Détecter si l'utilisateur quitte le plein écran avec ESC
        document.addEventListener('fullscreenchange', function() {
            if (!document.fullscreenElement && !document.webkitFullscreenElement) {
                isFullscreen = false;
            }
        });
