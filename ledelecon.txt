<script>
    // Banque de questions : le déterminant manquant dans la phrase
    const questionsBank = [
        { text: "____ livre est intéressant.", correct: "Le", options: ["Le", "Un", "Ce", "Mon"] },
        { text: "J'achète ____ pommes au marché.", correct: "des", options: ["des", "les", "ces", "mes"] },
        { text: "____ fille s'appelle Sophie.", correct: "La", options: ["La", "Une", "Ma", "Cette"] },
        { text: "____ stylo bleu est à moi.", correct: "Ce", options: ["Ce", "Un", "Le", "Mon"] },
        { text: "____ amis viendront demain.", correct: "Mes", options: ["Mes", "Les", "Des", "Ces"] },
        { text: "J'ai vu ____ oiseau rare.", correct: "un", options: ["un", "le", "ce", "mon"] },
        { text: "____ voiture est très rapide.", correct: "Cette", options: ["Cette", "La", "Une", "Ma"] },
        { text: "____ enfants jouent dans le jardin.", correct: "Les", options: ["Les", "Des", "Ces", "Mes"] },
        { text: "____ idée est géniale !", correct: "Ton", options: ["Ton", "Un", "Ce", "Le"] },
        { text: "____ professeur explique bien.", correct: "Le", options: ["Le", "Un", "Ce", "Notre"] },
        { text: "____ fleurs sont magnifiques.", correct: "Ces", options: ["Ces", "Les", "Des", "Mes"] },
        { text: "____ arbre centenaire est tombé.", correct: "L'", options: ["L'", "Un", "Cet", "Mon"] }
    ];

    let currentQuestions = [];      // 6 questions sélectionnées aléatoirement
    let userAnswers = [];
    let answerStatus = [];
    let score = 0;
    let totalQ = 0;

    // Fonction utilitaire pour mélanger un tableau
    function shuffleArray(arr) {
        for (let i = arr.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
        return arr;
    }

    // Initialisation du quiz : pioche aléatoire de 6 questions
    function initQuiz() {
        const shuffledAll = shuffleArray([...questionsBank]);
        currentQuestions = shuffledAll.slice(0, 6);
        totalQ = currentQuestions.length;
        document.getElementById('totalQuestions').innerText = totalQ;
        userAnswers = new Array(totalQ).fill(null);
        answerStatus = new Array(totalQ).fill(null);
        score = 0;
        updateScoreDisplay();
        renderQuestions();
    }

    // Rendu des questions dans le DOM
    function renderQuestions() {
        const container = document.getElementById('questionsContainer');
        if (!container) return;
        container.innerHTML = '';

        currentQuestions.forEach((q, idx) => {
            // On utilise les options fournies, on les mélange pour plus de variété
            let opts = [...q.options];
            // S'assurer que la bonne réponse est bien incluse (c'est le cas dans notre banque mais sécurité)
            if (!opts.includes(q.correct)) {
                opts.push(q.correct);
            }
            opts = shuffleArray([...new Set(opts)]); // enlever doublons éventuels

            const questionDiv = document.createElement('div');
            questionDiv.className = 'question';
            questionDiv.setAttribute('data-qidx', idx);

            const questionText = document.createElement('div');
            questionText.className = 'question-text';
            questionText.innerHTML = `<span style="color:#a5f3fc;">${idx+1}.</span> ${q.text}`;

            const optionsDiv = document.createElement('div');
            optionsDiv.className = 'options';

            opts.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = 'option-btn';
                btn.innerText = opt;
                btn.setAttribute('data-value', opt);
                if (userAnswers[idx] === opt) {
                    if (answerStatus[idx] === 'correct') btn.classList.add('correct-glow');
                    else if (answerStatus[idx] === 'wrong') btn.classList.add('wrong-glow');
                }
                btn.addEventListener('click', (e) => {
                    handleAnswer(idx, opt, q.correct);
                });
                optionsDiv.appendChild(btn);
            });

            const feedbackSpan = document.createElement('div');
            feedbackSpan.className = 'feedback';
            if (answerStatus[idx] === 'correct') {
                feedbackSpan.innerHTML = '✅ Exact ! Bon déterminant.';
                feedbackSpan.style.color = '#86efac';
            } else if (answerStatus[idx] === 'wrong') {
                feedbackSpan.innerHTML = `❌ Oups ! La bonne réponse était : "${q.correct}". Relis la leçon.`;
                feedbackSpan.style.color = '#fca5a5';
            } else {
                feedbackSpan.innerHTML = '🔍 Clique sur le bon déterminant.';
                feedbackSpan.style.color = '#94a3b8';
            }

            questionDiv.appendChild(questionText);
            questionDiv.appendChild(optionsDiv);
            questionDiv.appendChild(feedbackSpan);
            container.appendChild(questionDiv);
        });
    }

    // Gestion de la réponse
    function handleAnswer(qIdx, selected, correctAnswer) {
        const wasCorrect = (answerStatus[qIdx] === 'correct');
        const isNowCorrect = (selected === correctAnswer);

        // Mise à jour de la réponse utilisateur
        userAnswers[qIdx] = selected;

        if (isNowCorrect) {
            if (!wasCorrect) {
                answerStatus[qIdx] = 'correct';
                score++;
            } else {
                answerStatus[qIdx] = 'correct';
            }
        } else {
            if (wasCorrect) {
                answerStatus[qIdx] = 'wrong';
                score--;
            } else {
                answerStatus[qIdx] = 'wrong';
            }
        }

        if (score < 0) score = 0;
        updateScoreDisplay();
        refreshQuestionUI();
    }

    // Rafraîchir l'affichage des couleurs et feedback sans reconstruire tout le DOM
    function refreshQuestionUI() {
        const container = document.getElementById('questionsContainer');
        if (!container) return;
        const questionDivs = container.querySelectorAll('.question');
        questionDivs.forEach((div, idx) => {
            if (idx >= currentQuestions.length) return;
            const q = currentQuestions[idx];
            const optionsDiv = div.querySelector('.options');
            const feedbackSpan = div.querySelector('.feedback');

            // Mettre à jour l'apparence des boutons
            const btns = optionsDiv.querySelectorAll('.option-btn');
            btns.forEach(btn => {
                const val = btn.getAttribute('data-value');
                btn.classList.remove('correct-glow', 'wrong-glow');
                if (userAnswers[idx] === val) {
                    if (answerStatus[idx] === 'correct') {
                        btn.classList.add('correct-glow');
                    } else if (answerStatus[idx] === 'wrong') {
                        btn.classList.add('wrong-glow');
                    }
                }
            });

            // Mettre à jour le feedback
            if (answerStatus[idx] === 'correct') {
                feedbackSpan.innerHTML = '✅ Exact ! Bon déterminant.';
                feedbackSpan.style.color = '#86efac';
            } else if (answerStatus[idx] === 'wrong') {
                feedbackSpan.innerHTML = `❌ Oups ! La bonne réponse était : "${q.correct}". Relis la leçon.`;
                feedbackSpan.style.color = '#fca5a5';
            } else {
                feedbackSpan.innerHTML = '🔍 Clique sur le bon déterminant.';
                feedbackSpan.style.color = '#94a3b8';
            }
        });
    }

    function updateScoreDisplay() {
        const scoreSpan = document.getElementById('scoreValue');
        if (scoreSpan) scoreSpan.innerText = score;
    }

    function resetQuiz() {
        // Réinitialisation complète
        userAnswers = new Array(totalQ).fill(null);
        answerStatus = new Array(totalQ).fill(null);
        score = 0;
        updateScoreDisplay();
        // Efface les styles sur les boutons et remet feedback neutre
        const container = document.getElementById('questionsContainer');
        if (container) {
            const questionDivs = container.querySelectorAll('.question');
            questionDivs.forEach((div, idx) => {
                const feedbackSpan = div.querySelector('.feedback');
                if (feedbackSpan) {
                    feedbackSpan.innerHTML = '🔍 Clique sur le bon déterminant.';
                    feedbackSpan.style.color = '#94a3b8';
                }
                const btns = div.querySelectorAll('.option-btn');
                btns.forEach(btn => {
                    btn.classList.remove('correct-glow', 'wrong-glow');
                });
            });
        }
        // Réinitialiser les tableaux internes
        for (let i = 0; i < totalQ; i++) {
            answerStatus[i] = null;
            userAnswers[i] = null;
        }
        refreshQuestionUI();
    }

    // Transition : 2 secondes d'écran d'attente puis affichage de l'exercice
    function startExerciseAfterDelay() {
        const waitingDiv = document.getElementById('waitingScreen');
        const exerciseDiv = document.getElementById('exerciseContainer');
        if (waitingDiv && exerciseDiv) {
            setTimeout(() => {
                waitingDiv.style.opacity = '0';
                setTimeout(() => {
                    waitingDiv.style.display = 'none';
                    exerciseDiv.style.display = 'block';
                    initQuiz();
                    const resetBtn = document.getElementById('resetQuizBtn');
                    if (resetBtn) {
                        // Remplacer pour éviter les doublons d'écouteurs
                        const newReset = resetBtn.cloneNode(true);
                        resetBtn.parentNode.replaceChild(newReset, resetBtn);
                        newReset.addEventListener('click', () => {
                            resetQuiz();
                        });
                    }
                }, 500);
            }, 1800);
        } else {
            if (exerciseDiv) exerciseDiv.style.display = 'block';
            if (waitingDiv) waitingDiv.style.display = 'none';
            initQuiz();
        }
    }

    window.addEventListener('DOMContentLoaded', () => {
        startExerciseAfterDelay();
    });
</script>
