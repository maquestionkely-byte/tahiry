class BlobURLManager {
    constructor() {
        this.urls = new Map();
        this.cleanupInterval = null;
        this.startCleanup();
    }

    create(url, filename) {
        if (this.urls.has(filename)) {
            this.revoke(filename);
        }
        this.urls.set(filename, { url, timestamp: Date.now(), refCount: 0 });
        return url;
    }

    revoke(filename) {
        if (this.urls.has(filename)) {
            const data = this.urls.get(filename);
            try {
                URL.revokeObjectURL(data.url);
            } catch(e) {}
            this.urls.delete(filename);
        }
    }

    incrementRef(filename) {
        if (this.urls.has(filename)) {
            const data = this.urls.get(filename);
            data.refCount++;
            this.urls.set(filename, data);
        }
    }

    decrementRef(filename) {
        if (this.urls.has(filename)) {
            const data = this.urls.get(filename);
            data.refCount = Math.max(0, data.refCount - 1);
            this.urls.set(filename, data);
        }
    }

    cleanup() {
        const now = Date.now();
        const maxAge = 60000;
        for (const [filename, data] of this.urls) {
            if (data.refCount === 0 && (now - data.timestamp) > maxAge) {
                this.revoke(filename);
            }
        }
    }

    startCleanup() {
        if (this.cleanupInterval) clearInterval(this.cleanupInterval);
        this.cleanupInterval = setInterval(() => this.cleanup(), 30000);
    }

    stopCleanup() {
        if (this.cleanupInterval) {
            clearInterval(this.cleanupInterval);
            this.cleanupInterval = null;
        }
    }

    destroy() {
        this.stopCleanup();
        for (const [filename] of this.urls) {
            this.revoke(filename);
        }
        this.urls.clear();
    }
}

class AudioThumbnailGenerator {
    constructor() {
        this.cache = new Map();
    }

    async generate(audioBlob, filename) {
        if (this.cache.has(filename)) {
            return this.cache.get(filename);
        }

        return new Promise((resolve) => {
            try {
                const canvas = document.createElement('canvas');
                canvas.width = 320;
                canvas.height = 180;
                const ctx = canvas.getContext('2d');
                
                // Fond avec dégradé
                const gradient = ctx.createLinearGradient(0, 0, 320, 180);
                gradient.addColorStop(0, '#1a1a2e');
                gradient.addColorStop(0.5, '#2d1b3d');
                gradient.addColorStop(1, '#1a1a2e');
                ctx.fillStyle = gradient;
                ctx.fillRect(0, 0, 320, 180);

                // Effet de vagues sonores
                for (let i = 0; i < 12; i++) {
                    const x = 40 + i * 22;
                    const height = 15 + Math.random() * 45;
                    const alpha = 0.05 + Math.random() * 0.25;
                    ctx.fillStyle = `rgba(255, 180, 70, ${alpha})`;
                    ctx.fillRect(x, 140 - height, 3, height);
                }

                // Icône audio
                const text = '🎵';
                ctx.font = '70px Arial';
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                ctx.shadowColor = 'rgba(255, 180, 70, 0.4)';
                ctx.shadowBlur = 30;
                ctx.fillStyle = '#FFB347';
                ctx.fillText(text, 160, 95);

                // Texte "AUDIO"
                ctx.shadowBlur = 0;
                ctx.font = '12px Arial';
                ctx.fillStyle = 'rgba(255,255,255,0.12)';
                ctx.textAlign = 'center';
                ctx.textBaseline = 'bottom';
                ctx.fillText('🎵 AUDIO', 160, 170);

                const poster = canvas.toDataURL('image/jpeg', 0.8);
                this.cache.set(filename, poster);
                resolve(poster);
            } catch(e) {
                this.cache.set(filename, null);
                resolve(null);
            }
        });
    }

    clear() { this.cache.clear(); }
}

    if (!bytes || bytes === 0) return '';
    const sizes = ['o', 'Ko', 'Mo', 'Go'];
    const i = Math.floor(Math.log(bytes) / Math.log(1024));
    return (bytes / Math.pow(1024, i)).toFixed(1) + ' ' + sizes[i];
}

async function openDB(galleryId) {
    return new Promise((resolve, reject) => {
        const req = indexedDB.open(`Gallery_${galleryId}`, 4);
        req.onerror = () => reject(req.error);
        req.onsuccess = (e) => resolve(e.target.result);
        req.onupgradeneeded = (e) => {
            const db = e.target.result;
            if (!db.objectStoreNames.contains('medias')) {
                const store = db.createObjectStore('medias', { keyPath: 'filename' });
                store.createIndex('status', 'status', { unique: false });
            }
            if (!db.objectStoreNames.contains('metadata')) {
                db.createObjectStore('metadata', { keyPath: 'key' });
            }
        };
    });
}

async function saveMedia(galleryId, filename, blob, numero, titre, status = 'ok') {
    const db = await openDB(galleryId);
    return new Promise((resolve, reject) => {
        const tx = db.transaction(['medias'], 'readwrite');
        tx.objectStore('medias').put({ filename, blob, numero, titre, status });
        tx.oncomplete = resolve;
        tx.onerror = () => reject(tx.error);
    });
}

async function getAllMedias(galleryId) {
    const db = await openDB(galleryId);
    return new Promise((resolve, reject) => {
        const req = db.transaction(['medias'], 'readonly').objectStore('medias').getAll();
        req.onsuccess = () => resolve(req.result || []);
        req.onerror = () => reject(req.error);
    });
}

async function getStoredNum(galleryId) {

}

}

async function fetchManifest(gallery) {
    try {
        const resp = await fetch(gallery.manifest, { cache: 'no-cache' });
        if (!resp.ok) return [];
        const text = await resp.text();
        const lines = text.split(/\r?\n/).filter(l => l.trim() !== '');
        const map = new Map();
        let currentNum = null, currentFilename = null, currentTitre = '';
        const regex = new RegExp(`^${gallery.prefix}(\\d+)\\.([a-zA-Z0-9]+)$`);
        for (const line of lines) {
            const match = line.match(regex);
            if (match) {
                if (currentNum !== null && currentFilename !== null) {
                    map.set(currentNum, { filename: currentFilename, numero: currentNum, titre: currentTitre || `Audio ${currentNum}` });
                }
                currentNum = parseInt(match[1]);
                currentFilename = line;
                currentTitre = '';
            } else if (currentNum !== null) {
                const titreMatch = line.match(/^Titre\s*:\s*(.+)$/i);
                if (titreMatch) currentTitre = titreMatch[1].trim();
            }
        }
        if (currentNum !== null && currentFilename !== null) {
            map.set(currentNum, { filename: currentFilename, numero: currentNum, titre: currentTitre || `Audio ${currentNum}` });
        }
        return Array.from(map.values()).sort((a,b) => a.numero - b.numero);
    } catch(e) { 
        console.error('Erreur fetchManifest:', e);
        return []; 
    }
}

// ============================================================

async function telechargerAudioIndividuel(galleryId, filename, titre, numero) {
    const url = `${WORKER_URL_AUDIO}/${filename}?deviceModel=${encodeURIComponent(deviceModel)}`;
    console.log(`📥 Téléchargement: ${filename}`);
    
    try {
        const res = await fetch(url);
        
        // ✅ CAS 1 : Le Worker répond 403 (code requis)
        if (res.status === 403) {
            const errorData = await res.json();
            if (errorData.error === "access_denied") {
                // 🔒 Statut = "verrouille"
                await saveMedia(galleryId, filename, null, numero, titre, 'verrouille');
                console.log(`🔒 ${filename} → VERROUILLÉ (code requis)`);
                return { status: 'verrouille', data: errorData };
            }
            return { status: 'error', data: errorData };
        }
        
        // ✅ CAS 2 : Le Worker répond OK (téléchargement réussi)
        if (res.ok) {
            const blob = await res.blob();
            // ✅ Statut = "ok" (déverrouillé)
            await saveMedia(galleryId, filename, blob, numero, titre, 'ok');
            console.log(`✅ ${filename} → OK (téléchargé)`);
            
            // 🔥 Forcer la mise à jour de currentMediaList
            const indexInList = currentMediaList.findIndex(item => item.filename === filename);
            if (indexInList !== -1) {
                const db = await openDB(galleryId);
                const updated = await new Promise((resolve) => {
                    const req = db.transaction(['medias'], 'readonly').objectStore('medias').get(filename);
                    req.onsuccess = () => resolve(req.result);
                });
                if (updated) {
                    currentMediaList[indexInList] = updated;
                }
            }
            
            return { status: 'ok', data: null };
        }
        
        // ✅ CAS 3 : Autre erreur (pas de statut particulier)
        console.log(`⚠️ ${filename} → ERREUR (${res.status})`);
        return { status: 'error', data: { message: `HTTP ${res.status}` } };
        
    } catch(e) {
        console.error(`❌ Erreur ${filename}:`, e);
        return { status: 'error', data: { message: e.message } };
    }
}

        codeError.textContent = '⚠️ Ampidiro ny code';
        codeError.classList.add('visible');

    }

        codeError.textContent = '⚠️ Tsy misy audio voafidy';
        codeError.classList.add('visible');
        return;
    }
    const { galleryId, filename, titre, numero, fromTrash } = codePending;
    btnValidateCode.disabled = true;
    btnValidateCode.textContent = '⏳ Vérification...';
    try {
        const niveau = extrairePrefixe(filename);
        const authResponse = await fetch(`${WORKER_URL_API}/api/authorize`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ code, deviceModel: deviceModel, niveau: niveau })
        });
        const authData = await authResponse.json();

async function telechargerTousAudiosVerrouilles(galleryId, manifest) {
    if (isDownloading) return;
    isDownloading = true;
    
    let stored = await getAllMedias(galleryId);
    const deleted = getItemsSupprimes(galleryId);
    const deletedFilenames = deleted.map(v => v.filename);
    stored = stored.filter(item => !deletedFilenames.includes(item.filename));
   // Décroissant (nouveaux en premier)
stored = stored.sort((a,b) => b.numero - a.numero);
    
    // 🔍 Trouver la première vidéo verrouillée
    let premiereVerrouillee = null;
    for (let i = 0; i < stored.length; i++) {
        if (stored[i].status === 'verrouille') {
            premiereVerrouillee = stored[i];
            break;
        }
    }
    
    if (!premiereVerrouillee) {
        console.log('✅ Aucun audio verrouillé trouvé');
        isDownloading = false;
        return;
    }
    
    const numeroDepart = premiereVerrouillee.numero;
    const audiosATelecharger = manifest.filter(m => m.numero >= numeroDepart);
    console.log(`📥 ${audiosATelecharger.length} audio(s) à vérifier à partir de ${numeroDepart}`);
    
    let compteTelechargees = 0;
    
    for (const media of audiosATelecharger) {
        // Vérifier si déjà OK
        const existeDeja = stored.find(item => item.filename === media.filename && item.status === 'ok');
        if (existeDeja) {
            console.log(`⏭️ ${media.filename} déjà OK, skip`);
            continue;
        }
        
        // 🔥 Télécharger et laisser le Worker décider du statut
        const result = await telechargerAudioIndividuel(galleryId, media.filename, media.titre, media.numero);
        
        if (result.status === 'ok') {
            compteTelechargees++;
            await setStoredNum(galleryId, media.numero);
            
            // Mettre à jour currentMediaList
            const indexInList = currentMediaList.findIndex(item => item.filename === media.filename);
            if (indexInList !== -1) {
                const allMedias = await getAllMedias(galleryId);
                const mediaUpdated = allMedias.find(item => item.filename === media.filename);
                if (mediaUpdated) {
                    currentMediaList[indexInList] = mediaUpdated;
                }
            }
            
            await renderGallery(galleryId);
            
        } else if (result.status === 'verrouille') {
            // 🔒 Le Worker dit que c'est verrouillé → on arrête
            console.log(`🔒 ${media.filename} → VERROUILLÉ (arrêt)`);
            // 🔥 Mettre à jour l'affichage pour montrer le verrou
            await renderGallery(galleryId);
            break;
            
        } else {
            // ❌ Erreur
            console.log(`❌ Erreur ${media.filename}, arrêt`);
            break;
        }
    }
    
    console.log(`✅ ${compteTelechargees} audio(s) téléchargé(s) avec succès`);
    isDownloading = false;
    await renderGallery(galleryId);
    
    if (compteTelechargees > 0) {
        afficherNotificationCentre(`✅ ${compteTelechargees} audio(s) déverrouillé(s) !`, 'success');
    }
}

async function telechargerAudiosManquants(galleryId, manifest, dernierNumOk) {
    if (isDownloading) return;
    isDownloading = true;
    
    let maxNumManifest = 0;
    for (const media of manifest) {
        if (media.numero > maxNumManifest) maxNumManifest = media.numero;
    }
    if (maxNumManifest <= dernierNumOk) {
        console.log(`✅ Aucun nouvel audio (${dernierNumOk} >= ${maxNumManifest})`);
        isDownloading = false;
        return;
    }
    
    const debut = dernierNumOk + 1;
    const fin = maxNumManifest;
    const audiosManquants = manifest.filter(m => m.numero >= debut && m.numero <= fin);
    console.log(`📥 ${audiosManquants.length} audio(s) à traiter (${debut} à ${fin})`);
    
    let dernierNum = dernierNumOk;
    for (const media of audiosManquants) {
        const db = await openDB(galleryId);
        const existing = await new Promise((resolve) => {
            const req = db.transaction(['medias'], 'readonly').objectStore('medias').get(media.filename);
            req.onsuccess = () => resolve(req.result);
            req.onerror = () => resolve(null);
        });
        
        if (!existing) {
            await saveMedia(galleryId, media.filename, null, media.numero, media.titre, 'verrouille');
        }
        
        const result = await telechargerAudioIndividuel(galleryId, media.filename, media.titre, media.numero);
        
        if (result.status === 'ok') {
            if (media.numero > dernierNum) {
                dernierNum = media.numero;
                await setStoredNum(galleryId, dernierNum);
            }
            console.log(`✅ ${media.filename} téléchargé`);
            await renderGallery(galleryId);
        } else if (result.status === 'verrouille') {
            console.log(`🔒 ${media.filename} verrouillé`);
        }
    }
    
    console.log(`✅ Traitement terminé, dernier numéro: ${dernierNum}`);
    isDownloading = false;
    await renderGallery(galleryId);
}

// ATTACHER LES ÉVÉNEMENTS
// ============================================================
function attachCardEvents(galleryId) {
    document.querySelectorAll('.media-card:not(.verrouille)').forEach(card => {
        card.addEventListener('click', function(e) {
            if (e.target.classList.contains('delete-btn')) return;
            const index = parseInt(this.dataset.index);
            if (isNaN(index)) return;
            const item = currentMediaList[index];
            if (!item || !item.blob) {
                afficherNotificationCentre('⚠️ Audio non disponible', 'error');
                return;
            }
            const filteredIndex = currentFilteredList.findIndex(
                f => f.filename === currentMediaList[index].filename
            );
            openLightbox(filteredIndex >= 0 ? filteredIndex : 0);
        });
    });

    document.querySelectorAll('.media-card.verrouille').forEach(card => {
        card.addEventListener('click', function() {
            const index = parseInt(this.dataset.index);
            if (isNaN(index)) return;
            const item = currentMediaList[index];
            if (!item) return;
            codePending = {
                galleryId: galleryId,
                filename: item.filename,
                titre: item.titre || `Audio ${item.numero}`,
                numero: item.numero,
                fromTrash: false
            };
            ouvrirModalCode(item.filename, item.titre, item.numero);
        });
    });

    document.querySelectorAll('.delete-btn').forEach(btn => {
        btn.addEventListener('click', function(e) {
            e.stopPropagation();
            e.preventDefault();
            const filename = this.dataset.filename;
            const titre = this.dataset.titre;
            const numero = parseInt(this.dataset.numero);
            supprimerAudio(galleryId, filename, titre, numero);
        });
    });

    let deleteMode = false;
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Delete' || e.key === 'Suppr') {
            deleteMode = !deleteMode;
            document.querySelectorAll('.media-card').forEach(card => {
                if (deleteMode) {
                    card.classList.add('delete-mode');
                    const btn = card.querySelector('.delete-btn');
                    if (btn) btn.style.display = 'flex';
                } else {
                    card.classList.remove('delete-mode');
                    const btn = card.querySelector('.delete-btn');
                    if (btn) btn.style.display = 'none';
                }
            });
            afficherNotificationCentre(
                deleteMode ? '🗑️ Mode suppression activé' : '✅ Mode suppression désactivé',
                deleteMode ? 'error' : 'success'
            );
        }
    });
}

// SUPPRIMER UN AUDIO
// ============================================================
async function supprimerAudio(galleryId, filename, titre, numero) {
    if (!confirm(`Supprimer "${titre}" de la bibliothèque ?`)) return;
    try {
        const db = await openDB(galleryId);
        await new Promise((resolve, reject) => {
            const tx = db.transaction(['medias'], 'readwrite');
            const req = tx.objectStore('medias').delete(filename);
            req.onsuccess = () => {
                ajouterItemSupprime(galleryId, filename, titre, numero);
                resolve();
            };
            req.onerror = () => reject(req.error);
        });
        afficherNotificationCentre(`🗑️ "${titre}" supprimé`, 'success');
        await renderGallery(galleryId);
    } catch(e) {
        console.error('Erreur:', e);
        afficherNotificationCentre('❌ Erreur lors de la suppression', 'error');
    }
}

            const scale3 = 1.2 + Math.sin(phase + 1) * 0.3;
            
            wave1.style.transform = `translate(-50%,-50%) scale(${scale1})`;
            wave1.style.opacity = 0.6 + Math.sin(phase) * 0.2;
            wave1.style.borderWidth = `${2 + Math.sin(phase) * 1}px`;
            
            wave2.style.transform = `translate(-50%,-50%) scale(${scale2})`;
            wave2.style.opacity = 0.4 + Math.sin(phase + 0.5) * 0.2;
            wave2.style.borderWidth = `${2 + Math.sin(phase + 0.5) * 1}px`;
            
            wave3.style.transform = `translate(-50%,-50%) scale(${scale3})`;
            wave3.style.opacity = 0.2 + Math.sin(phase + 1) * 0.15;
            wave3.style.borderWidth = `${2 + Math.sin(phase + 1) * 1}px`;
        }, 50);
    }
    
    function stopAnimation() {
        icon.style.animation = 'none';
        icon.style.textShadow = '0 0 20px rgba(255, 180, 70, 0.2)';
        
        if (waveInterval) {
            clearInterval(waveInterval);
            waveInterval = null;
        }
        
        wave1.style.transform = 'translate(-50%,-50%) scale(1)';
        wave1.style.opacity = '0';
        wave2.style.transform = 'translate(-50%,-50%) scale(1)';
        wave2.style.opacity = '0';
        wave3.style.transform = 'translate(-50%,-50%) scale(1)';
        wave3.style.opacity = '0';
    }
    
    if (audio) {
        audio.addEventListener('play', startAnimation);
        audio.addEventListener('pause', stopAnimation);
        audio.addEventListener('ended', stopAnimation);
        
        // Si déjà en lecture, démarrer l'animation
        if (!audio.paused) {
            startAnimation();
        }
    }
}

function filterVideos(query) {
    searchTerm = query.toLowerCase().trim();
    const cards = document.querySelectorAll('.media-card');
    let hasVisible = false;
    cards.forEach((card, index) => {
        if (index >= currentMediaList.length) return;
        const item = currentMediaList[index];
        const title = (item.titre || '').toLowerCase();
        const filename = (item.filename || '').toLowerCase();
        const match = !searchTerm || title.includes(searchTerm) || filename.includes(searchTerm);
        if (match) {
            card.classList.remove('hidden-card');
            hasVisible = true;
        } else {
            card.classList.add('hidden-card');
        }
    });
    let noResult = document.querySelector('.no-result');
    if (!hasVisible && searchTerm && cards.length > 0) {
        if (!noResult) {
            noResult = document.createElement('div');
            noResult.className = 'no-result';
            noResult.innerHTML = `<span>🔍</span>Aucun audio trouvé pour "<strong>${searchTerm}</strong>"`;
            galleryContainer.appendChild(noResult);
        } else {
            noResult.innerHTML = `<span>🔍</span>Aucun audio trouvé pour "<strong>${searchTerm}</strong>"`;
            noResult.style.display = 'block';
        }
    } else if (noResult) {
        noResult.style.display = 'none';
    }
    currentFilteredList = currentMediaList.filter((item, index) => {
        const card = cards[index];
        return card && !card.classList.contains('hidden-card');
    });
}

function afficherCorbeille() {
    const galleryId = currentGallery;
    const items = getItemsSupprimes(galleryId);
    trashList.innerHTML = '';
    if (items.length === 0) {
        trashList.innerHTML = `<div class="trash-empty"><span>🧹</span>Aucun audio supprimé</div>`;
        return;
    }
    for (const item of items) {
        const div = document.createElement('div');
        div.className = 'trash-item';
        div.dataset.filename = item.filename;
        div.innerHTML = `
            <div class="info">
                <div class="title">${item.titre || `Audio ${item.numero}`}</div>
                <div class="detail">${item.filename} • N°${item.numero}</div>
            </div>
            <button class="btn-redownload" onclick="retelechargerAudio('${galleryId}', '${item.filename}', '${item.titre.replace(/'/g, "\\'")}', ${item.numero})">
                🔄 Retélécharger
            </button>
        `;
        trashList.appendChild(div);
    }
}

async function retelechargerAudio(galleryId, filename, titre, numero) {
    const btn = document.querySelector(`.trash-item[data-filename="${filename}"] .btn-redownload`);
    if (btn) { btn.textContent = '⏳...'; btn.classList.add('loading'); }
    try {
        const result = await telechargerAudioIndividuel(galleryId, filename, titre, numero);
        if (result.status === 'ok') {
            supprimerItemSupprime(galleryId, filename);
            afficherNotificationCentre(`✅ ${titre} retéléchargé`, 'success');
            await renderGallery(galleryId);
            afficherCorbeille();
        } else if (result.status === 'verrouille') {
            afficherNotificationCentre('🔒 Code requis pour cet audio', 'error');
            codePending = { galleryId, filename, titre, numero, fromTrash: true };
            ouvrirModalCode(filename, titre, numero);
        } else {
            afficherNotificationCentre('❌ Erreur lors du téléchargement', 'error');
        }
    } catch(e) {
        console.error('Erreur retéléchargement:', e);
        afficherNotificationCentre('❌ Erreur lors du téléchargement', 'error');
    }
    if (btn) { btn.textContent = '🔄 Retélécharger'; btn.classList.remove('loading'); }
}

let scrollTimer = null;

function pauseAllAudios() {
    document.querySelectorAll('.lb-audio, .audio-wrap audio').forEach(audio => {
        try { audio.pause(); audio.currentTime = 0; } catch(e) {}
    });
}

function blockAudioOnScroll() {
    isScrolling = true;
    pauseAllAudios();
    if (scrollTimer) clearTimeout(scrollTimer);
    scrollTimer = setTimeout(() => { isScrolling = false; pauseAllAudios(); }, 500);
}

window.addEventListener('scroll', blockAudioOnScroll, { passive: true });
window.addEventListener('touchmove', blockAudioOnScroll, { passive: true });
window.addEventListener('touchend', () => { setTimeout(pauseAllAudios, 300); }, { passive: true });

function setupAudioObservers() {
    document.querySelectorAll('.audio-wrap audio, .lb-audio').forEach(audio => {
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                try {
                    if (!entry.isIntersecting) {
                        entry.target.pause();
                        entry.target.currentTime = 0;
                    }
                } catch(e) {}
            });
        }, { threshold: 0.1 });
        observer.observe(audio);
    });
}

function applyScrollProtection() {
    pauseAllAudios();
    setupAudioObservers();
}

const scrollObserver = new MutationObserver(() => { applyScrollProtection(); });
scrollObserver.observe(document.body, { childList: true, subtree: true });

document.addEventListener('DOMContentLoaded', () => {
    setTimeout(applyScrollProtection, 500);
    document.querySelectorAll('.audio-wrap audio, .lb-audio').forEach(audio => {
        audio.addEventListener('click', function(e) {
            e.preventDefault(); e.stopPropagation();
            this.pause(); this.currentTime = 0;
        });
        audio.addEventListener('dblclick', function(e) {
            e.preventDefault(); e.stopPropagation();
            this.pause(); this.currentTime = 0;
        });
    });
});

let lastScrollY = 0, ticking = false;
function handleScroll() {
    if (!ticking) {
        window.requestAnimationFrame(() => {
            const currentScrollY = window.scrollY;
            const isScrollingDown = currentScrollY > lastScrollY;
            if (isScrollingDown && currentScrollY > 60) {
                heroHeader.classList.add('hide-header');
            } else if (!isScrollingDown || currentScrollY <= 30) {
                heroHeader.classList.remove('hide-header');
            }
            searchWrapper.classList.toggle('scrolled', currentScrollY > 60);
            lastScrollY = currentScrollY;
            ticking = false;
        });
        ticking = true;
    }
}
window.addEventListener('scroll', handleScroll, { passive: true });
