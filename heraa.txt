            <div style="text-align:center"><span class="final-score">${totalCorrect} / ${TOTAL} (${percent}%)</span></div>
            <p style="text-align:center; margin: 15px 0;">${finalMessage}</p>
            <div class="questions-list">
                <h3>📋 Corrigé complet avec explications :</h3>
        `;
        for (let i = 0; i < TOTAL; i++) {
            const q = questions[i];
            const userChoice = userSelections[i];
            const letters = ['A','B','C','D'];
            const userText = userChoice ? q.options[letters.indexOf(userChoice)] : "(non répondue)";
            const correctAnswerText = q.options[letters.indexOf(q.correct)];
            const isOk = (userChoice === q.correct);
            const borderClass = isOk ? "correct" : "wrong";
            listHtml += `
                <div class="question-summary ${borderClass}">
                    <strong>${isOk ? "✅" : "❌"} Q${i+1} : ${q.q}</strong><br>
                    <span>✏️ Votre réponse : ${userChoice ? userChoice : "—"} → "${userText}"</span><br>
                    <span>📌 Bonne réponse : ${q.correct} → "${correctAnswerText}"</span><br>
                    <small>📖 Explication : ${q.explanation}</small>
                    ${q.mg ? `<br><span style="font-size:0.8rem;">🇲🇬 Malagasy: ${q.mg}</span>` : ''}
                </div>
            `;
        }
        listHtml += `<button class="restart-btn" id="restartQuizBtn">🔄 Recommencer (nouveau tirage de 60 questions)</button>`;
        listHtml += `</div></div>`;
        finalContainer.innerHTML = listHtml;
        
        const restartBtn = document.getElementById('restartQuizBtn');
        if (restartBtn) restartBtn.addEventListener('click', () => resetAndRestart());
    }
    
    function resetAndRestart() {
        let freshMix = shuffleArray([...masterList]);
        while(freshMix.length < 60) freshMix.push(...baseQuestions.slice(0,5));
        questions = freshMix.slice(0,60);
        userSelections = new Array(TOTAL).fill(null);
        isLocked = new Array(TOTAL).fill(false);
        currentScore = 0;
        currentIdx = 0;
        quizFinished = false;
        updateScoreUI();
        finalContainer.innerHTML = '';
        if (motivationDiv) motivationDiv.style.display = 'none';
        clearTranslation();
        renderCurrent();
    }

        a.download = 'qcm_etre_avoir_60.html';
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
