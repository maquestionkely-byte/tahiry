    // ======================= INDEXEDDB POUR LE SYSTÈME DE MISE À JOUR =======================
    function ouvrirDB(nomDB) {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(nomDB, DB_VERSION);
            request.onerror = () => reject(request.error);
            request.onsuccess = (event) => resolve(event.target.result);
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains('contenu')) {
                    db.createObjectStore('contenu');
                }
                if (!db.objectStoreNames.contains('version')) {
                    db.createObjectStore('version');
                }
            };
        });
    }
    
    async function stockerContenuEtVersion(nomDB, contenu, version) {
        const db = await ouvrirDB(nomDB);
        return new Promise((resolve, reject) => {
            const transaction = db.transaction(['contenu', 'version'], 'readwrite');
            transaction.objectStore('contenu').put(contenu, 'htmlContent');
            transaction.objectStore('version').put(version, 'versionDate');
            transaction.oncomplete = () => resolve();
            transaction.onerror = () => reject(transaction.error);
        });
    }
    
    async function recupererVersion(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['version'], 'readonly');
                const request = transaction.objectStore('version').get('versionDate');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch {
            return null;
        }
    }
    
    async function recupererContenu(nomDB) {
        try {
            const db = await ouvrirDB(nomDB);
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['contenu'], 'readonly');
                const request = transaction.objectStore('contenu').get('htmlContent');
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch {
            return null;
        }
    }
    
    // ======================= RÉCUPÉRATION DATE VERSION DEPUIS WORKER =======================
    async function getRemoteVersionDate(nomFichier) {
        try {
            const response = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            const content = await response.text();
            
            const blocks = content.split(/DERNIÈRE MISE À JOUR/);
            
            for (const block of blocks) {
                const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
                if (fileMatch && fileMatch[1].trim() === nomFichier) {
                    const dateMatch = block.match(/📆\s*([\d\/]+\s+[\d:]+)/);
                    if (dateMatch) {
                        return dateMatch[1].trim();
                    }
                    const dayMatch = block.match(/📅\s*([\d\/]+)/);
                    const timeMatch = block.match(/⏰\s*([\d:]+)/);
                    if (dayMatch && timeMatch) {
                        return `${dayMatch[1]} ${timeMatch[1]}`;
                    }
                }
            }
            return null;
        } catch (error) {
            console.error(`Erreur récupération date pour ${nomFichier}:`, error);
            return null;
        }
    }
    
    // ======================= TÉLÉCHARGER UN FICHIER =======================

        if (!response.ok) throw new Error(`HTTP ${response.status}`);
        const content = await response.text();
        if (!content || content.length < 50) throw new Error("Contenu invalide");
        return content;

                return true;
            }
            
            console.log(`🆕 Nouvelle version de ${nom} détectée: ${storedVersion} → ${remoteVersion}`);
            
            const nouveauContenu = await telechargerFichier(nom);
            await stockerContenuEtVersion(dbName, nouveauContenu, remoteVersion);
            
            if (afficherNotification) {
                showUpdateNotification(`${nom} mis à jour (${remoteVersion})`);
            }
            
            return true;
            
        } catch (error) {
            console.error(`❌ Erreur mise à jour ${nom}:`, error);
            if (afficherNotification) {
                showUpdateNotification(`Erreur: ${nom}`, true);
            }
            return false;
        }
    }

            request.onupgradeneeded = (e) => { if (!e.target.result.objectStoreNames.contains('images')) e.target.result.createObjectStore('images'); };
            request.onsuccess = (e) => {
                const db = e.target.result;
                const tx = db.transaction('images', 'readwrite');
                tx.objectStore('images').put(blob, url);
                tx.oncomplete = () => { db.close(); resolve(); };
            };
        });
    }
    
    async function chargerImage(element, url) {
        const request = indexedDB.open('image_cache', 7);
        request.onsuccess = async (e) => {
            const db = e.target.result;
            if (db.objectStoreNames.contains('images')) {
                const tx = db.transaction('images', 'readonly');
                const store = tx.objectStore('images');
                const get = store.get(url);
                get.onsuccess = () => {
                    if (get.result) {
                        element.src = URL.createObjectURL(get.result);
                        db.close();
                        return;
                    }
                    db.close();
                    fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
                };
            } else {
                db.close();
                fetch(url).then(res => res.blob()).then(blob => { stockerImage(url, blob); element.src = URL.createObjectURL(blob); }).catch(() => element.src = url);
            }
        };
    }

let navigationStack = []; // Pile d'historique complète
let currentScreen = 'accueil';
let isInBlobView = false;
let blobType = null;
let ignoreNextPop = false;

// Sauvegarder l'écran actuel
function getCurrentScreenId() {
    const activeScreen = document.querySelector('.screen.active');
    if (!activeScreen) return currentScreen;
    
    const id = activeScreen.id;
    if (id === 'screenAccueil') return 'accueil';
    if (id === 'screenDebutant') return 'debutant';
    if (id === 'screenIntermediaire') return 'intermediaire';
    if (id === 'screenAvance') return 'avance';
    if (id === 'screenDictionnaire') return 'dictionnaire';
    if (id === 'screenConjugaison') return 'conjugaison';
    if (id === 'screenDialoguer') return 'dialoguer';
    return currentScreen;
}

// Mettre à jour currentScreen
function updateCurrentScreen() {
    currentScreen = getCurrentScreenId();
    return currentScreen;
}

// Ajouter un écran à l'historique (appelé à chaque navigation)
function pushToHistory(screenName, context = {}) {
    updateCurrentScreen();
    
    // Éviter les doublons consécutifs
    if (navigationStack.length > 0 && navigationStack[navigationStack.length - 1].screen === screenName) {
        return;
    }
    
    navigationStack.push({
        screen: screenName,
        from: currentScreen,
        timestamp: Date.now(),
        context: context
    });
    
    // Limiter la taille
    if (navigationStack.length > 50) {
        navigationStack.shift();
    }
    
    // Push dans l'historique du navigateur
    history.pushState({ stackIndex: navigationStack.length - 1, screen: screenName }, '', window.location.href);
    
    return navigationStack.length - 1;
}

// Revenir à un écran spécifique
function popToScreen(screenName) {
    // Trouver la dernière occurrence de cet écran dans la pile
    let index = -1;
    for (let i = navigationStack.length - 1; i >= 0; i--) {
        if (navigationStack[i].screen === screenName) {
            index = i;
            break;
        }
    }
    
    if (index !== -1 && index < navigationStack.length - 1) {
        // Supprimer l'historique après cet index
        navigationStack = navigationStack.slice(0, index + 1);
        return true;
    }
    return false;
}

// Gérer le blob
function checkIfInBlob() {
    isInBlobView = window.location.href.startsWith('blob:');
    
    if (isInBlobView) {
        const exerciceInfo = sessionStorage.getItem('exerciceInfo');
        const leconInfo = localStorage.getItem('dernierContexteLecon');
        
        if (exerciceInfo) {
            blobType = 'exercice';
        } else if (leconInfo) {
            blobType = 'lecon';
        } else {
            blobType = 'unknown';
        }
    } else {
        blobType = null;
    }
    return isInBlobView;
}

// Fermer blob et retourner
function closeBlobAndReturn() {
    const exerciceInfo = sessionStorage.getItem('exerciceInfo');
    const leconInfo = localStorage.getItem('dernierContexteLecon');
    
    if (exerciceInfo && blobType === 'exercice') {
        const info = JSON.parse(exerciceInfo);
        sessionStorage.removeItem('exerciceInfo');
        const mainUrl = localStorage.getItem('pagePrincipaleUrl') || originalPageUrl;
        localStorage.setItem('retourExerciceLecon', info.leconTitre);
        localStorage.setItem('retourExerciceNiveau', info.niveau || 'debutant');
        window.location.href = mainUrl;
        return true;
    } else if (leconInfo && blobType === 'lecon') {
        localStorage.removeItem('dernierContexteLecon');
        const mainUrl = localStorage.getItem('pagePrincipaleUrl') || originalPageUrl;
        window.location.href = mainUrl;
        return true;
    } else {
        const mainUrl = localStorage.getItem('pagePrincipaleUrl') || originalPageUrl;
        window.location.href = mainUrl;
        return true;
    }
}

// Fonction principale retour
function handleAndroidBack() {
    console.log("🔙 Retour - Pile actuelle:", navigationStack.map(s => s.screen));
    
    // 1. Priorité: Blob
    if (checkIfInBlob()) {
        closeBlobAndReturn();
        return true;
    }
    
    // 2. Restaurer liste exercices
    const retourExerciceLecon = localStorage.getItem('retourExerciceLecon');
    if (retourExerciceLecon) {
        const leconTitre = retourExerciceLecon;
        const niveau = localStorage.getItem('retourExerciceNiveau') || 'debutant';
        localStorage.removeItem('retourExerciceLecon');
        localStorage.removeItem('retourExerciceNiveau');
        
        // Ajouter à l'historique
        pushToHistory('exercices', { leconTitre, niveau });
        
        setTimeout(() => {
            if (typeof afficherListeExercices === 'function') {
                afficherListeExercices(leconTitre, niveau);
            }
        }, 100);
        return true;
    }
    
    // 3. Fermer modal
    const modals = document.querySelectorAll('.modal-fullscreen');
    if (modals.length > 0) {
        const lastModal = modals[modals.length - 1];
        lastModal.remove();
        console.log("✅ Modal fermé");
        return true;
    }
    
    // 4. Navigation dans l'historique (le plus important!)
    if (navigationStack.length > 1) {
        // Retirer l'écran actuel
        navigationStack.pop();
        // Récupérer l'écran précédent
        const previous = navigationStack[navigationStack.length - 1];
        
        if (previous && previous.screen !== currentScreen) {
            console.log(`↩ Retour vers: ${previous.screen} (depuis ${currentScreen})`);
            
            if (typeof originalShowScreen === 'function') {
                originalShowScreen(previous.screen);
            } else if (typeof showScreen === 'function') {
                showScreen(previous.screen);
            }
            
            updateCurrentScreen();
            return true;
        }
    }
    
    // 5. Retour à l'accueil si pas dans l'historique
    const activeScreen = document.querySelector('.screen.active');
    if (activeScreen && activeScreen.id !== 'screenAccueil') {
        console.log("🏠 Retour forcé à l'accueil");
        
        // Ajouter l'accueil à l'historique
        pushToHistory('accueil');
        
        if (typeof originalShowScreen === 'function') {
            originalShowScreen('accueil');
        } else if (typeof showScreen === 'function') {
            showScreen('accueil');
        }
        
        updateCurrentScreen();
        return true;
   
    } else {
        window.lastBackTime = now;
    }
    
    return false;
}

    event.stopPropagation();
    
    if (ignoreNextPop) {
        ignoreNextPop = false;
        return;
    }
    
    handleAndroidBack();
});

// Nouvelle fonction showScreen avec enregistrement auto
window.showScreen = function(screenName, ...args) {
    const currentId = getCurrentScreenId();
    
    // Ne pas enregistrer si c'est le même écran
    if (currentId !== screenName) {
        // Enregistrer dans l'historique
        pushToHistory(screenName, { from: currentId, args: args });
        console.log(`📜 Navigation enregistrée: ${currentId} → ${screenName}`);
    }
    
    // Appeler la fonction originale
    if (typeof originalShowScreen === 'function') {
        return originalShowScreen(screenName, ...args);
    }
    
    updateCurrentScreen();
};
