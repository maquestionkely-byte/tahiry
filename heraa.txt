            currentTranslationVisible = false;
        }, 5000);
    }
    
    function renderCurrent() {
        if (quizFinished) return;
        clearTranslation();
        const q = questions[currentIdx];
        const alreadyAnswered = (userSelections[currentIdx] !== null);
        const letters = ['A', 'B', 'C', 'D'];
        let optsHtml = '';
        q.options.forEach((opt, idx) => {
            const letter = letters[idx];
            let selectedClass = (alreadyAnswered && userSelections[currentIdx] === letter) ? 'selected' : '';
            optsHtml += `
                <div class="option-card ${selectedClass}" data-opt-letter="${letter}">
                    <div class="option-letter">${letter}</div>
                    <span class="option-text">${opt}</span>
                </div>
            `;
        });
        const html = `
            <div class="question-header">
                <div class="question-text">${currentIdx+1}. ${q.q}</div>
                <button class="mg-translate-btn" id="mgTranslateBtn">🇲🇬 MG</button>
            </div>
            <div class="options-list" id="optionsContainer">
                ${optsHtml}
            </div>
        `;
        questionBlock.innerHTML = html;
        const mgBtn = document.getElementById('mgTranslateBtn');
        if (mgBtn) mgBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            toggleTranslation();
        });
        if (!alreadyAnswered && !isLocked[currentIdx] && !quizFinished) {
            const optionDivs = document.querySelectorAll('.option-card');
            optionDivs.forEach(div => {
                div.addEventListener('click', (e) => {
                    if (isLocked[currentIdx] || quizFinished) return;
                    const chosenLetter = div.getAttribute('data-opt-letter');
                    if (chosenLetter) handleAnswer(chosenLetter);
                });
            });
        } else if (alreadyAnswered) {
            document.querySelectorAll('.option-card').forEach(div => {
                div.style.cursor = "default";
                div.style.pointerEvents = "none";
            });
        }
        qCounterSpan.innerText = `Question ${currentIdx+1} / ${TOTAL}`;
    }
    
    function handleAnswer(selectedLetter) {
        if (isLocked[currentIdx] || quizFinished) return;
        isLocked[currentIdx] = true;
        const q = questions[currentIdx];
        const isCorrect = (selectedLetter === q.correct);
        userSelections[currentIdx] = selectedLetter;
        if (isCorrect) {
            currentScore++;
            updateScoreUI();
        }
        renderCurrent();
        const answeredCount = userSelections.filter(a => a !== null).length;
        if (answeredCount === 20 || answeredCount === 40) {
            showPalierMotivation(answeredCount, currentScore);
        }
        setTimeout(() => {
            if (currentIdx + 1 < TOTAL) {
                currentIdx++;
                renderCurrent();
            } else {
                finishQuizAndShowFullReport();
            }
        }, 380);
    }
    
    function finishQuizAndShowFullReport() {
        quizFinished = true;
        questionBlock.innerHTML = `<div style="text-align:center; padding:2rem;">✅ Quiz terminé ! Voici votre note et le corrigé détaillé. ✅</div>`;
        clearTranslation();
        const totalCorrect = currentScore;
        const percent = Math.round((totalCorrect / TOTAL) * 100);
        let finalMessage = "";
        if (totalCorrect === TOTAL) finalMessage = "🏆 PARFAIT ! Maîtrise totale des verbes avoir/être au présent ! 🏆";
        else if (totalCorrect >= 50) finalMessage = "🌟 Très bonne maîtrise ! Révisez les petites erreurs ci-dessous.";
        else if (totalCorrect >= 40) finalMessage = "👍 Bonnes bases, mais revoyez bien les expressions.";
        else finalMessage = "📖 Revoyez les conjugaisons de 'avoir' et 'être' au présent, et les expressions courantes. Corrigé détaillé ci-dessous !";
        let listHtml = `<div class="final-report">
            <div class="final-header">📊 RÉSULTAT FINAL</div>
            <div style="text-align:center"><span class="final-score">${totalCorrect} / ${TOTAL} (${percent}%)</span></div>
            <p style="text-align:center; margin: 15px 0;">${finalMessage}</p>
            <div class="questions-list"><h3>📋 Liste complète des 60 questions avec corrections :</h3>`;
        for (let i = 0; i < TOTAL; i++) {
            const q = questions[i];
            const userChoice = userSelections[i];
            const lettersIdx = ['A','B','C','D'];
            const isOk = (userChoice === q.correct);
            const userText = userChoice ? q.options[lettersIdx.indexOf(userChoice)] : "(non répondue)";
            const correctText = q.options[lettersIdx.indexOf(q.correct)];
            const statusIcon = isOk ? "✅" : "❌";
            const borderClass = isOk ? "correct" : "wrong";
            listHtml += `<div class="question-summary ${borderClass}">
                <strong>${statusIcon} Q${i+1} : ${q.q}</strong><br>
                <span>✏️ Votre réponse : ${userChoice ? userChoice : "—"} → "${userText}"</span><br>
                <span>📌 Bonne réponse : ${q.correct} → "${correctText}"</span><br>
                <small>📖 Explication : ${q.explanation}</small>
                ${q.mg ? `<br><span style="font-size:0.8rem;">🇲🇬 Malagasy: ${q.mg}</span>` : ''}
            </div>`;
        }
        listHtml += `<button class="restart-btn" id="restartQuizBtn">🔄 Recommencer un nouveau quiz (60 questions)</button></div></div>`;
        finalContainer.innerHTML = listHtml;
        const restartBtn = document.getElementById('restartQuizBtn');
        if (restartBtn) restartBtn.addEventListener('click', () => resetAndRestart());
    }
    
    function resetAndRestart() {
        questions = [...FULL_QUESTION_SET.slice(0, TOTAL)];
        userSelections = new Array(TOTAL).fill(null);
        isLocked = new Array(TOTAL).fill(false);
        currentScore = 0;
        currentIdx = 0;
        quizFinished = false;
        updateScoreUI();
        finalContainer.innerHTML = '';
        if (motivationDiv) motivationDiv.style.display = 'none';
        clearTranslation();
        renderCurrent();
    }
    
    downloadBtn.addEventListener('click', () => {
        const htmlContent = document.documentElement.outerHTML;
        const blob = new Blob([htmlContent], { type: 'text/html' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'qcm_avoir_etre_malagasy.html';
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    });
    
    renderCurrent();
    updateScoreUI();
