        // Détection de la console ouverte
        function detectDevTools() {
            const startTime = performance.now();
            debugger;

            if (endTime - startTime > 100) {
                showNotification('🔒 Console développeur détectée et bloquée');
                return true;

            return false;

        // ============================================
        // SAUVEGARDE AVEC INDEXEDDB
        // ============================================
        
        let db = null;
        const DB_NAME = "QuizDB_Secure";
        const STORE_NAME = "progress_secure";
        
        // Ouvrir la base de données
        function ouvrirDB() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.open(DB_NAME, 1);
                request.onerror = () => reject(request.error);
                request.onsuccess = () => {
                    db = request.result;
                    resolve(db);
                };
                request.onupgradeneeded = (event) => {
                    const db = event.target.result;
                    if (!db.objectStoreNames.contains(STORE_NAME)) {
                        db.createObjectStore(STORE_NAME, { keyPath: 'id' });
                    }
                };
            });
        }
        
        // Sauvegarder la progression
        async function sauvegarderProgres() {
            try {
                const q1Selected = document.querySelector('input[name="q1"]:checked');
                const q2Selected = document.querySelector('input[name="q2"]:checked');
                
                const progression = {
                    id: "quiz_hita_secure",
                    q1: q1Selected ? q1Selected.value : null,
                    q2: q2Selected ? q2Selected.value : null,
                    date: new Date().toISOString(),
                    timestamp: Date.now()
                };
                
                await ouvrirDB();
                const tx = db.transaction(STORE_NAME, 'readwrite');
                const store = tx.objectStore(STORE_NAME);
                await store.put(progression);
                await tx.done;
                
                showNotification('✅ Progression sauvegardée !');
                return true;
            } catch (error) {
                console.error("Erreur sauvegarde:", error);
                showNotification('❌ Erreur lors de la sauvegarde');
                return false;
            }
        }
        
        // Charger la progression
        async function chargerProgres() {
            try {
                await ouvrirDB();
                const tx = db.transaction(STORE_NAME, 'readonly');
                const store = tx.objectStore(STORE_NAME);
                const progression = await new Promise((resolve) => {
                    const request = store.get("quiz_hita_secure");
                    request.onsuccess = () => resolve(request.result);
                });
                
                if (progression) {
                    if (progression.q1 !== null) {
                        const q1Radio = document.querySelector(`input[name="q1"][value="${progression.q1}"]`);
                        if (q1Radio) {
                            q1Radio.checked = true;
                            const q1Option = q1Radio.closest('.option');
                            if (q1Option) {
                                const parentOptions = q1Option.closest('.options');
                                parentOptions.querySelectorAll('.option').forEach(opt => {
                                    opt.classList.remove('selected');
                                });
                                q1Option.classList.add('selected');
                            }
                        }
                    }
                    
                    if (progression.q2 !== null) {
                        const q2Radio = document.querySelector(`input[name="q2"][value="${progression.q2}"]`);
                        if (q2Radio) {
                            q2Radio.checked = true;
                            const q2Option = q2Radio.closest('.option');
                            if (q2Option) {
                                const parentOptions = q2Option.closest('.options');
                                parentOptions.querySelectorAll('.option').forEach(opt => {
                                    opt.classList.remove('selected');
                                });
                                q2Option.classList.add('selected');
                            }
                        }
                    }
                    
                    showNotification(`📀 Progression chargée (${new Date(progression.date).toLocaleDateString()})`);
                } else {
                    showNotification('📭 Aucune progression sauvegardée');
                }
                
            } catch (error) {
                console.error("Erreur chargement:", error);
                showNotification('❌ Erreur lors du chargement');
            }
        }
