<script>
    // Éléments DOM
    const tokenInput = document.getElementById('token');
    const repoInput = document.getElementById('repo');
    const branchInput = document.getElementById('branch');
    const fileInput = document.getElementById('file');
    const commitMsgInput = document.getElementById('commitMsg');
    const btnUpload = document.getElementById('btnUpload');
    const resultDiv = document.getElementById('result');
    const urlBox = document.getElementById('urlBox');
    const rawUrlSpan = document.getElementById('rawUrl');
    const btnCopy = document.getElementById('btnCopy');
    const fileInfo = document.getElementById('fileInfo');
    const detectedPathDiv = document.getElementById('detectedPath');
    
    let currentFilePath = "";
    const DATE_FILE_PATH = "date_mise_a_jour.txt";
    
    // Détection automatique du nom du fichier
    fileInput.addEventListener('change', (e) => {
        const file = e.target.files[0];
        if (file) {
            fileInfo.style.display = 'block';
            fileInfo.innerHTML = `📄 ${file.name} (${(file.size/1024).toFixed(1)} Ko)`;
            currentFilePath = file.name;
            detectedPathDiv.style.display = 'block';
            detectedPathDiv.innerHTML = `🎯 Chemin détecté : <strong>${currentFilePath}</strong><br>✅ Si le fichier existe déjà → REMPLACÉ. Sinon → CRÉÉ.`;
        } else {
            fileInfo.style.display = 'none';
            detectedPathDiv.style.display = 'none';
            currentFilePath = "";
        }
    });
    
    function showMessage(msg, type) {
        resultDiv.textContent = msg;
        resultDiv.className = `result ${type}`;
        setTimeout(() => {
            if(resultDiv.className === `result ${type}`) resultDiv.className = 'result';
        }, 5000);
    }
    
    // Appel API GitHub
    async function githubRequest(method, repo, path, token, body = null) {
        const url = `https://api.github.com/repos/${repo}/contents/${encodeURIComponent(path)}`;
        const headers = {
            'Authorization': `token ${token}`,
            'Accept': 'application/vnd.github.v3+json'
        };
        const options = { method, headers };
        if (body) {
            options.body = JSON.stringify(body);
            headers['Content-Type'] = 'application/json';
        }
        const response = await fetch(url, options);
        const data = await response.json();
        return { ok: response.ok, status: response.status, data };
    }
    
    // Récupérer le SHA d'un fichier (s'il existe)
    async function getSha(repo, path, token) {
        const { ok, data } = await githubRequest('GET', repo, path, token);
        if (ok && data.sha) return data.sha;
        return null;
    }
    
    // Lire le contenu d'un fichier
    async function readFileContent(repo, path, token) {
        const { ok, data } = await githubRequest('GET', repo, path, token);
        if (!ok) return null;
        return decodeURIComponent(escape(atob(data.content)));
    }
    
    // Uploader un fichier (crée ou remplace)
    async function uploadFile(repo, path, content, message, token, sha = null) {
        const body = {
            message: message,
            content: btoa(unescape(encodeURIComponent(content))),
            branch: branchInput.value.trim()
        };
        if (sha) body.sha = sha;
        const { ok, data } = await githubRequest('PUT', repo, path, token, body);
        if (!ok) return { success: false, error: data.message };
        return { success: true, data };
    }
    
    // Générer une ligne de date pour un fichier
    function generateDateLine(fileName) {
        const now = new Date();
        const day = now.getDate().toString().padStart(2, '0');
        const month = (now.getMonth() + 1).toString().padStart(2, '0');
        const year = now.getFullYear();
        const hours = now.getHours().toString().padStart(2, '0');
        const minutes = now.getMinutes().toString().padStart(2, '0');
        const seconds = now.getSeconds().toString().padStart(2, '0');
        
        const dateStr = `${day}/${month}/${year}`;
        const timeStr = `${hours}:${minutes}:${seconds}`;
        
        return `DERNIÈRE MISE À JOUR
📅 ${dateStr}
⏰ ${timeStr}
📆 ${dateStr} ${timeStr}
📁 Fichier: ${fileName}`;
    }
    
    // Mettre à jour le fichier date_mise_a_jour.txt (une entrée par fichier)
    async function updateDateFile(repo, fileName, token) {
        try {
            // Lire le contenu actuel du fichier date
            let currentContent = await readFileContent(repo, DATE_FILE_PATH, token);
            let lines = [];
            let found = false;
            let newContent = "";
            
            if (currentContent) {
                // Séparer les blocs (chaque bloc commence par "DERNIÈRE MISE À JOUR")
                const blocks = currentContent.split(/(?=DERNIÈRE MISE À JOUR)/);
                
                for (let block of blocks) {
                    // Vérifier si ce bloc correspond au fichier actuel
                    if (block.includes(`📁 Fichier: ${fileName}`)) {
                        // Remplacer par le nouveau bloc
                        newContent += generateDateLine(fileName) + "\n\n";
                        found = true;
                    } else {
                        newContent += block;
                        if (!block.endsWith("\n\n") && !block.endsWith("\n")) {
                            newContent += "\n";
                        }
                    }
                }
            }
            
            // Si le fichier n'a pas été trouvé, ajouter un nouveau bloc
            if (!found) {
                if (newContent && !newContent.endsWith("\n\n")) {
                    newContent += "\n";
                }
                newContent += generateDateLine(fileName) + "\n\n";
            }
            
            // Nettoyer le contenu
            newContent = newContent.replace(/\n{3,}/g, "\n\n").trim() + "\n";
            
            // Upload du fichier date mis à jour
            const existingSha = await getSha(repo, DATE_FILE_PATH, token);
            const result = await uploadFile(repo, DATE_FILE_PATH, newContent, `Mise à jour date: ${fileName}`, token, existingSha);
            
            return result.success;
        } catch (error) {
            console.error(`❌ Erreur: ${error.message}`);
            return false;
        }
    }
    
    // Upload principal
    btnUpload.addEventListener('click', async () => {
        const token = tokenInput.value.trim();
        const repo = repoInput.value.trim();
        const branch = branchInput.value.trim();
        const commitMsg = commitMsgInput.value.trim();
        const file = fileInput.files[0];
        
        // Vérifications
        if (!token) { showMessage("❌ Entrez votre token GitHub", "error"); return; }
        if (!repo) { showMessage("❌ Entrez le dépôt (ex: user/repo)", "error"); return; }
        if (!repo.includes('/')) { showMessage("❌ Dépôt invalide. Format: owner/repo", "error"); return; }
        if (!branch) { showMessage("❌ Entrez la branche (main ou master)", "error"); return; }
        if (!file) { showMessage("❌ Sélectionnez un fichier", "error"); return; }
        if (!currentFilePath) { showMessage("❌ Erreur: chemin du fichier non détecté", "error"); return; }
        
        btnUpload.disabled = true;
        btnUpload.textContent = "⏳ Upload en cours...";
        urlBox.style.display = 'none';
        
        try {
            const reader = new FileReader();
            reader.onload = async (e) => {
                const content = e.target.result;
                
                // Vérifier si le fichier existe déjà
                const existingSha = await getSha(repo, currentFilePath, token);
                const action = existingSha ? "remplacé" : "créé";
                
                // Upload du fichier principal
                const result = await uploadFile(repo, currentFilePath, content, commitMsg, token, existingSha);
                
                if (result.success) {
                    // 🔥 Mettre à jour le fichier date_mise_a_jour.txt
                    const dateUpdated = await updateDateFile(repo, currentFilePath, token);
                    
                    // URL raw
                    const rawUrl = `https://raw.githubusercontent.com/${repo}/${branch}/${currentFilePath}`;
                    rawUrlSpan.textContent = rawUrl;
                    urlBox.style.display = 'block';
                    
                    if (dateUpdated) {
                        showMessage(`✅ Fichier "${currentFilePath}" ${action} avec succès !\n📅 date_mise_a_jour.txt mis à jour.`, "success");
                    } else {
                        showMessage(`✅ Fichier "${currentFilePath}" ${action} avec succès !\n⚠️ date_mise_a_jour.txt non mis à jour.`, "success");
                    }
                    
                    // Nettoyer
                    fileInput.value = '';
                    fileInfo.style.display = 'none';
                    detectedPathDiv.style.display = 'none';
                    currentFilePath = "";
                } else {
                    showMessage(`❌ Erreur: ${result.error}`, "error");
                }
            };
            reader.readAsText(file);
        } catch (err) {
            showMessage(`❌ Erreur: ${err.message}`, "error");
        } finally {
            btnUpload.disabled = false;
            btnUpload.textContent = "📤 Uploader / Remplacer";
        }
    });
    
    // Copier URL
    btnCopy.addEventListener('click', () => {
        navigator.clipboard.writeText(rawUrlSpan.textContent);
        showMessage("✅ URL copiée !", "success");
    });
</script>
