  // ==================== CONFIGURATION ====================
  const DB_NAME = 'dialunalagasy';
  const STORE_NAME = 'dialgunmalagasy';
  const FILE_KEY = 'diallemasanguny_content';
  const VERSION_KEY = 'dialan_version';
  const FILE_URL = 'https://passage.ahofaly62.workers.dev/diallefrancais.txt';
  const DATE_INFO_URL = 'https://passage.ahofaly62.workers.dev/date_rehetra.txt';
  const TARGET_FILE = 'diallefrancais.txt';
  
  const GARCONS = ['Hery', 'Safidy', 'Andry', 'Avotra', 'Dera'];
  const FILLES = ['Faniry', 'Rojo', 'Rova', 'Nomena', 'Nirina', 'Meva'];
  
  let cachedRubriques = [];
  let cachedRubriquesData = {};
  let cachedDescriptions = {};
  let currentVersionDate = null;
  let isUpdating = false;
  let currentSearch = "";
  let currentUtterance = null;
  
  let rubriquesGrid = null;
  let searchBadgeSpan = null;
  
  function getGenre(nom) {
    if (GARCONS.includes(nom)) return 'garcon';
    if (FILLES.includes(nom)) return 'fille';
    return 'garcon';
  }
  
  function showToast(msg, duration = 2000) {
    const toast = document.getElementById('toast');
    toast.textContent = msg;
    toast.style.opacity = '1';
    setTimeout(() => { toast.style.opacity = '0'; }, duration);
  }
  
  function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/[&<>]/g, m => m === '&' ? '&amp;' : m === '<' ? '&lt;' : '&gt;');
  }
  
  // ==================== INDEXEDDB ====================
  function openDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, 2);
      request.onupgradeneeded = (e) => {
        const db = e.target.result;
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          db.createObjectStore(STORE_NAME);
        }
      };
      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }
  
  async function saveToDB(content, version) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
      const tx = db.transaction(STORE_NAME, 'readwrite');
      const store = tx.objectStore(STORE_NAME);
      store.put(content, FILE_KEY);
      store.put(version, VERSION_KEY);
      tx.oncomplete = () => resolve();
      tx.onerror = () => reject(tx.error);
    });
  }
  
  async function loadFromDB() {
    const db = await openDB();
    return new Promise((resolve) => {
      const tx = db.transaction(STORE_NAME, 'readonly');
      const store = tx.objectStore(STORE_NAME);
      const reqContent = store.get(FILE_KEY);
      const reqVersion = store.get(VERSION_KEY);
      let content = null, version = null;
      let completed = 0;
      function check() {
        completed++;
        if (completed === 2) resolve({ content, version });
      }
      reqContent.onsuccess = () => { content = reqContent.result; check(); };
      reqVersion.onsuccess = () => { version = reqVersion.result; check(); };
      reqContent.onerror = () => check();
      reqVersion.onerror = () => check();
    });
  }
  
  // ==================== VERSION DISTANTE ====================
  async function getRemoteVersionDate() {
    try {
      const res = await fetch(DATE_INFO_URL, { cache: 'no-cache' });
      if (!res.ok) return null;
      const text = await res.text();
      const blocks = text.split(/DERNIÈRE MISE À JOUR/);
      for (const block of blocks) {
        const fileMatch = block.match(/📁 Fichier:\s*([^\n]+)/);
        if (fileMatch && fileMatch[1].trim() === TARGET_FILE) {
          const dayMatch = block.match(/📅\s*([\d\/]+)/);
          const timeMatch = block.match(/⏰\s*([\d:]+)/);
          if (dayMatch && timeMatch) return `${dayMatch[1]} ${timeMatch[1]}`;
        }
      }
      return null;
    } catch (e) { return null; }
  }
  
  async function fetchRemoteTxt() {
    const res = await fetch(FILE_URL, { cache: 'no-store' });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    return await res.text();
  }
  
  // ==================== PARSEUR ====================
  
 

// ==================== PARSEUR AMÉLIORÉ (DÉTECTE TOUS LES NOMS) ====================
function parseCompleteFile(rawText) {
    const lines = rawText.split(/\r?\n/);
    const rubriques = [];
    const rubriquesData = {};
    const rubriquesDescriptions = {};
    
    let currentRubrique = null;
    let currentNom = null;
    let currentMessage = [];
    let i = 0;
    
    // ✅ Fonction pour détecter TOUT nom en début de ligne
    // Un nom est une suite de lettres majuscules/minuscules suivie de ":" ou " :"
    function estNomAuDebut(ligne) {
        const trimmed = ligne.trim();
        // Détecte un mot composé uniquement de lettres (A-Z, a-z, accents) suivi de ":" ou " :"
        const match = trimmed.match(/^([A-Za-zÀ-ÿ]+)\s*[:;]\s*/);
        if (match) {
            return match[1];
        }
        return null;
    }
    
    function extraireMessage(ligne, nom) {
        const trimmed = ligne.trim();
        const pattern = new RegExp('^' + nom + '\\s*[:;]\\s*(.*)$', 'i');
        const match = trimmed.match(pattern);
        return match ? match[1].trim() : null;
    }
    
    function sauvegarderMessage() {
        if (currentRubrique && currentNom && currentMessage.length > 0) {
            const messageTexte = currentMessage.join('\n').trim();
            if (messageTexte) {
                if (!rubriquesData[currentRubrique]) {
                    rubriquesData[currentRubrique] = [];
                }
                rubriquesData[currentRubrique].push({
                    nom: currentNom,
                    message: messageTexte,
                    genre: getGenre(currentNom)
                });
            }
        }
        currentNom = null;
        currentMessage = [];
    }
    
    while (i < lines.length) {
        let line = lines[i];
        let trimmed = line.trim();
        
        // ✅ 1. Détecter TITRE
        const titreMatch = trimmed.match(/^[Tt]itre\s*[:;]\s*(.+)$/);
        if (titreMatch) {
            sauvegarderMessage();
            const nouveauTitre = titreMatch[1].trim();
            currentRubrique = nouveauTitre;
            
            if (!rubriques.includes(currentRubrique)) {
                rubriques.push(currentRubrique);
                rubriquesData[currentRubrique] = [];
                rubriquesDescriptions[currentRubrique] = "";
            }
            
            i++;
            if (i < lines.length) {
                let nextLine = lines[i].trim();
                const descMatch = nextLine.match(/^[Dd]escription\s*[:;]\s*(.+)$/);
                if (descMatch) {
                    rubriquesDescriptions[currentRubrique] = descMatch[1].trim();
                    i++;
                }
            }
            continue;
        }
        
        // ✅ 2. Détecter TOUT NOM en début de ligne
        const nomTrouve = estNomAuDebut(trimmed);
        if (nomTrouve && currentRubrique) {
            sauvegarderMessage();
            const message = extraireMessage(trimmed, nomTrouve);
            currentNom = nomTrouve;
            if (message) {
                currentMessage.push(message);
            } else {
                currentMessage = [];
            }
            i++;
            continue;
        }
        
        // ✅ 3. Sinon, c'est du texte (suite du message)
        if (currentRubrique && currentNom) {
            // Garder la ligne telle quelle (vide ou pas)
            currentMessage.push(line);
        }
        
        i++;
    }
    
    sauvegarderMessage();
    
    const rubriquesValides = rubriques.filter(r => rubriquesData[r] && rubriquesData[r].length > 0);
    const rubriquesDataFiltre = {};
    const rubriquesDescriptionsFiltre = {};
    for (const r of rubriquesValides) {
        rubriquesDataFiltre[r] = rubriquesData[r];
        rubriquesDescriptionsFiltre[r] = rubriquesDescriptions[r] || "";
    }
    
    return {
        rubriques: rubriquesValides,
        rubriquesData: rubriquesDataFiltre,
        rubriquesDescriptions: rubriquesDescriptionsFiltre
    };
}



  // ==================== MISE À JOUR SILENCIEUSE ====================
  async function silentUpdate() {
    if (isUpdating || !navigator.onLine) return false;
    try {
      const remoteVer = await getRemoteVersionDate();
      if (!remoteVer) return false;
      if (currentVersionDate === remoteVer) return false;
      
      isUpdating = true;
      const newContent = await fetchRemoteTxt();
      if (!newContent || newContent.length < 100) throw new Error("Contenu invalide");
      
      await saveToDB(newContent, remoteVer);
      currentVersionDate = remoteVer;
      
      const parsed = parseCompleteFile(newContent);
      cachedRubriques = parsed.rubriques;
      cachedRubriquesData = parsed.rubriquesData;
      cachedDescriptions = parsed.rubriquesDescriptions;
      
      updateListOnly();
      
      isUpdating = false;
      showToast('✅ Dialogues mis à jour');
      return true;
    } catch (e) {
      console.error(e);
      isUpdating = false;
      return false;
    }
  }
  
  // ==================== RECHERCHE ====================
  function fuzzyMatch(text, search) {
    if (!search.trim()) return true;
    const t = text.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    const s = search.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");
    return t.includes(s);
  }
  
  function filterRubriques() {
    if (!currentSearch.trim()) return cachedRubriques;
    const term = currentSearch.trim();
    if (/^\d+$/.test(term)) {
      const num = parseInt(term);
      const found = cachedRubriques.find((_, idx) => idx + 1 === num);
      return found ? [found] : [];
    }
    return cachedRubriques.filter(t => fuzzyMatch(t, term));
  }
  
  // ============ MISE À JOUR UNIQUEMENT DE LA LISTE ============
  function updateListOnly() {
    if (!rubriquesGrid) return;
    const filtered = filterRubriques();
    
    if (searchBadgeSpan) {
      if (currentSearch.trim()) {
        searchBadgeSpan.innerHTML = `🔍 ${filtered.length} résultat(s) pour « ${escapeHtml(currentSearch)} »`;
      } else {
        searchBadgeSpan.innerHTML = ``;
      }

    }

  async function initApp() {
    const app = document.getElementById('app-root');
    app.innerHTML = `<div class="loading-box"><div class="spinner"></div><p>Chargement des dialogues...</p></div>`;
    
    if (window.speechSynthesis) {
      window.speechSynthesis.getVoices();
    }
    
    try {
      const cached = await loadFromDB();
      
      if (cached.content && cached.content.trim().length > 50) {
        currentVersionDate = cached.version;
        const parsed = parseCompleteFile(cached.content);
        cachedRubriques = parsed.rubriques;
        cachedRubriquesData = parsed.rubriquesData;
        cachedDescriptions = parsed.rubriquesDescriptions;
        
        if (cachedRubriques.length > 0) {
          renderList();
        } else {
          throw new Error("Fichier corrompu");
        }
      } else {
        if (!navigator.onLine) throw new Error("Pas de connexion internet");
        
        const fileContent = await fetchRemoteTxt();
        const remoteVersion = await getRemoteVersionDate();
        const finalVersion = remoteVersion || new Date().toLocaleString('fr-FR');
        
        await saveToDB(fileContent, finalVersion);
        currentVersionDate = finalVersion;
        
        const parsed = parseCompleteFile(fileContent);
        cachedRubriques = parsed.rubriques;
        cachedRubriquesData = parsed.rubriquesData;
        cachedDescriptions = parsed.rubriquesDescriptions;
        
        if (cachedRubriques.length === 0) {
          const fallback = getFallbackData();
          cachedRubriques = fallback.rubriques;
          cachedRubriquesData = fallback.rubriquesData;
          cachedDescriptions = fallback.rubriquesDescriptions;
        }
        
        renderList();
      }
      
      if (navigator.onLine) setTimeout(() => silentUpdate(), 2000);
      window.addEventListener('online', () => setTimeout(() => silentUpdate(), 1000));
      
    } catch (err) {
      console.error(err);
      const fallback = getFallbackData();
      cachedRubriques = fallback.rubriques;
      cachedRubriquesData = fallback.rubriquesData;
      cachedDescriptions = fallback.rubriquesDescriptions;
      currentVersionDate = "2025-01-01";
      
      if (cachedRubriques.length > 0) {
        renderList();
      } else {
        app.innerHTML = `<div class="loading-box" style="color:#b33;">
          <span>❌ Erreur: ${err.message}</span><br/><br/>
          <button onclick="location.reload()" style="background:#c56985; border:none; padding:10px 20px; border-radius:40px; color:white; cursor:pointer;">🔄 Réessayer</button>
        </div>`;
      }
    }
  }
